\newcommand{\TNBlockDiagonalGauge}[1]{%
  \begin{array}{ccccccc}
    \fbox{$\mu_0 A_0$} & \fbox{$\mu_1 A_1$} & \cdots & \fbox{$\mu_{r-1} A_{r-1}$}
    & \xrightarrow{\ \operatorname{diag}(#1_k)\ } &
    \fbox{$\mu_0 B_0$} & \fbox{$\mu_{r-1} B_{r-1}$}
  \end{array}%
}

\newcommand{\TNLeadingBlockPeeling}[2]{%
  \begin{array}{ccccc}
    \fbox{$#1_0^N V(A_0)$} & + & \fbox{$\sum_{k\ge 1} #1_k^N V(A_k)$}
    & = &
    \fbox{$#1_0^N V(B_0)$} + \fbox{$\sum_{k\ge 1} #1_k^N V(B_k)$} \\
    && O(#2^N) && \\
    && \Downarrow && \\
    && V(A_0)=V(B_0) &&
  \end{array}%
}

\newcommand{\TNSummedBlocks}[2]{%
  \begin{array}{ccccc}
    \fbox{$\sum_k #1_k^{#2} V(A_k)$} & = & \fbox{$\sum_k #1_k^{#2} V(B_k)$} \\
    \fbox{$#1_0^{#2} V(A_0)$} & + & \fbox{$\sum_{k\ge1} #1_k^{#2} V(A_k)$} &
    = &
    \fbox{$#1_0^{#2} V(B_0)$} + \fbox{$\sum_{k\ge1} #1_k^{#2} V(B_k)$}
  \end{array}%
}
