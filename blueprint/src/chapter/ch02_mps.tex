\chapter{Matrix Product States}\label{ch:mps}

Fix a physical dimension $d \ge 1$ and a bond dimension $D \ge 1$.
All MPS in this blueprint are translation-invariant with periodic
boundary conditions (PBC).

\section{Basic definitions}

\begin{definition}[MPS tensor]\label{def:mps_tensor}
    \lean{MPSTensor}
    \leanok
    A (translation-invariant, PBC) \emph{MPS tensor} with physical
    dimension~$d$ and bond dimension~$D$
    is a collection of matrices
    \[
        \{A^i\}_{i=0}^{d-1}, \qquad A^i \in \MN{D},
    \]
    indexed by a physical index $i \in \{0, \ldots, d{-}1\}$.
\end{definition}

\begin{definition}[Word evaluation]\label{def:eval_word}
    \lean{MPSTensor.evalWord}
    \leanok
    \uses{def:mps_tensor}
    Given a word $w = (i_1, \ldots, i_L) \in \{0,\ldots,d{-}1\}^L$,
    the \emph{word evaluation} is the matrix product
    \[
        A^w \;:=\; A^{i_1} A^{i_2} \cdots A^{i_L} \;\in\; \MN{D},
    \]
    with the convention that the empty word evaluates to the identity:
    $A^\varnothing = \Id_D$.
\end{definition}

\begin{lemma}[Word evaluation respects concatenation]\label{lem:eval_word_append}
    \lean{MPSTensor.evalWord_append}
    \leanok
    \uses{def:eval_word}
    For any words $w_1, w_2$,
    $A^{w_1 \cdot w_2} = A^{w_1} \, A^{w_2}$.
\end{lemma}

\begin{proof}\leanok\uses{def:eval_word}
    Induction on $w_1$.
\end{proof}

\begin{definition}[Trace coefficient]\label{def:coeff}
    \lean{MPSTensor.coeff}
    \leanok
    \uses{def:eval_word}
    The \emph{trace coefficient} of a word $w$ is
    $c_w(A) := \tr(A^w)$.
\end{definition}

\begin{definition}[Matrix product vector]\label{def:mpv}
    \lean{MPSTensor.mpv}
    \leanok
    \uses{def:coeff}
    The \emph{matrix product vector} (MPV) of an MPS tensor~$A$
    at system size~$N$ is the vector
    \[
        \ket{V^{(N)}(A)}
        = \sum_{i_1, \ldots, i_N = 0}^{d-1}
        \tr\!(A^{i_1} A^{i_2} \cdots A^{i_N})
        \ket{i_1, \ldots, i_N}
        \in  (\C^d)^{\otimes N}.
    \]
    The \emph{MPV family} generated by $A$ is the collection
    $\mc{V}(A) = \bigl\{\ket{V^{(N)}(A)}\bigr\}_{N \ge 1}$.
\end{definition}

\begin{definition}[Transfer map]\label{def:transfer_map}
    \lean{MPSTensor.transferMap}
    \leanok
    \uses{def:mps_tensor}
    The \emph{transfer map} associated to an MPS tensor~$A$ is the
    linear map $\E_A : \MN{D} \to \MN{D}$ defined by
    \[
        \E_A(X) \;=\; \sum_{i=0}^{d-1} A^i \, X \, (A^i)^\dagger.
    \]
\end{definition}

\begin{lemma}[Transfer map is positive]\label{lem:transfer_pos}
    \lean{MPSTensor.transferMap_pos}
    \leanok
    \uses{def:transfer_map}
    If $X \ge 0$, then $\E_A(X) \ge 0$.
\end{lemma}

\begin{proof}\leanok\uses{def:transfer_map}
    Each summand $A^i X (A^i)^\dagger$ is PSD when $X$ is PSD.
\end{proof}

\section{Gauge equivalence and same MPV}

\begin{definition}[Gauge equivalence]\label{def:gauge_equiv}
    \lean{MPSTensor.GaugeEquiv}
    \leanok
    \uses{def:mps_tensor}
    Two MPS tensors $A$ and $B$ of the same bond dimension~$D$ are
    \emph{gauge equivalent} if there exists an invertible matrix
    $X \in \GL_D(\C)$ such that
    \[
        B^i \;=\; X \, A^i \, X^{-1}
        \qquad \text{for all } i \in \{0,\ldots,d{-}1\}.
    \]
\end{definition}

\begin{definition}[Same MPV]\label{def:same_mpv}
    \lean{MPSTensor.SameMPV}
    \leanok
    \uses{def:mpv}
    Two MPS tensors $A$ and $B$ (of the same bond dimension)
    \emph{generate the same MPV family}
    if $\ket{V^{(N)}(A)} = \ket{V^{(N)}(B)}$ for every system size~$N \ge 1$.
\end{definition}

\begin{definition}[Same MPV — different bond dimensions]\label{def:same_mpv2}
    \lean{MPSTensor.SameMPV₂}
    \leanok
    \uses{def:mpv}
    For MPS tensors $A$ and $B$ of possibly different bond
    dimensions $D_1$ and $D_2$,
    we write $\mc{V}(A) = \mc{V}(B)$ if
    $\ket{V^{(N)}(A)} = \ket{V^{(N)}(B)}$ for every $N \ge 1$.
\end{definition}

\begin{definition}[Proportional MPV]\label{def:proportional_mpv}
    \lean{MPSTensor.ProportionalMPV₂}
    \leanok
    \uses{def:mpv}
    Two MPS tensors $A$ and $B$ (of possibly different bond dimensions)
    generate \emph{proportional MPV families} if for every~$N$ there
    exists $c_N \in \C$ such that
    $\ket{V^{(N)}(A)} = c_N \, \ket{V^{(N)}(B)}$.
\end{definition}

\begin{lemma}[Word evaluation under scaling]\label{lem:eval_word_smul}
    \lean{MPSTensor.evalWord_smul}
    \leanok
    \uses{def:eval_word}
    Let $\zeta A$ denote the tensor with matrices
    $\{\zeta \, A^i\}_{i=0}^{d-1}$.
    Then for any word $w$,
    \[
        (\zeta A)^w \;=\; \zeta^{|w|} \, A^w,
    \]
    where $|w|$ is the length of~$w$.
\end{lemma}

\begin{proof}\leanok\uses{def:eval_word}
    Induction on $w$: the base case gives $\zeta^0 = 1$,
    and each step picks up one factor of $\zeta$.
\end{proof}

\begin{definition}[Gauge-phase equivalence]\label{def:gauge_phase_equiv}
    \lean{MPSTensor.GaugePhaseEquiv}
    \leanok
    \uses{def:mps_tensor}
    Two MPS tensors $A$ and $B$ are \emph{gauge-phase equivalent}
    if there exist $X \in \GL_D(\C)$ and $\zeta \in \C \setminus \{0\}$
    such that
    \[
        B^i \;=\; \zeta \, X \, A^i \, X^{-1}
        \qquad \text{for all } i.
    \]
    The requirement $\zeta \neq 0$ ensures that gauge-phase equivalence
    preserves injectivity.
\end{definition}

\begin{lemma}[Word evaluation under conjugation]\label{lem:eval_word_gauge}
    \lean{MPSTensor.evalWord_gauge}
    \leanok
    \uses{def:eval_word}
    If $B^i = X \, A^i \, X^{-1}$ for all~$i$, then for every word~$w$,
    \[
        B^w \;=\; X \, A^w \, X^{-1}.
    \]
\end{lemma}

\begin{proof}\leanok\uses{def:eval_word}
    Induction on the word length, using $X X^{-1} = \Id$.
\end{proof}

\begin{theorem}[Gauge invariance of MPVs]\label{thm:gauge_invariance}
    \lean{MPSTensor.GaugeEquiv.sameMPV}
    \leanok
    \uses{def:gauge_equiv, def:same_mpv}
    If $A$ and $B$ are gauge equivalent, then they generate the same
    MPV family.
\end{theorem}

\begin{proof}
    \leanok
    \uses{lem:eval_word_gauge}
    Let $B^i = X A^i X^{-1}$.
    By Lemma~\ref{lem:eval_word_gauge}, $B^w = X \, A^w \, X^{-1}$
    for every word~$w$.
    Then $\tr(B^w) = \tr(X \, A^w \, X^{-1}) = \tr(A^w)$
    by cyclicity of the trace.
\end{proof}

\section{Injectivity and normality}

\begin{definition}[Injectivity]\label{def:injective}
    \lean{MPSTensor.IsInjective}
    \leanok
    \uses{def:mps_tensor}
    An MPS tensor~$A$ is \emph{injective} if its matrices span the
    full matrix algebra:
    $\spn_{\C}\{A^i : i = 0,\ldots,d{-}1\} = \MN{D}$.
\end{definition}

\begin{definition}[$L$-block injectivity]\label{def:l_blk_injective}
    \lean{MPSTensor.IsNBlkInjective}
    \leanok
    \uses{def:mps_tensor}
    An MPS tensor~$A$ is \emph{$L$-block injective} if
    $\spn_{\C}\{A^{i_1} \cdots A^{i_L}
      : (i_1,\ldots,i_L) \in \{0,\ldots,d{-}1\}^L\} = \MN{D}$.
    An injective tensor is $1$-block injective.
\end{definition}

\begin{definition}[Normal tensor]\label{def:normal}
    \lean{MPSTensor.IsNormal}
    \leanok
    \uses{def:l_blk_injective}
    An MPS tensor~$A$ is \emph{normal}
    \cite{Cirac2021Matrix}
    if there exists a finite $L_0$ such that $A$ is $L_0$-block
    injective.
\end{definition}

\section{Canonical form}

\begin{definition}[Canonical form]\label{def:canonical_form}
    \lean{MPSTensor.CanonicalForm}
    \leanok
    \uses{def:mps_tensor, def:injective}
    A \emph{canonical form} for an MPS consists of:
    \begin{enumerate}
        \item a number of blocks $r \ge 1$,
        \item bond dimensions $D_1, \ldots, D_r$,
        \item injective tensors
              $\{A_k^i\}_{i=0}^{d-1}$ with $A_k^i \in \MN{D_k}$
              for each block $k \in \{1,\ldots,r\}$,
        \item scaling factors $\mu_1, \ldots, \mu_r \in \C$.
    \end{enumerate}
    The associated MPS tensor is the block-diagonal tensor
    \[
        A^i \;=\; \bigoplus_{k=1}^{r} \mu_k \, A_k^i.
    \]
    This decomposition is defined in \cite{PerezGarcia2007Matrix}
    and reviewed in~\cite{Cirac2021Matrix}.
\end{definition}

\begin{remark}\label{rem:cf_injective_vs_normal}
    The formalization requires each block tensor to be injective
    (Definition~\ref{def:injective}).
    More generally, canonical forms in the literature
    \cite{Cirac2021Matrix} allow normal tensors
    (Definition~\ref{def:normal}), which become injective after
    sufficient blocking (Definition~\ref{def:blocked_tensor}).
\end{remark}

\begin{definition}[Block-diagonal MPS tensor]\label{def:block_diagonal_tensor}
    \lean{MPSTensor.toTensorFromBlocks}
    \leanok
    \uses{def:mps_tensor}
    Given $r$ blocks with bond dimensions $(D_k)_{k=1}^r$,
    per-block MPS tensors $\{A_k^i\}$,
    and scaling factors $(\mu_k)_{k=1}^r$,
    the \emph{block-diagonal MPS tensor} is the tensor of
    total bond dimension $D = \sum_{k=1}^r D_k$ defined by
    \[
        A^i \;=\; \bigoplus_{k=1}^{r} \mu_k \, A_k^i.
    \]
\end{definition}

\begin{theorem}[MPV decomposition]\label{thm:mpv_decomposition}
    \lean{MPSTensor.mpv_toTensorFromBlocks_eq_sum}
    \leanok
    \uses{def:block_diagonal_tensor, def:mpv}
    The MPV of a block-diagonal tensor decomposes as
    \[
        \ket{V^{(N)}(A)}
        \;=\; \sum_{k=1}^{r} \mu_k^N \, \ket{V^{(N)}(A_k)}.
    \]
\end{theorem}

\begin{proof}
    \leanok
    \uses{def:block_diagonal_tensor, def:coeff, lem:eval_word_smul}
    For each physical index~$i$, the block-diagonal tensor
    $A^i = \bigoplus_k \mu_k A_k^i$ is a block-diagonal matrix.
    By Lemma~\ref{lem:eval_word_smul},
    the word evaluation of each block picks up a factor
    $\mu_k^{|w|}$, and the full word evaluation is block-diagonal
    with blocks $\mu_k^N A_k^w$.
    Taking the trace of a block-diagonal matrix gives the sum
    of the block traces:
    $\tr(A^w) = \sum_k \mu_k^N \tr(A_k^w)$.
\end{proof}

\section{Blocking}

\begin{definition}[Blocked tensor]\label{def:blocked_tensor}
    \lean{MPSTensor.blockTensor}
    \leanok
    \uses{def:mps_tensor}
    Given an MPS tensor~$A$ and a blocking length $L \ge 1$,
    the \emph{$L$-blocked tensor} is the MPS tensor with physical
    dimension $d^L$ and the same bond dimension~$D$, whose matrices
    are indexed by words $(i_1, \ldots, i_L) \in \{0,\ldots,d{-}1\}^L$:
    \[
        (A^{[L]})^{(i_1, \ldots, i_L)}
        \;=\; A^{i_1} A^{i_2} \cdots A^{i_L}.
    \]
\end{definition}

\begin{lemma}[Blocked word evaluation]\label{lem:eval_word_block}
    \lean{MPSTensor.evalWord_blockTensor}
    \leanok
    \uses{def:blocked_tensor, def:eval_word}
    Let $w = (J_1, \ldots, J_m)$ be a word in the blocked alphabet
    (each $J_k \in \{0, \ldots, d^L{-}1\}$).
    Decoding each blocked index $J_k$ back to an $L$-tuple of
    physical indices and concatenating gives a word
    $\widetilde{w}$ of length $mL$ in the original alphabet.
    Then $(A^{[L]})^w = A^{\widetilde{w}}$.
\end{lemma}

\begin{proof}\leanok\uses{def:blocked_tensor, def:eval_word}
    Induction on $w$, using the concatenation lemma
    (Lemma~\ref{lem:eval_word_append}).
\end{proof}

\begin{lemma}[Blocking preserves MPV coefficients]\label{lem:mpv_block}
    \lean{MPSTensor.mpv_blockTensor_eq_mpv}
    \leanok
    \uses{def:blocked_tensor, def:mpv}
    For each basis state $\sigma = (\sigma_1, \ldots, \sigma_N)$ of
    the blocked system (with $\sigma_k \in \{0, \ldots, d^L{-}1\}$),
    there exists a basis state
    $\widetilde{\sigma} = (\widetilde{\sigma}_1, \ldots, \widetilde{\sigma}_{NL})$
    of the original system such that
    \[
        V^{(N)}(A^{[L]})_\sigma
        \;=\;
        V^{(NL)}(A)_{\widetilde{\sigma}}.
    \]
\end{lemma}

\begin{proof}\leanok\uses{def:blocked_tensor, def:mpv, lem:eval_word_block}
    Flatten the blocked basis state and apply the blocked word
    evaluation lemma (Lemma~\ref{lem:eval_word_block}).
\end{proof}

\begin{theorem}[Blocking preserves same MPV]\label{thm:blocking_same_mpv}
    \lean{MPSTensor.SameMPV.blockTensor}
    \leanok
    \uses{def:same_mpv, def:blocked_tensor}
    If $A$ and $B$ generate the same MPV family, then for any
    blocking length~$L$, the blocked tensors $A^{[L]}$ and $B^{[L]}$
    also generate the same MPV family.
\end{theorem}

\begin{proof}
    \leanok
    \uses{lem:mpv_block}
    By Lemma~\ref{lem:mpv_block},
    each MPV coefficient of the blocked tensor $A^{[L]}$ equals
    a coefficient of $A$ at the flattened configuration.
    The same holds for $B^{[L]}$ (with the same flattened
    configuration).
    Since $A$ and $B$ agree on all configurations, so do
    $A^{[L]}$ and $B^{[L]}$.
\end{proof}

\section{MPV overlap}

\begin{definition}[MPV as Hilbert-space vector]\label{def:mpv_state}
    \lean{MPSTensor.mpvState}
    \leanok
    \uses{def:mpv}
    We regard $\ket{V^{(N)}(A)}$ as an element of $\C^{d^N}$
    indexed by configurations
    $\sigma \in \{0,\ldots,d{-}1\}^N$,
    with $\sigma$-component $V^{(N)}(A)_\sigma$ as in
    Definition~\ref{def:mpv}.
    This allows us to use the standard Euclidean inner product
    on $\C^{d^N}$.
\end{definition}

\begin{definition}[MPV inner product]\label{def:mpv_inner}
    \lean{MPSTensor.mpvInner}
    \leanok
    \uses{def:mpv_state}
    The \emph{MPV inner product} between two MPS tensors $A$ and $B$
    at system size~$N$ is
    \[
        \braket{V^{(N)}(A)}{V^{(N)}(B)}
        \;:=\;
        \sum_{\sigma \in \{0,\ldots,d{-}1\}^N}
        \overline{V^{(N)}(A)_\sigma} \; V^{(N)}(B)_\sigma,
    \]
    i.e.\ the Euclidean inner product on $\C^{d^N}$
    (conjugate-linear in the first argument).
\end{definition}

\begin{definition}[MPV overlap]\label{def:mpv_overlap}
    \lean{MPSTensor.mpvOverlap}
    \leanok
    \uses{def:mpv}
    The \emph{MPV overlap} between two MPS tensors $A$ (of bond
    dimension $D_1$) and $B$ (of bond dimension $D_2$)
    at system size~$N$ is
    \[
        O_{AB}(N)
        \;:=\;
        \sum_{\sigma \in \{0,\ldots,d{-}1\}^N}
        V^{(N)}(A)_\sigma \; \overline{V^{(N)}(B)_\sigma}.
    \]
\end{definition}

\begin{lemma}[Overlap equals conjugate inner product]\label{lem:overlap_eq_inner}
    \lean{MPSTensor.mpvOverlap_eq_star_mpvInner}
    \leanok
    \uses{def:mpv_overlap, def:mpv_inner}
    $O_{AB}(N) = \overline{\braket{V^{(N)}(A)}{V^{(N)}(B)}}$.
\end{lemma}

\begin{proof}\leanok\uses{def:mpv_overlap, def:mpv_inner}
    The overlap sums $V_\sigma \, \overline{W_\sigma}$
    while the inner product sums $\overline{V_\sigma} \, W_\sigma$.
    Complex conjugation swaps these.
\end{proof}
