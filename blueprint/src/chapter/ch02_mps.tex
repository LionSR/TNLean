\chapter{Matrix Product States}\label{ch:mps}

Fix a physical dimension $d \ge 1$ and a bond dimension $D \ge 1$.
All MPS in this blueprint are translation-invariant with periodic
boundary conditions (PBC).

\section{Basic definitions}

\begin{definition}[MPS tensor]\label{def:mps_tensor}
    \lean{MPSTensor}
    \leanok
    A (translation-invariant, PBC) \emph{MPS tensor} with physical
    dimension~$d$ and bond dimension~$D$
    is a collection of matrices
    \[
        \{A^i\}_{i=0}^{d-1}, \qquad A^i \in \MN{D},
    \]
    indexed by a physical index $i \in \{0, \ldots, d{-}1\}$.
\end{definition}

\begin{definition}[Matrix product vector]\label{def:mpv}
    \lean{MPSTensor.mpv}
    \leanok
    \uses{def:mps_tensor}
    The \emph{matrix product vector} (MPV) of an MPS tensor~$A$
    at system size~$N$ is the vector
    \[
        \ket{V^{(N)}(A)}
        = \sum_{i_1, \ldots, i_N = 0}^{d-1}
        \tr\!(A^{i_1} A^{i_2} \cdots A^{i_N})
        \ket{i_1, \ldots, i_N}
        \in  (\C^d)^{\otimes N}.
    \]
    The \emph{MPV family} generated by $A$ is the collection
    $\mc{V}(A) = \bigl\{\ket{V^{(N)}(A)}\bigr\}_{N \ge 1}$.
\end{definition}

\begin{definition}[Transfer map]\label{def:transfer_map}
    \lean{MPSTensor.transferMap}
    \leanok
    \uses{def:mps_tensor}
    The \emph{transfer map} associated to an MPS tensor~$A$ is the
    linear map $\E_A : \MN{D} \to \MN{D}$ defined by
    \[
        \E_A(X) \;=\; \sum_{i=0}^{d-1} A^i \, X \, (A^i)^\dagger.
    \]
\end{definition}

\section{Gauge equivalence and same MPV}

\begin{definition}[Gauge equivalence]\label{def:gauge_equiv}
    \lean{MPSTensor.GaugeEquiv}
    \leanok
    \uses{def:mps_tensor}
    Two MPS tensors $A$ and $B$ of the same bond dimension~$D$ are
    \emph{gauge equivalent} if there exists an invertible matrix
    $X \in \GL_D(\C)$ such that
    \[
        B^i \;=\; X \, A^i \, X^{-1}
        \qquad \text{for all } i \in \{0,\ldots,d{-}1\}.
    \]
\end{definition}

\begin{definition}[Same MPV]\label{def:same_mpv}
    \lean{MPSTensor.SameMPV}
    \leanok
    \uses{def:mpv}
    Two MPS tensors $A$ and $B$ (of the same bond dimension)
    \emph{generate the same MPV family}
    if $\ket{V^{(N)}(A)} = \ket{V^{(N)}(B)}$ for every system size~$N \ge 1$.
\end{definition}

\begin{definition}[Same MPV — different bond dimensions]\label{def:same_mpv2}
    \lean{MPSTensor.SameMPV₂}
    \leanok
    \uses{def:mpv}
    For MPS tensors $A$ and $B$ of possibly different bond
    dimensions $D_1$ and $D_2$,
    we write $\mc{V}(A) = \mc{V}(B)$ if
    $\ket{V^{(N)}(A)} = \ket{V^{(N)}(B)}$ for every $N \ge 1$.
\end{definition}

\begin{definition}[Proportional MPV]\label{def:proportional_mpv}
    \lean{MPSTensor.ProportionalMPV₂}
    \leanok
    \uses{def:mpv}
    Two MPS tensors $A$ and $B$ (of possibly different bond dimensions)
    generate \emph{proportional MPV families} if for every~$N$ there
    exists $c_N \in \C$ such that
    $\ket{V^{(N)}(A)} = c_N \, \ket{V^{(N)}(B)}$.
\end{definition}

\begin{definition}[Gauge-phase equivalence]\label{def:gauge_phase_equiv}
    \lean{MPSTensor.GaugePhaseEquiv}
    \leanok
    \uses{def:mps_tensor}
    Two MPS tensors $A$ and $B$ are \emph{gauge-phase equivalent}
    if there exist $X \in \GL_D(\C)$ and $\zeta \in \C$ such that
    \[
        B^i \;=\; \zeta \, X \, A^i \, X^{-1}
        \qquad \text{for all } i.
    \]
\end{definition}

\begin{theorem}[Gauge invariance of MPVs]\label{thm:gauge_invariance}
    \lean{MPSTensor.GaugeEquiv.sameMPV}
    \leanok
    \uses{def:gauge_equiv, def:same_mpv}
    If $A$ and $B$ are gauge equivalent, then they generate the same
    MPV family.
\end{theorem}

\begin{proof}
    \leanok
    \uses{def:gauge_equiv, def:mpv}
    Let $B^i = X A^i X^{-1}$.
    By induction on word length,
    $B^{i_1} \cdots B^{i_N} = X \, A^{i_1} \cdots A^{i_N} \, X^{-1}$.
    Then $\tr(B^{i_1} \cdots B^{i_N}) = \tr(A^{i_1} \cdots A^{i_N})$
    by cyclicity of the trace.
\end{proof}

\section{Injectivity and normality}

\begin{definition}[Injectivity]\label{def:injective}
    \lean{MPSTensor.IsInjective}
    \leanok
    \uses{def:mps_tensor}
    An MPS tensor~$A$ is \emph{injective} if its matrices span the
    full matrix algebra:
    $\spn_{\C}\{A^i : i = 0,\ldots,d{-}1\} = \MN{D}$.
\end{definition}

\begin{definition}[$L$-block injectivity]\label{def:l_blk_injective}
    \lean{MPSTensor.IsNBlkInjective}
    \leanok
    \uses{def:mps_tensor}
    An MPS tensor~$A$ is \emph{$L$-block injective} if
    $\spn_{\C}\{A^{i_1} \cdots A^{i_L}
      : (i_1,\ldots,i_L) \in \{0,\ldots,d{-}1\}^L\} = \MN{D}$.
    An injective tensor is $1$-block injective.
\end{definition}

\begin{definition}[Normal tensor]\label{def:normal}
    \lean{MPSTensor.IsNormal}
    \leanok
    \uses{def:l_blk_injective}
    An MPS tensor~$A$ is \emph{normal}
    \cite{Cirac2021Matrix}
    if there exists a finite $L_0$ such that $A$ is $L_0$-block
    injective.
\end{definition}

\section{Canonical form}

\begin{definition}[Canonical form]\label{def:canonical_form}
    \lean{MPSTensor.CanonicalForm}
    \leanok
    \uses{def:mps_tensor, def:injective}
    A \emph{canonical form} for an MPS consists of:
    \begin{enumerate}
        \item a number of blocks $r \ge 1$,
        \item bond dimensions $D_1, \ldots, D_r$,
        \item injective (normal) tensors
              $\{A_k^i\}_{i=0}^{d-1}$ with $A_k^i \in \MN{D_k}$
              for each block $k \in \{1,\ldots,r\}$,
        \item scaling factors $\mu_1, \ldots, \mu_r \in \C$.
    \end{enumerate}
    The associated MPS tensor is the block-diagonal tensor
    \[
        A^i \;=\; \bigoplus_{k=1}^{r} \mu_k \, A_k^i.
    \]
    This decomposition is defined in \cite{PerezGarcia2007Matrix}
    and reviewed in~\cite{Cirac2021Matrix}.
\end{definition}

\begin{definition}[Block-diagonal MPS tensor]\label{def:block_diagonal_tensor}
    \lean{MPSTensor.toTensorFromBlocks}
    \leanok
    \uses{def:mps_tensor}
    Given $r$ blocks with bond dimensions $(D_k)_{k=1}^r$,
    per-block MPS tensors $\{A_k^i\}$,
    and scaling factors $(\mu_k)_{k=1}^r$,
    the \emph{block-diagonal MPS tensor} is the tensor of
    total bond dimension $D = \sum_{k=1}^r D_k$ defined by
    \[
        A^i \;=\; \bigoplus_{k=1}^{r} \mu_k \, A_k^i.
    \]
\end{definition}

\begin{theorem}[MPV decomposition]\label{thm:mpv_decomposition}
    \lean{MPSTensor.mpv_toTensorFromBlocks_eq_sum}
    \leanok
    \uses{def:block_diagonal_tensor, def:mpv}
    The MPV of a block-diagonal tensor decomposes as
    \[
        \ket{V^{(N)}(A)}
        \;=\; \sum_{k=1}^{r} \mu_k^N \, \ket{V^{(N)}(A_k)}.
    \]
\end{theorem}

\begin{proof}
    \leanok
    \uses{def:block_diagonal_tensor, def:mpv}
    The trace of a block-diagonal matrix is the sum of the
    block traces, and the word evaluation of a block-diagonal
    tensor is block-diagonal with per-block scaling
    $\mu_k^N$.
\end{proof}

\section{Blocking}

\begin{definition}[Blocked tensor]\label{def:blocked_tensor}
    \lean{MPSTensor.blockTensor}
    \leanok
    \uses{def:mps_tensor}
    Given an MPS tensor~$A$ and a blocking length $L \ge 1$,
    the \emph{$L$-blocked tensor} is the MPS tensor with physical
    dimension $d^L$ and the same bond dimension~$D$, whose matrices
    are indexed by words $(i_1, \ldots, i_L) \in \{0,\ldots,d{-}1\}^L$:
    \[
        (A^{[L]})^{(i_1, \ldots, i_L)}
        \;=\; A^{i_1} A^{i_2} \cdots A^{i_L}.
    \]
\end{definition}

\begin{theorem}[Blocking preserves same MPV]\label{thm:blocking_same_mpv}
    \lean{MPSTensor.SameMPV.blockTensor}
    \leanok
    \uses{def:same_mpv, def:blocked_tensor}
    If $A$ and $B$ generate the same MPV family, then for any
    blocking length~$L$, the blocked tensors $A^{[L]}$ and $B^{[L]}$
    also generate the same MPV family.
\end{theorem}

\begin{proof}
    \leanok
    \uses{def:same_mpv, def:blocked_tensor}
    Every word in the blocked indices flattens to an ordinary
    word in the original indices, and the hypothesis gives
    equality of traces for all ordinary words.
\end{proof}

\section{MPV overlap}

\begin{definition}[MPV overlap]\label{def:mpv_overlap}
    \lean{MPSTensor.mpvOverlap}
    \leanok
    \uses{def:mpv}
    The \emph{MPV overlap} between two MPS tensors $A$ (of bond
    dimension $D_1$) and $B$ (of bond dimension $D_2$)
    at system size~$N$ is
    \[
        \langle A | B \rangle_N
        \;=\;
        \sum_{\sigma \in \{0,\ldots,d{-}1\}^N}
        V^{(N)}(A)_\sigma \; \overline{V^{(N)}(B)_\sigma}
        \;=\;
        \braket{V^{(N)}(A)}{V^{(N)}(B)}.
    \]
\end{definition}
