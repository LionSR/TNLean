\chapter{The Algebraic Fundamental Theorem}%
\label{ch:algebraic_ft}

This chapter formalises the one-dimensional MPS results of
\cite{Molnar2018NormalPEPS}: the Fundamental Theorem for injective
and normal MPS on a finite periodic chain of fixed length.
The approach is purely algebraic---it never appeals to operator-norm
limits or spectral theory of transfer maps, working instead with the
matrix algebra $\MN{D}$ directly.
Two preparatory lemmas (the virtual bond gauge and the proportionality
lemma) carry the entire argument; their combination yields the main
theorem and its corollaries for translation-invariant chains, non-TI
descriptions of TI states, physical symmetries, and normal MPS.

Throughout we fix a constant bond dimension $D$ and physical
dimension $d$ with $d \ge D^2$.
The paper \cite{Molnar2018NormalPEPS} also treats injective PEPS on
arbitrary graphs, normal PEPS, and an improved system-size bound
($2L+1$ in place of $3L$); those extensions are left for future work.

The chain bookkeeping layer (Definition~\ref{def:non_ti_chain}),
including the notions of same chain state, injectivity, and cyclic gauge
equivalence, was introduced in Chapter~\ref{ch:single}.

%% ---------------------------------------------------------
\section{One-sided inverse and decomposition}%
\label{sec:decomposition_map}
%% ---------------------------------------------------------

Let $A = \{A^i\}_{i=0}^{d-1}$ be an injective MPS tensor with bond
dimension $D$.
Recall from Definition~\ref{def:injective} that injectivity means that
the matrices $\{A^i\}$ span $\MN{D}$.
The \emph{linear combination map}
\[
    \Phi_A : \C^d \to \MN{D},
    \qquad
    \Phi_A(c) := \sum_{i=0}^{d-1} c_i A^i,
\]
is therefore surjective, and a right inverse exists.

\begin{lemma}[Existence of a right inverse]\label{lem:exists_rightInverse}
    \lean{MPSTensor.IsInjective.exists_rightInverse}
    \leanok
    \uses{def:injective}
    If $A$ is injective, then there exists a $\C$-linear map
    $\Psi : \MN{D} \to \C^d$ satisfying
    $\Phi_A \circ \Psi = \mathrm{id}_{\MN{D}}$.
\end{lemma}

\begin{proof}
    \leanok
    Surjectivity of $\Phi_A$ yields a linear section by choice.
\end{proof}

\begin{definition}[Decomposition map]\label{def:decomposition_map}
    \lean{MPSTensor.decompositionMap}
    \leanok
    \uses{def:injective, lem:exists_rightInverse}
    Let $A$ be an injective MPS tensor.
    The \emph{decomposition map} $\Psi_A : \MN{D} \to \C^d$ is a
    fixed choice of right inverse of $\Phi_A$.
\end{definition}

\begin{lemma}[Decomposition property]\label{lem:exists_decomposition}
    \lean{MPSTensor.IsInjective.exists_decomposition}
    \leanok
    \uses{def:decomposition_map}
    If $A$ is injective, then for every $X \in \MN{D}$ there exist
    coefficients $c \in \C^d$ such that
    \[
        X = \sum_{i=0}^{d-1} c_i\, A^i.
    \]
\end{lemma}

\begin{proof}
    \leanok
    Take $c = \Psi_A(X)$.
    Then $\Phi_A(c) = \Phi_A(\Psi_A(X)) = X$.
\end{proof}

\begin{remark}[Non-uniqueness of the decomposition]
    \label{rem:decomposition_nonunique}
    When $d > D^2$ the map $\Phi_A$ has a nontrivial kernel and the
    decomposition is not unique.
    The decomposition map $\Psi_A$ is any fixed section; its choice
    does not affect the statements that follow because those statements
    only depend on the image $\Phi_A \circ \Psi_A = \mathrm{id}$.
    Uniqueness of the decomposition holds if and only if $d = D^2$ and
    the matrices $\{A^i\}$ form a basis of $\MN{D}$.
\end{remark}

\section{Physical realisation of virtual insertions}%
\label{sec:phys_realize}
%% ---------------------------------------------------------

Given an injective tensor $A$ and a matrix $X \in \MN{D}$, one can
ask: is there a $d \times d$ matrix $O$ such that inserting $X$ on the
virtual bond has the same effect on any multi-site amplitude as acting
with $O$ on a physical index?

\begin{definition}[Physical realisation map]\label{def:phys_realize}
    \lean{MPSTensor.physRealize}
    \leanok
    \uses{def:decomposition_map}
    Let $A$ be an injective tensor.
    For $X \in \MN{D}$, define the $d \times d$ matrix
    $O_A(X) \in \MN{d}$ by
    \[
        O_A(X)_{ij} := (\Psi_A(A^i X))_j,
    \]
    so that row~$i$ of $O_A(X)$ contains the decomposition coefficients
    of $A^i X$ in the spanning set $\{A^j\}$.
    The map $X \mapsto O_A(X)$ is $\C$-linear.
\end{definition}

\begin{lemma}[Defining property of the physical realisation map]%
    \label{lem:physRealize_spec}
    \lean{MPSTensor.physRealize_spec}
    \leanok
    \uses{def:phys_realize, def:decomposition_map}
    For every $X \in \MN{D}$ and every physical index $i$,
    \[
        A^i\, X = \sum_{j=0}^{d-1} O_A(X)_{ij}\, A^j.
    \]
\end{lemma}

\begin{proof}
    \leanok
    By definition, $O_A(X)_{ij} = (\Psi_A(A^i X))_j$.
    Applying $\Phi_A$ gives
    $\sum_j O_A(X)_{ij} A^j = \Phi_A(\Psi_A(A^i X)) = A^i X$,
    where the last step uses $\Phi_A \circ \Psi_A = \mathrm{id}$.
\end{proof}

\begin{theorem}[Physical realisation is multiplicative -- basis case]%
    \label{thm:physRealize_mul}
    \lean{MPSTensor.physRealize_mul}
    \leanok
    \uses{def:phys_realize, lem:physRealize_spec, def:injective}
    Let $A$ be an injective MPS tensor such that $\{A^i\}_{i=0}^{d-1}$
    is a \emph{basis} of $\MN{D}$ (i.e.\ $d = D^2$ and the matrices are
    linearly independent).
    For all $X, Y \in \MN{D}$,
    \[
        O_A(XY) = O_A(X)\, O_A(Y).
    \]
\end{theorem}

\begin{proof}
    \leanok
    By Lemma~\ref{lem:physRealize_spec} applied to $Y$,
    \[
        A^j\, Y = \sum_{k} O_A(Y)_{jk}\, A^k.
    \]
    Apply Lemma~\ref{lem:physRealize_spec} to $XY$ and also to $X$:
    \begin{align*}
        A^i\,(XY)
        &= \sum_j O_A(XY)_{ij}\, A^j, \\
        A^i\, X
        &= \sum_j O_A(X)_{ij}\, A^j.
    \end{align*}
    Using associativity and the formula for $A^j Y$,
    \begin{align*}
        A^i\,(XY) = (A^i X)\,Y
        &= \left(\sum_j O_A(X)_{ij}\, A^j\right) Y
         = \sum_j O_A(X)_{ij}\, (A^j Y) \\
        &= \sum_j O_A(X)_{ij} \sum_k O_A(Y)_{jk}\, A^k
         = \sum_k (O_A(X)\, O_A(Y))_{ik}\, A^k.
    \end{align*}
    Thus $\sum_j O_A(XY)_{ij} A^j = \sum_k (O_A(X) O_A(Y))_{ik} A^k$
    for every $i$.
    Since $\{A^j\}$ is a basis of $\MN{D}$, the decomposition coefficients
    are unique, and comparing both sides gives
    $O_A(XY)_{ik} = (O_A(X)\, O_A(Y))_{ik}$ for all $i, k$.
\end{proof}

\begin{remark}[General injective case -- open problem]
    \label{rem:physRealize_mul_general}
    \uses{thm:physRealize_mul, def:injective}
    The multiplicativity $O_A(XY) = O_A(X)\,O_A(Y)$ is expected to hold
    for any injective $A$ (not just the basis case $d = D^2$), but the
    proof above breaks down when $d > D^2$ because the decomposition is
    no longer unique.
    In this over-complete setting the coefficient vectors
    $\{O_A(XY)_{i\bullet}\}$ and $\{(O_A(X)\,O_A(Y))_{i\bullet}\}$
    need not be equal even though they define the same element of $\MN{D}$.

    The paper \cite{Molnar2018NormalPEPS} sidesteps this difficulty by
    reformulating the key comparison in Lemma~2 in terms of
    \emph{traces} (see Lemma~\ref{lem:tensor_proportional} and
    Section~\ref{sec:tensor_proportional}): instead of requiring
    equality of operator-level virtual insertions, the proof works with
    the trace pairing $\tr(\,\cdot\,)$, which is non-degenerate and
    therefore recovers the matrix product $A_2^j A_1^i = B_2^j B_1^i$
    from the trace identity.
    The general multiplicativity of $O_A$ for $d > D^2$ remains
    unverified in the Lean formalisation and is left as an open problem.
\end{remark}

\section{Intermediate lemmas}%
\label{sec:intermediate}
%% ---------------------------------------------------------

Two auxiliary facts are used repeatedly. We state them here so that the
dependency graph records them as explicit nodes.

\begin{lemma}[Blocking preserves injectivity]\label{lem:blocking_preserves_injectivity}
    \notready
    \uses{def:injective, def:blocked_tensor}
    Let $A_0, \dots, A_{m-1}$ be injective MPS tensors with compatible
    bond dimensions.
    Then the blocked tensor
    $(A_0 \otimes \cdots \otimes A_{m-1})^{(i_0,\dots,i_{m-1})}
    := A_0^{i_0} A_1^{i_1} \cdots A_{m-1}^{i_{m-1}}$
    is again injective.
\end{lemma}

\begin{proof}
    The span of the blocked matrices contains all products
    $A_0^{i_0} \cdots A_{m-1}^{i_{m-1}}$.
    Since each $\{A_k^i\}_i$ spans $\MN{D}$, these products span
    all of $\MN{D}$.
\end{proof}

\begin{lemma}[Centre of $\MN{D}$]\label{lem:center_scalar}
    \notready
    If $Z \in \MN{D}$ commutes with every element of $\MN{D}$, then
    $Z = \lambda \Id_D$ for some $\lambda \in \C$.
\end{lemma}

\begin{proof}
    Standard: $Z$ commutes with every matrix unit $E_{ij}$, which
    forces all off-diagonal entries to vanish and all diagonal entries
    to agree.
\end{proof}

\section{Same state forces a virtual bond gauge}%
\label{sec:virtual_bond_gauge}
%% ---------------------------------------------------------

This section proves Lemma~1 of \cite{Molnar2018NormalPEPS}: if two
injective chains of length $n \ge 3$ generate the same state,
then on each bond the virtual insertion algebras are related by an
inner automorphism.

\begin{lemma}[Virtual bond gauge -- Lemma~1 of \protect\cite{Molnar2018NormalPEPS}]%
    \label{lem:virtual_bond_gauge}
    \notready
    \uses{thm:physRealize_mul, thm:simplicity, thm:skolem_noether,
          def:non_ti_chain, lem:blocking_preserves_injectivity}
    Let $\mathbf{A} = (A_0, \dots, A_{n-1})$ and
    $\mathbf{B} = (B_0, \dots, B_{n-1})$ be two injective non-TI
    periodic chains of length $n \ge 3$ that generate the same chain
    state.
    Fix a bond position $k \in \{0, \dots, n-1\}$.
    Then there exists an invertible matrix $Z_k \in \GL_D(\C)$, unique
    up to a nonzero scalar multiple, such that for every
    $X \in \MN{D}$,
    \[
        O_{A, k}(X) = O_{B, k}(Z_k^{-1} X Z_k),
    \]
    where $O_{A,k}$ (resp.\ $O_{B,k}$) denotes the physical realisation
    map obtained by blocking the sites adjacent to bond~$k$ in the
    $\mathbf{A}$-chain (resp.\ $\mathbf{B}$-chain) to form a three-site
    chain.
    Equivalently, inserting $X$ on bond~$k$ in the $\mathbf{A}$-chain
    gives the same physical operation as inserting $Z_k^{-1} X Z_k$ on
    bond~$k$ in the $\mathbf{B}$-chain.
\end{lemma}

\begin{proof}
    \uses{thm:physRealize_mul, thm:simplicity, thm:skolem_noether,
          lem:blocking_preserves_injectivity}

    \textbf{Step 1} (blocking to three sites).
    Block the sites on either side of bond~$k$ into a single site each,
    producing a three-site chain.
    By Lemma~\ref{lem:blocking_preserves_injectivity}, all three blocked
    tensors are injective.
    Define $O_{A,k}(X) \in \MN{d^{\ell}}$ (where $\ell$ is the block
    size) via the physical realisation map for the blocked tensor adjacent
    to bond~$k$.
    The map $X \mapsto O_{A,k}(X)$
    is a $\C$-algebra homomorphism from $\MN{D}$ to $\MN{d^{\ell}}$.

    \emph{Note.} The blocked tensor has physical dimension $d^{\ell}$,
    which generically exceeds $D^2$; thus
    Theorem~\ref{thm:physRealize_mul} (which is proved only for the
    basis case $d = D^2$) does not directly apply.
    The paper \cite{Molnar2018NormalPEPS} establishes multiplicativity
    by a different route: it constructs the algebra homomorphisms
    $X \mapsto O_1$, $O_1 \mapsto Y$ through diagrammatic inversion,
    avoiding the coefficient-uniqueness issue entirely
    (see Remark~\ref{rem:physRealize_mul_general}).
    The formalisation of this alternative argument is future work.

    \textbf{Step 2} (transfer of physical operations).
    Since $\mathbf{A}$ and $\mathbf{B}$ generate the same state,
    inserting any virtual matrix $X$ on bond~$k$ in the
    $\mathbf{A}$-chain produces the same physical-level amplitude
    change as some virtual insertion in the $\mathbf{B}$-chain.
    The physical operation $O_{A,k}(X)$ can therefore be realised on
    the $\mathbf{B}$-chain: there exists $Y \in \MN{D}$ with
    $O_{A,k}(X) = O_{B,k}(Y)$.
    By injectivity of $\mathbf{B}$, this $Y$ is unique.

    \textbf{Step 3} (the map $T$).
    Define $T : \MN{D} \to \MN{D}$ by $T(X) = Y$ as in Step~2.
    Linearity of $O_{A,k}$ and $O_{B,k}$ and uniqueness of $Y$
    imply that $T$ is $\C$-linear.

    \textbf{Step 4} (multiplicativity).
    Since both $X \mapsto O_{A,k}(X)$ and $X \mapsto O_{B,k}(X)$ are
    algebra homomorphisms (see the note in Step~1),
    $T(XY) = T(X)T(Y)$.

    \textbf{Step 5} (bijectivity).
    $T$ is a nonzero $\C$-algebra endomorphism of $\MN{D}$.
    By Theorem~\ref{thm:simplicity}, $T$ is bijective.

    \textbf{Step 6} (Skolem--Noether).
    By Theorem~\ref{thm:skolem_noether}, every automorphism of
    $\MN{D}$ is inner.
    Hence $T(X) = Z_k^{-1} X Z_k$ for some $Z_k \in \GL_D(\C)$.
    The matrix $Z_k$ is unique up to a scalar factor because two
    inner automorphisms $\operatorname{Ad}(Z)$ and $\operatorname{Ad}(Z')$
    coincide iff $Z' Z^{-1}$ is central, i.e.\ scalar
    (Lemma~\ref{lem:center_scalar}).
\end{proof}

\section{Aligned gauges force proportionality}%
\label{sec:tensor_proportional}
%% ---------------------------------------------------------

After applying Lemma~\ref{lem:virtual_bond_gauge} to each bond and
absorbing the gauge matrices, one obtains two chains whose
virtual-insertion operations agree on every bond.
Lemma~2 of \cite{Molnar2018NormalPEPS} shows that agreement on
\emph{both} bonds of a two-site block forces the local tensors to be
proportional.

\begin{lemma}[Aligned gauges force proportionality -- Lemma~2 of \protect\cite{Molnar2018NormalPEPS}]%
    \label{lem:tensor_proportional}
    \notready
    \uses{def:injective, def:decomposition_map, lem:center_scalar}
    Let $A_1, A_2$ and $B_1, B_2$ be injective MPS tensors with
    the same bond dimension $D$.
    Suppose that for all $X \in \MN{D}$ (on the internal bond) and
    all $Y \in \MN{D}$ (on the external bond),
    \begin{align}
        \operatorname{tr}(A_1^{i}\, X\, A_2^{j})
        &= \operatorname{tr}(B_1^{i}\, X\, B_2^{j})
        \qquad \text{for all } i, j,
        \label{eq:internal_bond} \\
        \operatorname{tr}(A_2^{j}\, Y\, A_1^{i})
        &= \operatorname{tr}(B_2^{j}\, Y\, B_1^{i})
        \qquad \text{for all } i, j.
        \label{eq:external_bond}
    \end{align}
    (Here \eqref{eq:internal_bond} says: inserting $X$ on the bond
    between sites $1$ and $2$ gives the same trace -- i.e.\ the same
    contribution to the state amplitude -- in both chains.
    Condition~\eqref{eq:external_bond} says the same for the
    external bond, with $Y$ inserted and the cyclic order reversed.)

    Then there exists a nonzero scalar $\lambda \in \C^{\times}$ such
    that
    \[
        A_1^{i} = \lambda\, B_1^{i}
        \qquad\text{and}\qquad
        A_2^{j} = \lambda^{-1}\, B_2^{j}
    \]
    for all physical indices $i, j$.
\end{lemma}

\begin{proof}
    \uses{def:decomposition_map, lem:center_scalar}

    \textbf{Step 1} (trace pairing to matrix products).
    The trace pairing $(M, N) \mapsto \operatorname{tr}(MN)$ on $\MN{D}$
    is non-degenerate: if $\operatorname{tr}(MX) = 0$ for all $X$, then
    $M = 0$.
    Condition~\eqref{eq:internal_bond} gives
    $\operatorname{tr}\bigl((A_2^{j} A_1^{i} - B_2^{j} B_1^{i}) X\bigr)
    = \operatorname{tr}(A_1^{i} X A_2^{j}) - \operatorname{tr}(B_1^{i} X B_2^{j}) = 0$
    for all $X$ (using cyclicity of the trace).
    By non-degeneracy,
    \begin{equation}\label{eq:prodL}
        A_2^{j} A_1^{i} = B_2^{j} B_1^{i} \qquad \text{for all } i,j.
    \end{equation}
    Likewise, \eqref{eq:external_bond} yields
    $\operatorname{tr}\bigl((A_1^{i} A_2^{j} - B_1^{i} B_2^{j}) Y\bigr) = 0$
    for all $Y$, giving
    \begin{equation}\label{eq:prodR}
        A_1^{i} A_2^{j} = B_1^{i} B_2^{j} \qquad \text{for all } i,j.
    \end{equation}

    \textbf{Step 2} (extract left and right factors via decomposition maps).
    Since $A_2$ is injective, the matrices $\{A_2^{j}\}$ span $\MN{D}$.
    Write $\Id_D = \Phi_{A_2}(\Psi_{A_2}(\Id_D)) = \sum_j c_j\, A_2^{j}$,
    where $c = \Psi_{A_2}(\Id_D) \in \C^d$.
    Using \eqref{eq:prodL},
    \[
        A_1^{i}
        = \Id_D \cdot A_1^{i}
        = \Bigl(\sum_j c_j\, A_2^{j}\Bigr) A_1^{i}
        = \sum_j c_j\, A_2^{j} A_1^{i}
        = \sum_j c_j\, B_2^{j} B_1^{i}
        = Z\, B_1^{i},
    \]
    where we define
    \begin{equation}\label{eq:A1_ZB1}
        Z := \Phi_{B_2}(\Psi_{A_2}(\Id_D))
           = \sum_j c_j\, B_2^{j} \;\in\; \MN{D}.
    \end{equation}
    Hence $A_1^{i} = Z\, B_1^{i}$ for all~$i$.

    For the right factor, use \eqref{eq:prodR} and the same coefficients $c$:
    \[
        A_1^{i}
        = A_1^{i} \cdot \Id_D
        = A_1^{i} \Bigl(\sum_j c_j\, A_2^{j}\Bigr)
        = \sum_j c_j\, A_1^{i} A_2^{j}
        = \sum_j c_j\, B_1^{i} B_2^{j}
        = B_1^{i}\, W,
    \]
    where
    \begin{equation}\label{eq:A1_B1W}
        W := \Phi_{B_2}(\Psi_{A_2}(\Id_D)) = Z.
    \end{equation}
    Hence $A_1^{i} = B_1^{i}\, W$ for all~$i$ with $W = Z$.

    \textbf{Step 3} (both factors are scalar).
    From \eqref{eq:A1_ZB1} and \eqref{eq:A1_B1W} we have
    $Z\, B_1^{i} = A_1^{i} = B_1^{i}\, Z$ for all $i$.
    Since $\{B_1^{i}\}_i$ spans $\MN{D}$ (injectivity of $B_1$), for any
    $M \in \MN{D}$ write $M = \sum_i e_i\, B_1^{i}$; then
    \[
        Z M = Z \sum_i e_i\, B_1^{i}
            = \sum_i e_i\, Z B_1^{i}
            = \sum_i e_i\, B_1^{i} Z
            = M Z.
    \]
    Thus $Z$ commutes with every element of $\MN{D}$.
    Setting $M = \Id_D$ confirms $Z = W$.
    By Lemma~\ref{lem:center_scalar}, $Z = \lambda \Id_D$ for some
    $\lambda \in \C$.

    \textbf{Step 4} (conclusion).
    Hence $A_1^{i} = \lambda\, B_1^{i}$ for all $i$.
    Substituting into \eqref{eq:prodL} gives
    $\lambda\, A_2^{j} B_1^{i} = B_2^{j} B_1^{i}$.
    Since $\{B_1^{i}\}_i$ spans $\MN{D}$, this forces
    $A_2^{j} = \lambda^{-1} B_2^{j}$ for all $j$.
    The tensors are nonzero, so $\lambda \neq 0$.
\end{proof}

\section{The Fundamental Theorem for injective chains}%
\label{sec:ft_chain}
%% ---------------------------------------------------------

Combining Lemma~\ref{lem:virtual_bond_gauge} with
Lemma~\ref{lem:tensor_proportional} yields the main theorem of
\cite{Molnar2018NormalPEPS} for non-translation-invariant injective MPS.

\begin{theorem}[Fundamental Theorem for injective chains -- Theorem~1 of \protect\cite{Molnar2018NormalPEPS}]%
    \label{thm:fundamentalTheorem_injective_chain}
    \notready
    \uses{lem:virtual_bond_gauge, lem:tensor_proportional,
          def:non_ti_chain}
    Let $\mathbf{A} = (A_0, \dots, A_{n-1})$ and
    $\mathbf{B} = (B_0, \dots, B_{n-1})$ be two injective non-TI periodic
    chains of length $n \ge 3$ that generate the same chain state.
    Then $\mathbf{A}$ and $\mathbf{B}$ are cyclically gauge equivalent:
    there exist invertible matrices $Z_0, \dots, Z_{n-1} \in \GL_D(\C)$,
    unique up to a common nonzero scalar factor, such that
    \[
        B_k^i = Z_k^{-1}\, A_k^i\, Z_{k+1}
    \]
    for every site $k \in \{0, \dots, n-1\}$ and every physical index
    $i$, with $Z_n = Z_0$.
\end{theorem}

\begin{proof}
    \uses{lem:virtual_bond_gauge, lem:tensor_proportional,
          lem:blocking_preserves_injectivity}

    \textbf{Step 1} (gauge on each bond via Lemma~1).
    Fix a bond $(k, k{+}1)$.
    Block all sites except $k$ and $k{+}1$ into a single site to
    form a three-site chain, which is injective by
    Lemma~\ref{lem:blocking_preserves_injectivity}.
    Apply Lemma~\ref{lem:virtual_bond_gauge} to obtain
    $Z_k \in \GL_D(\C)$ such that inserting $X$ on bond~$k$ in the
    $\mathbf{A}$-chain equals inserting $Z_k^{-1} X Z_k$ in the
    $\mathbf{B}$-chain.
    Repeat for every bond.

    \textbf{Step 2} (absorbing gauges).
    Define $\widetilde{B}_k^i := Z_k\, B_k^i\, Z_{k+1}^{-1}$ for each
    site~$k$.
    After this redefinition, the virtual insertion operations for
    $\mathbf{A}$ and $\widetilde{\mathbf{B}}$ agree on every bond:
    for every bond $k$ and every $X \in \MN{D}$, inserting $X$ gives
    the same amplitude in both chains.

    \textbf{Step 3} (proportionality via Lemma~2).
    Fix a pair of adjacent sites $(k, k{+}1)$.
    Block all other sites so that the chain becomes a two-site MPS.
    The agreement of virtual insertions on both the internal bond
    (between $k$ and $k{+}1$) and the external bond (through the
    blocked complement) gives precisely the two hypotheses
    \eqref{eq:internal_bond}--\eqref{eq:external_bond} of
    Lemma~\ref{lem:tensor_proportional}.
    Hence $A_k^i = \lambda_k \widetilde{B}_k^i$ for some
    $\lambda_k \in \C^{\times}$.
    Applying this to each adjacent pair gives scalars
    $\lambda_0, \dots, \lambda_{n-1}$.
    To see that $\prod_{k=0}^{n-1} \lambda_k = 1$: since both chains
    generate the same state, for any configuration
    $\sigma = (\sigma_0, \dots, \sigma_{n-1})$ we have
    \[
        \operatorname{tr}\!\bigl(A_0^{\sigma_0} \cdots A_{n-1}^{\sigma_{n-1}}\bigr)
        =
        \operatorname{tr}\!\bigl(\widetilde{B}_0^{\sigma_0} \cdots
                                 \widetilde{B}_{n-1}^{\sigma_{n-1}}\bigr).
    \]
    Substituting $A_k^{\sigma_k} = \lambda_k\, \widetilde{B}_k^{\sigma_k}$
    yields
    $\bigl(\prod_{k=0}^{n-1} \lambda_k\bigr)
     \operatorname{tr}(\widetilde{B}_0^{\sigma_0} \cdots
                       \widetilde{B}_{n-1}^{\sigma_{n-1}})
     = \operatorname{tr}(\widetilde{B}_0^{\sigma_0} \cdots
                         \widetilde{B}_{n-1}^{\sigma_{n-1}})$
    for all $\sigma$.
    Choosing $\sigma$ so that the trace is nonzero (possible by
    injectivity) gives $\prod_{k=0}^{n-1} \lambda_k = 1$.

    \textbf{Step 4} (absorbing scalars).
    Define $\mu_0 = 1$ and $\mu_k = \prod_{j=0}^{k-1} \lambda_j$ for
    $k = 1, \dots, n-1$.
    The condition $\prod_k \lambda_k = 1$ ensures $\mu_n = 1$.
    Replace $Z_k$ by $\mu_k\, Z_k$.
    Then $B_k^i = Z_k^{-1}\, A_k^i\, Z_{k+1}$ for all $k$ and $i$,
    with $Z_n = Z_0$.

    Uniqueness of the gauges up to a common scalar follows from the
    uniqueness clause in Lemma~\ref{lem:virtual_bond_gauge}.
\end{proof}

\begin{remark}[Relationship to the translation-invariant all-sizes theorem]
    \label{rem:ft_vs_singleBlock}
    \uses{thm:fundamentalTheorem_injective_chain, def:injective}
    The Lean library also contains a proof of a different, and in some
    respects stronger, version of the Fundamental Theorem
    (\lean{MPSTensor.fundamentalTheorem_singleBlock}).
    This result states: if $A$ is injective and $A$, $B$ generate the
    same MPV family \emph{for all system sizes} $N$ (i.e.\ \texttt{SameMPV A B}),
    then $A$ and $B$ are gauge equivalent without any phase ambiguity
    ($B^i = X A^i X^{-1}$ for some invertible $X$).

    The two results are complementary:
    \begin{itemize}
        \item \textbf{Theorem~\ref{thm:fundamentalTheorem_injective_chain}}
              (this chapter, from \cite{Molnar2018NormalPEPS}) works for
              \emph{non-TI} chains of \emph{fixed} length $n \ge 3$, and
              concludes gauge equivalence up to a common scalar factor.
              It does not require translation invariance.
        \item \textbf{\texttt{fundamentalTheorem\_singleBlock}} (Lean) is
              restricted to \emph{translation-invariant} pairs $A, B$ but
              assumes agreement for \emph{all} system sizes; the
              all-sizes hypothesis eliminates the phase ambiguity so the
              conclusion is exact gauge equivalence $B^i = X A^i X^{-1}$
              with no phase.
    \end{itemize}
    The fixed-size version is more general in allowing non-TI chains;
    the all-sizes version yields a cleaner conclusion.
    Neither result implies the other in full generality.
\end{remark}

\subsection{Translation-invariant chains}

\begin{corollary}[TI case]\label{cor:ft_ti_chain}
    \notready
    \uses{thm:fundamentalTheorem_injective_chain, def:non_ti_chain}
    Let $A$ and $B$ be injective MPS tensors of bond dimension $D$.
    Suppose the two translation-invariant chains
    $(A, A, \dots, A)$ and $(B, B, \dots, B)$ of length $n \ge 3$
    generate the same chain state.
    Then there exists an invertible $Z \in \GL_D(\C)$, unique up to a
    scalar, and a scalar $\lambda \in \C^{\times}$ with $\lambda^n = 1$,
    such that
    \[
        B^i = \lambda\, Z^{-1}\, A^i\, Z
    \]
    for every physical index~$i$.
\end{corollary}

\begin{proof}
    Apply Theorem~\ref{thm:fundamentalTheorem_injective_chain}
    to the two constant chains.
    This yields invertible matrices $Z_0, \dots, Z_{n-1}$ with
    $B^i = Z_k^{-1} A^i Z_{k+1}$ for all $k$ and $i$.
    Setting $G_k = Z_k Z_0^{-1}$, the relation
    $Z_k^{-1} A^i Z_{k+1} = Z_{k+1}^{-1} A^i Z_{k+2}$ shows
    $A^i = G_k^{-1} A^i G_k$ for suitable $G_k$.
    Since $\{A^i\}$ spans $\MN{D}$, each $G_k$ commutes with all of
    $\MN{D}$, so $G_k = \mu_k \Id$ by Lemma~\ref{lem:center_scalar}.
    Hence $Z_k = \mu_k Z_0$.
    The cyclic condition $Z_n = Z_0$ forces $\mu_0 = 1$ and
    $(\mu_1)^n = 1$.
    Setting $Z = Z_0$ and $\lambda = \mu_1$ yields the result.
\end{proof}

\subsection{Non-TI descriptions of TI states}

\begin{corollary}[Non-TI realisation of a TI state]%
    \label{cor:non_ti_to_ti}
    \notready
    \uses{thm:fundamentalTheorem_injective_chain, def:non_ti_chain}
    Let $\mathbf{A} = (A_0, \dots, A_{n-1})$ be an injective non-TI
    chain of length $n \ge 3$.
    Suppose the state generated by $\mathbf{A}$ is translation-invariant,
    i.e.\ it equals the state generated by the cyclically shifted chain
    $(A_1, \dots, A_{n-1}, A_0)$.
    Then $\mathbf{A}$ admits a translation-invariant description with the
    same bond dimension: there exist an injective tensor $B$ such that
    the TI chain $(B, \dots, B)$ generates the same state as $\mathbf{A}$.
    Explicitly, $B^i = A_0^i\, Z_0^{-1}$ (right multiplication,
    \emph{not} conjugation), where $Z_0$ is defined below.
\end{corollary}

\begin{proof}
    \uses{thm:fundamentalTheorem_injective_chain}
    The cyclic shift $\sigma(\mathbf{A}) = (A_1, \dots, A_{n-1}, A_0)$
    generates the same state by hypothesis.
    Apply Theorem~\ref{thm:fundamentalTheorem_injective_chain}:
    there exist invertible $Z_0, \dots, Z_{n-1}$ with
    $A_{k+1}^i = Z_k^{-1}\, A_k^i\, Z_{k+1}$ for all $k$ and $i$
    (indices mod~$n$).

    Define $L_k = Z_0 Z_1 \cdots Z_{k-1}$ and
    $R_k = Z_1 Z_2 \cdots Z_k$ for $k \ge 1$, with $L_0 = R_0 = \Id$.
    Iterating the gauge relation gives
    \[
        A_k^i = L_k^{-1}\, A_0^i\, R_k.
    \]
    Observe that $R_k\, L_{k+1}^{-1} = Z_0^{-1}$ for every $k$
    (since $L_{k+1} = Z_0\, R_k$).

    For the periodic chain ($A_n = A_0$) we need
    $R_n = L_n$, i.e.\ $Z_1 \cdots Z_n = Z_0 Z_1 \cdots Z_{n-1}$.
    The gauge relation at $k = n{-}1$ gives this closure.
    Substituting $A_k = L_k^{-1} A_0 R_k$ into the trace yields
    \begin{align*}
        \tr(A_0^{\sigma_0} \cdots A_{n-1}^{\sigma_{n-1}})
        &= \tr\bigl(A_0^{\sigma_0}\, Z_0^{-1}\, A_0^{\sigma_1}
           Z_0^{-1} \cdots A_0^{\sigma_{n-1}}\, Z_0^{-1}\bigr),
    \end{align*}
    using $R_k L_{k+1}^{-1} = Z_0^{-1}$ at each junction.
    Hence the state is generated by the TI chain with tensor
    $B^i := A_0^i\, Z_0^{-1}$.
    Since $A_0$ is injective and $Z_0$ is invertible, $B$ is injective.
\end{proof}

\subsection{Physical symmetries give virtual gauges}

\begin{corollary}[Physical symmetry implies virtual gauge]%
    \label{cor:symmetry_virtual_gauge}
    \notready
    \uses{cor:ft_ti_chain, def:injective, lem:center_scalar}
    Let $A$ be an injective MPS tensor and let $U \in \GL_d(\C)$ be a
    unitary acting on the physical index.
    Fix $n \ge 3$ and suppose the $n$-site TI state satisfies
    \[
        U^{\otimes n}\, \ket{\psi_n(A)} = \ket{\psi_n(A)}.
    \]
    Then there exist $V \in \GL_D(\C)$ and a scalar
    $\lambda \in \C^{\times}$ with $\lambda^n = 1$ such that
    \[
        \sum_{j=0}^{d-1} U_{ij}\, A^j
        = \lambda\, V^{-1}\, A^i\, V
    \]
    for every physical index~$i$.
\end{corollary}

\begin{proof}
    Define $B^i := \sum_j U_{ij} A^j$.
    Since $U$ is invertible, $\{B^i\}$ spans $\MN{D}$, so $B$ is
    injective.
    The symmetry condition means the TI chains $(A, \dots, A)$ and
    $(B, \dots, B)$ of length $n$ generate the same state.
    Corollary~\ref{cor:ft_ti_chain} gives $V \in \GL_D(\C)$
    and $\lambda$ with $\lambda^n = 1$ and
    $B^i = \lambda\, V^{-1}\, A^i\, V$ for all~$i$.
\end{proof}

\subsection{Normal MPS (blocked injectivity)}

\begin{corollary}[Normal MPS Fundamental Theorem]%
    \label{cor:normal_mps_ft}
    \notready
    \uses{thm:fundamentalTheorem_injective_chain, def:injective,
          def:blocked_tensor, def:normal,
          lem:blocking_preserves_injectivity}
    Let $\{A_k\}_{k=0}^{n-1}$ and $\{B_k\}_{k=0}^{n-1}$ be two
    non-TI periodic MPS on $n$ sites.
    Suppose that blocking any $L$ consecutive sites in either chain
    yields an injective tensor, and that $n \ge 3L$.
    If both chains generate the same state, then there exist invertible
    matrices $Z_0, \dots, Z_{n-1} \in \GL_D(\C)$, unique up to a
    common scalar, such that for every site $k$ and physical index $i$,
    \[
        B_k^i = Z_k^{-1}\, A_k^i\, Z_{k+1},
    \]
    with $Z_n = Z_0$.
    That is, gauge equivalence holds for the \emph{original} (unblocked)
    tensors.
\end{corollary}

\begin{proof}
    \uses{thm:fundamentalTheorem_injective_chain,
          lem:blocking_preserves_injectivity}
    Block every $L$ consecutive sites to form a chain of length
    $m = \lfloor n/L \rfloor \ge 3$ whose local tensors are the
    $L$-blocked tensors $A^{[L]}$ and $B^{[L]}$.
    These are injective by hypothesis.
    The blocked chains generate the same state (blocking is an isometry
    on the physical Hilbert space), so
    Theorem~\ref{thm:fundamentalTheorem_injective_chain} gives
    invertible gauges $\widehat{Z}_0, \dots, \widehat{Z}_{m-1}$ with
    $(B^{[L]})_k = \widehat{Z}_k^{-1}\, (A^{[L]})_k\, \widehat{Z}_{k+1}$.

    Now apply Lemma~\ref{lem:tensor_proportional} within each $L$-block
    to decompose the blocked gauge equivalence into site-by-site gauge
    equivalence: the proportionality constants at each site absorb into
    the bond gauges, yielding $Z_0, \dots, Z_{n-1}$ for the original
    tensors.
\end{proof}

\begin{remark}
    The paper's Appendix~\cite{Molnar2018NormalPEPS} strengthens the
    normal-MPS bound from $n \ge 3L$ to $n \ge 2L + 1$ by a more
    refined blocking argument.
    The injective PEPS theorem and the normal PEPS theorem on arbitrary
    graphs both rest on the same two core lemmas
    (Lemma~\ref{lem:virtual_bond_gauge} and
    Lemma~\ref{lem:tensor_proportional}); the graph geometry enters
    only through the blocking step that reduces to a three-site chain.
\end{remark}
