\section{Source statement}

\cite[Lemma~\leanid{equalMPS}, line~1081]{Cirac2016MPDO_arXiv} gives
\begin{align}
    \lim_{N\to\infty}\langle V_\alpha^{(N)},V_\alpha^{(N)}\rangle &=1,\qquad
    \lim_{N\to\infty}|\langle V_b^{(N)},V_a^{(N)}\rangle|         &\in\{0,1\}.
\end{align}
In the unit case,
\begin{align}
    D_a   &=D_b,\\
    A_b^i &=e^{i\phi}XA_a^iX^{-1}.
\end{align}
It does not assume proportionality of all blockwise MPV families.

\section{Source transfer-matrix core}

The proof defines
\begin{align}
    \mathbb E_{ab} &=\sum_i A_a^i\otimes\overline{A_b^i},
\end{align}
and the eigen-equation
\begin{align}
    \sum_i A_b^{i\dagger}X A_a^i &=\lambda X.
\end{align}
From Cauchy--Schwarz with the fixed point
\begin{align}
    \sum_{i}A_a^{i\dagger}A_a^i &=\sum_i A_b^{i\dagger}A_b^i=I,
\end{align}
the unit-overlap branch has \(|\lambda|=1\) and equality
\begin{align}
    A_b^iX &=\alpha XA_a^i,\qquad |\alpha|=1.
\end{align}
Then
\begin{align}
    X^\dagger X &=I.
\end{align}
So \(X\) is an isometry.
Dimensional comparison gives \(D_a=D_b\).
