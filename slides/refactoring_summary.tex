%======================================================================
% SLIDE 8 - Pattern 1: Term-mode conversion
%======================================================================
\begin{frame}[fragile]{Pattern 1: Term-Mode Conversion}
    \vspace{-0.3em}
    Replace simple tactic proofs with direct term-mode expressions.

    \vspace{0.3em}
    \textbf{\textcolor{softred}{Before:}}
    \begin{lstlisting}
theorem foo (h : A) : B := by
  intro x
  exact bar h x
    \end{lstlisting}

    \textbf{\textcolor{leangreen}{After:}}
    \begin{lstlisting}
theorem foo (h : A) : B :=
  fun x => bar h x
    \end{lstlisting}

    \vspace{0.5em}
    {\small
    \textbf{Why it helps:}
    \begin{itemize}\itemsep0.1em
        \item Eliminates tactic overhead for trivially constructive proofs
        \item Common in lemmas that just compose hypotheses
        \item Also: \texttt{by exfalso; exact h} $\to$ \texttt{absurd \ldots\ h}
    \end{itemize}
    }
\end{frame}

%======================================================================
% SLIDE 9 - Pattern 2: Simp consolidation
%======================================================================
\begin{frame}[fragile]{Pattern 2: Simp Consolidation}
    \vspace{-0.3em}
    Merge scattered \texttt{rw}/\texttt{simp} sequences into a single call.

    \vspace{0.3em}
    \textbf{\textcolor{softred}{Before:}}
    \begin{lstlisting}
  rw [map_add]
  simp [mul_comm]
  rw [trace_transpose]
  ring
    \end{lstlisting}

    \textbf{\textcolor{leangreen}{After:}}
    \begin{lstlisting}
  simp only [map_add, mul_comm, trace_transpose, ring]
    \end{lstlisting}

    \vspace{0.4em}
    {\small
    \textbf{Variants found:}
    \begin{itemize}\itemsep0.1em
        \item \texttt{unfold X; rw [Y]; simpa using Z} $\to$ \texttt{simp only [X, Y, Z]}
        \item \texttt{show \ldots\ from by rw [\ldots]} $\to$ direct \texttt{simp only}
        \item \texttt{conv} blocks with simple rewrites $\to$ \texttt{simp only} with targeted lemmas
    \end{itemize}
    }
\end{frame}

%======================================================================
% SLIDE 10 - Pattern 3: Have-chain compression
%======================================================================
\begin{frame}[fragile]{Pattern 3: Have-Chain Compression}
    \vspace{-0.3em}
    Inline single-use \texttt{have} bindings to reduce proof length.

    \vspace{0.3em}
    \textbf{\textcolor{softred}{Before:}}
    \begin{lstlisting}
  have h1 := foo x
  have h2 := bar h1
  have h3 : baz := quux h2
  exact h3
    \end{lstlisting}

    \textbf{\textcolor{leangreen}{After:}}
    \begin{lstlisting}
  exact quux (bar (foo x))
    \end{lstlisting}

    \vspace{0.3em}
    {\small
    \textbf{Guidelines:}
    \begin{itemize}\itemsep0.1em
        \item Only inline \texttt{have}s used exactly once
        \item Keep named \texttt{have}s that aid readability or are reused
        \item Chain with dot notation: \texttt{h.symm.trans h2} instead of named intermediates
    \end{itemize}
    }
\end{frame}

\begin{frame}[fragile,shrink=2]{Pattern 5: Leveraging Mathlib API}
    \vspace{-0.3em}
    The biggest wins came from discovering existing Mathlib lemmas.

    \vspace{0.2em}
    \textbf{\textcolor{softred}{Before} --- \texttt{componentMap\_injective} (14 lines):}
    \begin{lstlisting}[basicstyle=\ttfamily\fontsize{5pt}{6pt}\selectfont\color{white}]
theorem componentMap_injective ... :
    Function.Injective (componentMap T σ i) := by
  let f := componentMapRingHom hσ i
  have hker : TwoSidedIdeal.ker (f : Matrix _ _ ℂ →+* Matrix _ _ ℂ) = ⊥ := by
    rcases eq_bot_or_eq_top (TwoSidedIdeal.ker ...) with h | h
    · exact h
    · exfalso
      have h1 : (1 : Matrix ...) ∈ (⊤ : TwoSidedIdeal _) := trivial
      rw [← h] at h1; rw [TwoSidedIdeal.mem_ker] at h1
      change componentMap T σ i 1 = 0 at h1
      rw [componentMap_map_one hσ i] at h1; exact one_ne_zero h1
  intro x y hxy
  have hsub : x - y ∈ TwoSidedIdeal.ker ... := by ...
  ...
    \end{lstlisting}

    \textbf{\textcolor{leangreen}{After} (1 line):}
    \begin{lstlisting}
theorem componentMap_injective ... :
    Function.Injective (componentMap T σ i) :=
  (componentMapRingHom hσ i).injective
    \end{lstlisting}

    {\small \textbf{Key:} \texttt{RingHom.injective} for \texttt{IsSimpleRing} does the kernel argument automatically.}
\end{frame}

% SLIDE 13 - Pattern 5 continued: densityMatrices_isCompact

\begin{frame}[fragile]{Pattern 5 (cont.): API Discovery --- \texttt{isCompact}}
    \vspace{-0.3em}
    \textbf{\textcolor{softred}{Before} --- \texttt{densityMatrices\_isCompact} (10 lines):}
    {\small from \texttt{PositiveMap.lean}}
    \begin{lstlisting}[basicstyle=\ttfamily\fontsize{5.5pt}{6.5pt}\selectfont\color{white}]
theorem densityMatrices_isCompact :
    IsCompact (densityMatrices D) := by
  haveI : ProperSpace (Matrix (Fin D) (Fin D) ℂ) := ...
  apply Metric.isCompact_of_isClosed_isBounded
  · apply IsClosed.inter isClosed_posSemidef
    exact isClosed_eq ... continuous_const
  · apply Bornology.IsBounded.subset
        (posSemidef_trace_bounded_isBounded 1)
    intro X ⟨hX_psd, hX_tr⟩
    exact ⟨hX_psd, by rw [hX_tr]; simp⟩
    \end{lstlisting}

    \textbf{\textcolor{leangreen}{After} (3 lines):}
    \begin{lstlisting}[basicstyle=\ttfamily\fontsize{5.5pt}{6.5pt}\selectfont\color{white}]
theorem densityMatrices_isCompact :
    IsCompact (densityMatrices D) := by
  haveI : ProperSpace (Matrix (Fin D) (Fin D) ℂ) := ...
  exact Metric.isCompact_of_isClosed_isBounded
    (isClosed_posSemidef.inter (isClosed_eq ... continuous_const))
    ((posSemidef_trace_bounded_isBounded 1).subset
      fun _ ⟨h1, h2⟩ => ⟨h1, by rw [h2]; simp⟩)
    \end{lstlisting}

    {\small \textbf{Key:} Dot notation chains (\texttt{.inter}, \texttt{.subset}) eliminate \texttt{apply}/\texttt{exact} scaffolding.}
\end{frame}

%======================================================================
% SLIDE 14 - Pattern 6: Tactic compression
%======================================================================
\begin{frame}[fragile]{Pattern 6: Tactic Compression}
    \vspace{-0.3em}
    Lean~4's combinator tactics replace verbose bullet-point proofs.

    \vspace{0.4em}
    \textbf{\textcolor{softred}{Before:}}
    \begin{lstlisting}
  constructor
  · intro x
    simp [mul_comm x]
  · intro y
    simp [mul_comm y]
    \end{lstlisting}

    \textbf{\textcolor{leangreen}{After:}}
    \begin{lstlisting}
  constructor <;> intro z <;> simp [mul_comm z]
    \end{lstlisting}

    \vspace{0.4em}
    {\small
    \textbf{Common compressions:}
    \begin{itemize}\itemsep0.1em
        \item \texttt{split; \textbullet\ simp; \textbullet\ simp} $\to$ \texttt{split <;> simp}
        \item \texttt{constructor} + bullets $\to$ \texttt{exact $\langle$\ldots, \ldots$\rangle$}
        \item \texttt{refine $\langle$\_, ?\_ $\rangle$; intro x; \ldots} $\to$ \texttt{refine $\langle$\_, fun x => ?\_ $\rangle$}
    \end{itemize}
    }
\end{frame}

% SLIDE 15 - Pattern 7: Calc chain simplification
%======================================================================
\begin{frame}[fragile,shrink=5]{Pattern 7: Calc Chain Simplification}
    \vspace{-0.3em}
    Multi-step \texttt{calc} blocks often collapse to a single \texttt{rw} or tactic.

    \vspace{0.2em}
    \textbf{\textcolor{softred}{Before:}}
    \begin{lstlisting}[basicstyle=\ttfamily\fontsize{6pt}{7.5pt}\selectfont\color{white}]
  calc ||f x|| ≤ ||f|| * ||x||     := f.le_opNorm x
    _ ≤ 1 * ||x||               := by nlinarith [hf_bound]
    _ = ||x||                   := one_mul _
    \end{lstlisting}

    \textbf{\textcolor{leangreen}{After:}}
    \begin{lstlisting}[basicstyle=\ttfamily\fontsize{6pt}{7.5pt}\selectfont\color{white}]
  nlinarith [f.le_opNorm x, hf_bound]
    \end{lstlisting}

    \vspace{0.15em}
    \textbf{\textcolor{softred}{Before:}}
    \begin{lstlisting}[basicstyle=\ttfamily\fontsize{6pt}{7.5pt}\selectfont\color{white}]
  calc A * B * C = A * (B * C)    := by ring
    _ = A * D                     := by rw [hBCD]
    _ = E                         := by rw [hAD]
    \end{lstlisting}

    \textbf{\textcolor{leangreen}{After:}}
    \begin{lstlisting}[basicstyle=\ttfamily\fontsize{6pt}{7.5pt}\selectfont\color{white}]
  rw [show A * B * C = A * (B * C) from by ring, hBCD, hAD]
    \end{lstlisting}

    {\small \textbf{Savings:} SpectralGap alone had ${\sim}$12 calc blocks simplified, saving ${\sim}$80 lines.
    {\tiny (from \texttt{SpectralGap.lean})}}
\end{frame}

    \textbf{\textcolor{softred}{Before} (3-step calc):}
    \begin{lstlisting}[basicstyle=\ttfamily\fontsize{6pt}{7.5pt}\selectfont\color{white}]
  calc (1 - P) * A i * P * (P * ((A i)(*@	extsuperscript{H}@*) * (1 - P)))
      = (1 - P) * A i * (P * P) * (A i)(*@	extsuperscript{H}@*) * (1 - P)
          := by noncomm_ring
    _ = (1 - P) * A i * P * (A i)(*@	extsuperscript{H}@*) * (1 - P)
          := by rw [hPP]
    _ = (1 - P) * (A i * P * (A i)(*@	extsuperscript{H}@*)) * (1 - P)
          := by noncomm_ring
    \end{lstlisting}

    \textbf{\textcolor{leangreen}{After} (2 lines):}
    \begin{lstlisting}[basicstyle=\ttfamily\fontsize{6pt}{7.5pt}\selectfont\color{white}]
  rw [show (1 - P) * A i * P * (P * ((A i)(*@	extsuperscript{H}@*) * (1 - P)))
      = (1 - P) * A i * (P * P) * (A i)(*@	extsuperscript{H}@*) * (1 - P)
      from by noncomm_ring, hProj.2]
  noncomm_ring
    \end{lstlisting}

    \vspace{0.3em}
    {\small
    \textbf{Pattern:}
    Use \texttt{noncomm\_ring} for associativity/distribution steps,
    interleave with hypothesis rewrites via \texttt{show \ldots\ from by noncomm\_ring}.

    \vspace{0.2em}
    \textbf{Impact:} Especially effective in \texttt{CPPrimitive.lean} and \texttt{SpectralGap.lean}.
    }
\end{frame}

\begin{frame}{Lesson 1: Agents Must \texttt{git commit}}
    \vspace{-0.5em}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{block}{\textcolor{softred}{Problem: Wave 1}}
            \begin{itemize}\itemsep0.2em\small
                \item 3 agents successfully simplified files
                \item Changes made via \texttt{edit\_file} on filesystem
                \item \textbf{No \texttt{git commit} issued}
                \item iCloud sync + concurrent agent builds\\
                      $\to$ \textcolor{softred}{4 files' work lost}
                \item Had to redo Wave 1 from scratch
            \end{itemize}
        \end{block}

        \column{0.48\textwidth}
        \begin{block}{\textcolor{leangreen}{Solution: Mandatory commit}}
            \begin{itemize}\itemsep0.2em\small
                \item Added to agent instructions:\\
                      ``\texttt{git add \& git commit after each file}''
                \item \textcolor{leangreen}{Zero lost work} after Wave 1 redo
                \item Bonus: \texttt{git diff} provides natural\\
                      progress tracking per file
                \item Enables easy rollback if a simplification\\
                      breaks downstream builds
            \end{itemize}
        \end{block}
    \end{columns}

    \vspace{0.5em}
    \begin{center}
    {\small\textcolor{amber}{$\blacktriangleright$}
    \textbf{Rule:} Any AI agent editing files \emph{must} commit atomically.
    Filesystem edits without version control are ephemeral.}
    \end{center}
\end{frame}

\begin{frame}{Lesson 2: Subdirectory Isolation}
    \vspace{-0.3em}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \textbf{\textcolor{softred}{Failure mode:}}
        \begin{itemize}\itemsep0.2em\small
            \item Two agents editing files in \emph{same} subdirectory
            \item Both run \texttt{lake build} simultaneously
            \item Shared \texttt{.olean} cache gets corrupted
            \item Both agents see spurious type errors
            \item Cascading retries waste 30+ minutes
        \end{itemize}

        \vspace{0.3em}
        \textbf{\textcolor{leangreen}{Solution:}}
        \begin{itemize}\itemsep0.2em\small
            \item Max \textcolor{leangreen}{3 concurrent agents}
            \item Each in a \textbf{different} subdirectory:\\
                  e.g., Channel/ + MPS/ + Algebra/
            \item Serialize agents within same subdir
        \end{itemize}

        \column{0.48\textwidth}
        \begin{center}
