\begin{frame}{Safety mechanisms}
    \vspace{-0.6em}
    \begin{columns}[T]
        \column{0.52\textwidth}
        {\small\textcolor{amber}{\textbf{Five layers of protection:}}}
        \begin{enumerate}\setlength{\itemsep}{1pt}
            \item \textbf{Iteration cap} --- max 5 consecutive bot commits
            \item \textbf{Label gate} --- review-fix needs \texttt{auto-fix-claude}
            \item \textbf{Concurrency groups} --- \texttt{bot-fix-\$\{branch\}}
            \item \textbf{Fork guard} --- same-repo PRs only
            \item \textbf{Outdated filter} --- \texttt{!isResolved \&\& !isOutdated}
        \end{enumerate}

        \column{0.46\textwidth}
        {\small\textcolor{amber}{\textbf{Iteration counting:}}}
        \vspace{0.2em}

        {\ttfamily\scriptsize\color{dimwhite}
        COUNT=0\\
        while read -r MSG; do\\
        ~~if echo "\$MSG" |\\
        ~~~~grep -qE \textquotesingle [claude-(auto|review)-fix]\textquotesingle;\\
        ~~then COUNT=\$((COUNT + 1))\\
        ~~else break \# human commit\\
        ~~fi\\
        done < <(git log -{}-pretty=\%s -n 5)
        }
        \vspace{0.2em}

        {\scriptsize \textcolor{leangreen}{A human commit resets the counter.} Both \texttt{[claude-auto-fix]} and \texttt{[claude-review-fix]} share the budget.}
    \end{columns}
\end{frame}

        \textbf{Can rebase?} & \textcolor{softred}{No} & \textcolor{softred}{No} \\
        \textbf{Can fetch branches?} & \textcolor{leangreen}{Yes} & \textcolor{softred}{No (HTTP 403)} \\
        \textbf{Typical PR size} & 50--500 lines & \textcolor{amber}{6--60 lines} \\

        \bottomrule
    \end{tabular}
    }
    \end{center}
    \vspace{0.3em}
    \textcolor{dimwhite}{\footnotesize Both agents create new PRs from issue comments --- neither can push to an existing PR from an issue. Critical distinction: @codex on \textbf{issues} $\neq$ @codex on \textbf{PRs}.}
\end{frame}

\begin{frame}{The bracket incompatibility}
    \vspace{-0.2em}
    \begin{columns}[T]
        \column{0.50\textwidth}
        \textcolor{softred}{\textbf{Root cause:}}
        \vspace{0.2em}
        \begin{itemize}
            \item Issue titles used \texttt{[brackets]}: ``\texttt{[Wolf Ch6] \ldots}''
            \item @codex generates branch slug from title
            \item Strips \texttt{[} but \textbf{keeps} \texttt{]}
            \item Branch: \texttt{codex/wolf-ch5]-schwarz\ldots}
            \item \texttt{claude-code-action} rejects \texttt{]} in branch names
            \item \textbf{100\% failure rate} for @claude on codex branches
        \end{itemize}
        \vspace{0.3em}
        \textcolor{leangreen}{\textbf{Solution adopted:}} Bracket-free issue naming.\\
        {\scriptsize ``Wolf Ch6 ---'' instead of ``[Wolf Ch6]''}

        \column{0.48\textwidth}
        \textcolor{softred}{\textbf{Other failure patterns:}}
        \vspace{0.2em}
        \begin{itemize}
            \item \textbf{PR proliferation}: every @codex on issue = new PR\\
              {\scriptsize \#162 $\to$ \#216, \#200 $\to$ \#208, \#165 $\to$ \#215}
            \item \textbf{Tiny PRs}: 6--60 lines per @codex PR\\
              {\scriptsize Review overhead disproportionate to changes}
            \item \textbf{``Couldn't complete''}: 7 documented failures\\
              {\scriptsize PRs \#212, \#211, \#206, \#201, \#196, \#167, \#91}
            \item \textbf{Phantom pushes}: @codex says ``committed'' but commit stays in cloud sandbox\\
              {\scriptsize User must click Codex portal link to finalize}
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{Lessons learned}
    \vspace{-0.2em}
    \begin{enumerate}
        \item \textcolor{leangreen}{\textbf{The fixed-point loop works}} --- but needs an iteration cap.\\
          {\scriptsize Most PRs converge in 1--2 rounds. Without a cap, Bugbot feedback loops run forever.}
        \item \textcolor{leangreen}{\textbf{Never merge without reading all 3 comment APIs.}}\\
          {\scriptsize Inline, PR-level, and review summaries live in separate endpoints. Miss one $\to$ merge bugs.}
        \item \textcolor{amber}{\textbf{@codex on issues $\neq$ @codex on PRs.}}\\
          {\scriptsize Issue comments create new PRs. PR comments push fixes. Confusing the two causes proliferation.}
        \item \textcolor{amber}{\textbf{Branch naming conventions propagate into CI compatibility.}}\\
          {\scriptsize \texttt{[brackets]} in issue titles $\to$ \texttt{]} in branch names $\to$ @claude 100\% failure rate.}
        \item \textcolor{leangreen}{\textbf{Auto-fix is multiplicative.}}\\
          {\scriptsize One human push triggers cascading improvements: build fix $\to$ blueprint fix $\to$ review fix.}
        \item \textcolor{leangreen}{\textbf{Tracking issues + follow-up automation = self-maintaining state.}}\\
          {\scriptsize The project tracks itself. Checklists update, follow-ups spawn, progress is reported automatically.}
    \end{enumerate}
\end{frame}
