\begin{frame}{Discovery During Formalization: A False Helper}
    \vspace{-0.5em}

    \begin{columns}[T]
        \column{0.50\textwidth}
        \textbf{\textcolor{softred}{The false claim:}}\\
        {\small \texttt{transferMap\_fixed\_mul\_conjTranspose}:}
        \[
            \mathcal{E}_A(XX^\dagger) = XX^\dagger
        \]
        {\scriptsize where $\mathcal{F}_{AB}(X) = \mu X$ with $|\mu|=1$.}

        \vspace{0.3em}
        \textbf{Why it's false:}
        \begin{itemize}\itemsep0.1em\small
            \item $\mathcal{E}_A(XX^\dagger) = \sum_i A_i X X^\dagger A_i^\dagger$
            \item $\mathcal{F}_{AB}(X) = \sum_i A_i X B_i^\dagger$
            \item Different operations; eigenvector of $\mathcal{F}_{AB}$ does \textbf{not} make $XX^\dagger$ a fixed point of $\mathcal{E}_A$
        \end{itemize}

        \column{0.46\textwidth}
        \textbf{\textcolor{leangreen}{The correct approach:}}\\
        {\small Work at the block level:}
        \begin{enumerate}\itemsep0.1em\small
            \item Embed into block channel $E_{\mathrm{block}}$
            \item $M = (\begin{smallmatrix}0 & X' \\ 0 & 0\end{smallmatrix})$ is peripheral eigenvector of $E_{\mathrm{block}}$
            \item $M^\dagger M = (\begin{smallmatrix}0 & 0 \\ 0 & X'^\dagger X'\end{smallmatrix})$
            \item KS equality holds for $E_{\mathrm{block}}$
            \item This gives the correct matrix identity
        \end{enumerate}

        \vspace{0.2em}
        {\scriptsize\textcolor{leangreen}{Formalization forces precision about which channel acts on which matrix.}}
