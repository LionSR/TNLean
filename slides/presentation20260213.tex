\begin{frame}{Proved: PSD Fixed Point $\Rightarrow$ Positive Definite (Injectivity)}
    \vspace{-0.3em}

    \textbf{Theorem} (\texttt{posSemidef\_fixedPoint\_isPosDef}):
    If $\{A^i\}$ spans $\MD$ and $\rho \geq 0$, $\rho \neq 0$,
    $\mathcal{E}(\rho) = \rho$, then $\rho > 0$.

    \vspace{0.3em}
    \textbf{Proof idea} (contradiction, $\sim$170 lines):
    \begin{enumerate}\itemsep0.2em
        \item {\small Suppose $\exists\, x \neq 0$ with $\rho x = 0$.}
        \item {\small From the fixed-point equation $\rho = \sum_i A_i \rho A_i^\dagger$:
            \[
                0 = \langle x | \rho | x\rangle = \sum_i \underbrace{\langle A_i^\dagger x | \rho | A_i^\dagger x\rangle}_{\geq 0}
                    \quad\Longrightarrow\quad \rho\, (A_i^\dagger x) = 0\ \ \forall\, i
            \]}
        \item {\small By injectivity, $\{A_i\}$ spans $\MD$, so $\{A_i^\dagger\}$ spans $\MD$.
            Hence $\rho\, (M x) = 0$ for \emph{all} $M$.}
        \item {\small Since $x \neq 0$, the map $M \mapsto Mx$ is surjective,
            so $\rho = 0$. Contradiction.}
    \end{enumerate}

    \vspace{0.2em}
    {\small \textbf{Key insight}: use injectivity directly at the vector level.
    No spectral projections, no functional calculus. The proof is short and elementary.}
\end{frame}

\begin{frame}{Proved: PSD Fixed Point $\Rightarrow$ Positive Definite (Irreducibility)}
    \vspace{-0.3em}

    \textbf{Theorem} (\texttt{posSemidef\_fixedPoint\_isPosDef\_of\_irreducible}):
    If $\mathcal{E}$ is irreducible CP and $\rho \geq 0$, $\rho \neq 0$,
    $\mathcal{E}(\rho) = \rho$, then $\rho > 0$.

    \vspace{0.2em}
    \textbf{Proof idea} (contradiction via support projection, $\sim$220 lines):
    \begin{enumerate}\itemsep0.15em
        \item {\small Assume $\rho$ PSD but not PD $\Rightarrow$ some eigenvalue is 0.}
        \item {\small Construct the \textbf{spectral support projection}
            $Q = U \cdot \diag(\mathbf{1}_{\lambda > 0}) \cdot U^\dagger$
            where $\rho = U \cdot \diag(\lambda) \cdot U^\dagger$.}
        \item {\small Show $Q$ is an orthogonal projection with $\ker(Q) = \ker(\rho)$.}
        \item {\small From ``$\ker(\rho)$ invariant under $A_i^\dagger$'' (reusing lemma from Part 1):
            \[
                (1 - Q) \cdot A_i \cdot Q = 0 \quad\Longrightarrow\quad
                Q \cdot \mathcal{E}(Q X Q) \cdot Q = \mathcal{E}(Q X Q) \quad \forall\, X
            \]
            (verified using \texttt{noncomm\_ring} tactic for the matrix algebra).}
        \item {\small By irreducibility: $Q = 0$ or $Q = 1$.
            But $Q \neq 0$ (since $\rho \neq 0$) and $Q \neq 1$ (since $\rho$ not PD). \qed}
    \end{enumerate}

    \vspace{0.2em}
    {\small \textbf{Key tools}: Mathlib's \texttt{IsHermitian.spectral\_theorem},
    \texttt{diagonal\_conjTranspose}, and \texttt{noncomm\_ring} for non-commutative identities.}
\end{frame}

\begin{frame}{Proved: Uniqueness of PSD Fixed Point}
    \vspace{-0.3em}

    \textbf{Theorem} (\texttt{posSemidef\_fixedPoint\_unique}):
    If $\rho, \sigma$ are both nonzero PSD fixed points of $\mathcal{E}_A$
    (with $A$ injective), then $\sigma = c\rho$ for some $c > 0$.

    \vspace{0.3em}
    \textbf{Proof idea} ($\sim$330 lines, using \texttt{exists\_critical\_scalar}):
    \begin{enumerate}\itemsep0.15em
        \item {\small Both $\rho, \sigma$ are PD (by the theorem above).}
        \item {\small Diagonalize $\rho^{-1/2}\sigma\,\rho^{-1/2}$ and find the minimum eigenvalue $c_0 > 0$.}
        \item {\small Then $\sigma - c_0 \rho$ is PSD, a fixed point, and \textbf{has a zero eigenvalue}.}
        \item {\small If $\sigma - c_0 \rho \neq 0$, it's a nonzero PSD fixed point that's not PD.
            Contradiction with the PD theorem!}
        \item {\small Hence $\sigma = c_0 \rho$.}
    \end{enumerate}

    \vspace{0.3em}
    {\small \textbf{Key Mathlib tools}:
    \texttt{PosDef.isUnit} (PD matrices are invertible),
    spectral theorem for the conjugated matrix,
    and the PD theorem from Part 1 for the contradiction.}
\end{frame}

\begin{frame}{The Real Blocker: Spectral Gap}
    \vspace{-0.3em}

    The multi-block theorem needs:
    \[
        A_k \neq B_\ell \text{ (not gauge-equivalent)} \quad\Longrightarrow\quad
        \rho\!\left(\mathcal{F}_{A_k B_\ell}\right) < 1
    \]

    \vspace{0.2em}
    \textbf{What this requires} (not yet formalized):
    \begin{enumerate}\itemsep0.15em\small
        \item Existence of PSD eigenvector for spectral radius \hfill\textcolor{softred}{[Krein--Rutman]}
        \item This eigenvector is PD (our new theorem gives this \textcolor{leangreen}{\checkmark})
        \item Spectral radius of $\mathcal{E}_A$ is simple and dominant \hfill\textcolor{softred}{[QPF spectral theory]}
        \item For the \emph{mixed} operator $\mathcal{F}_{AB}$: no PD fixed point when $A \not\sim B$
            \hfill\textcolor{softred}{[cross analysis]}
        \item Hence spectral radius $< 1$ \hfill\textcolor{softred}{[primitivity]}
    \end{enumerate}

\begin{frame}{Afternoon Breakthrough: Existence via Ces\`aro Mean}
    \vspace{-0.3em}

    \textbf{Problem}: prove $\exists\, \rho \geq 0,\; \rho \neq 0,\; \mathcal{E}(\rho) = \rho$
    \emph{without} Brouwer FPT (not in Mathlib).

    \vspace{0.3em}
    \textbf{Key idea} (Wolf Ch.\,6 \cite{Wolf2012Quantum}): use
    \textbf{Ces\`aro mean + compactness} (Markov--Kakutani style).

    \vspace{0.2em}
    \begin{enumerate}\itemsep0.15em\small
        \item Let $K = \{\rho \geq 0 \mid \tr(\rho) = 1\}$
            \hfill {\scriptsize\textcolor{dimwhite}{density matrices}}
        \item $K$ is \textbf{compact}
            {\scriptsize (Heine--Borel: closed PSD cone $\cap$ trace\,=\,1, entries bounded by trace norm)}
            \hfill \textcolor{leangreen}{\checkmark}
        \item $K$ is \textbf{convex} {\scriptsize (PSD cone convex + trace linear)}
            and \textbf{nonempty} {\scriptsize (witness: $\tfrac{1}{D}\cdot I$)}
            \hfill \textcolor{leangreen}{\checkmark}
        \item Channel $\mathcal{E}$ maps $K \to K$
            {\scriptsize (preserves PSD + trace)}
            \hfill \textcolor{leangreen}{\checkmark}
        \item Ces\`aro mean: $\sigma_N = \frac{1}{N}\sum_{n=0}^{N-1}\mathcal{E}^n(\rho_0) \in K$
            \hfill {\scriptsize\textcolor{dimwhite}{convexity}}
        \item Extract convergent subsequence $\sigma_{N_k} \to \sigma \in K$
            \hfill {\scriptsize\textcolor{dimwhite}{compactness}}
        \item \textbf{Telescoping}: $\mathcal{E}(\sigma_N) - \sigma_N
                = \frac{1}{N}\bigl(\mathcal{E}^N(\rho_0) - \rho_0\bigr) \to 0$
            \hfill \textcolor{leangreen}{\checkmark}
        \item Continuity: $\mathcal{E}(\sigma) = \sigma$,
            and $\sigma \neq 0$ since $\tr(\sigma) = 1$ \hfill $\square$
    \end{enumerate}

    \vspace{0.3em}
    \begin{center}
        \begin{tikzpicture}
            \node[draw=leangreen,thick,rounded corners=4pt,fill=cardbg,text=white,
                text width=0.85\textwidth,align=center,inner sep=6pt,font=\small] {
                \textbf{No Brouwer FPT needed!}
                The entire proof stays within Mathlib's existing topology.\\
                {\scriptsize \texttt{IsChannel.exists\_posSemidef\_fixedPoint}
                --- fully proved, 0 sorry}
            };
        \end{tikzpicture}
    \end{center}
\end{frame}
