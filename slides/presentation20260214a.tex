\begin{frame}{The Equal-or-Orthogonal Lemma (Key to Multi-Block)}
    \vspace{-0.5em}

    {\small For two normal tensors $A_1, A_2$ generating states $|V_N(A_1)\rangle,  |V_N(A_2)\rangle$:}

    \vspace{0.2em}
    \begin{center}
    \colorbox{cardbg}{\parbox{0.85\textwidth}{\centering
    $
    \langle V_N(A_1)\,|\,V_N(A_2)\rangle = 
    \tr\!(\mathcal{F}_{12}^N(\mathbf{I}))
    \qquad\text{where}\quad
    \mathcal{F}_{12}(X) = \sum_i A_1^i\, X\, (A_2^i)^\dagger
    $}}
    \end{center}

    \vspace{0.3em}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \textbf{Case 1:} $\rho(\mathcal{F}_{12}) < 1$
        \begin{itemize}\small
            \item $\tr(\mathcal{F}_{12}^N) \to 0$
            \item States become \textbf{\textcolor{leanblue}{orthogonal}}
            \item $\Rightarrow$ different blocks
        \end{itemize}

        \column{0.48\textwidth}
        \textbf{Case 2:} $\rho(\mathcal{F}_{12}) = 1$
        \begin{itemize}\small
            \item Overlap persists
            \item Tensors are \textbf{\textcolor{leangreen}{gauge equivalent}}
            \item $\Rightarrow$ same block
        \end{itemize}
    \end{columns}

    \vspace{0.3em}
    {\small \textbf{Our status:}}
    \begin{itemize}\itemsep0.05em\small
        \item[\textcolor{leangreen}{\checkmark}] $\rho(\mathcal{F}_{12}) \leq 1$ \hfill{\tiny Hilbert--Schmidt contraction, fully proved}
        \item[\textcolor{leangreen}{\checkmark}] $\rho < 1 \Rightarrow \mathcal{F}^n\to 0$ \hfill{\tiny Gelfand formula, fully proved}
        \item[\textcolor{softred}{$\boldsymbol\times$}] $\rho = 1 \Rightarrow$ gauge equiv \hfill{\tiny \textcolor{softred}{\texttt{axiom}} --- the hard part (Wolf Thm 6.6)}
    \end{itemize}

    \vspace{0.15em}
    {\footnotesize \textcolor{dimwhite}{Refs:~\cite{Perez-Garcia2007Matrix,Wolf2012Quantum}}}
\end{frame}

\begin{frame}{Normal Tensors: Blocking and the Wielandt Bound}
    \vspace{-0.5em}

    \textbf{Key idea}: a normal tensor is ``almost injective'' ---
    just block enough sites.

    \vspace{0.3em}
    \begin{columns}[T]
        \column{0.52\textwidth}
        \textbf{Blocking $L$ sites}:\\[3pt]
        {\small Given MPS tensor $A$ with physical dim $d$, bond dim $D$,
        define the \emph{$L$-blocked tensor}:}
        \[
            A^{[L]}_{i_1\ldots i_L} = A_{i_1}\, A_{i_2}\cdots A_{i_L}
        \]
        {\small New physical dim $= d^L$, same bond dim $D$.
        Transfer operator: $\mathcal{E}_{A^{[L]}} = \mathcal{E}_A^L$.}

        \vspace{0.4em}
        \textbf{Normal $\Leftrightarrow$ primitive CP}:\\[2pt]
        {\small $\mathcal{E}_A$ has 1 as the \emph{only} eigenvalue on $|z|=1$.\\
        Implies: $\mathcal{E}_A^n \to P$ (rank-1 projection).}

        \column{0.48\textwidth}
        \textbf{\textcolor{amber}{Quantum Wielandt bound}} \cite{Sanz2010Quantum}:
        \vspace{0.2em}

        \colorbox{cardbg}{\parbox{0.92\columnwidth}{\centering\small
        $L_0 \leq D^2(D^2 - d + 1)$
        }}

        \vspace{0.3em}
        {\small After $L_0$ sites of blocking, $\{A^{[L_0]}_\sigma\}_\sigma$
        spans $\MD$.\\[3pt]
        i.e.\ the blocked tensor is \textbf{injective}.}

        \vspace{0.4em}
        \textbf{Proof ingredients}:
        \begin{itemize}\itemsep0.08em\footnotesize
            \item Classical Wielandt inequality for nonneg.\ matrices
            \item Choi--Jamio{\l}kowski isomorphism
            \item Rank arguments on iterated Kraus operators
        \end{itemize}

        {\footnotesize\textcolor{dimwhite}{Tight example:
        $d=2$, $D=3$ gives $L_0 \leq 63$.}}
    \end{columns}

    \vspace{0.4em}
    {\footnotesize\textcolor{dimwhite}{%
    \textbf{For formalization}: we need either the Wielandt bound (hard but gives explicit $L_0$)
    or just the \emph{existence} of $L_0$ from primitivity
    (``$\mathcal{E}^n$ converges $\Rightarrow$ eventually full Kraus rank'').}}
\end{frame}

\begin{frame}{Periodic Tensors: The Off-Diagonal Decomposition}
    \vspace{-0.7em}

    {\small An irreducible tensor with period $m > 1$: $\mathcal{E}$ has
    eigenvalues $\{e^{2\pi i k/m}\}_{k=0}^{m-1}$ on the unit circle.}

    \vspace{0.2em}
    \begin{columns}[T]
        \column{0.55\textwidth}
        \textbf{Cyclic structure} \cite{Wolf2012Quantum,Cirac2021Matrix}:\\[2pt]
        {\small There exist orthogonal projectors $P_0, P_1, \ldots, P_{m-1}$
        with $\sum_u P_u = I$ such that:}
        \vspace{-0.3em}
        \[
            A^i = \sum_{u=0}^{m-1} P_u\, A^i\, P_{u+1\!\!\mod m}
        \]
        \vspace{-0.5em}
        {\small Each $A^i$ maps the $P_{u+1}$-subspace into the $P_u$-subspace.}

        \vspace{0.2em}
        \textbf{Consequence}: blocking $m$ sites gives
        \vspace{-0.3em}
        \[
            A^{[m]}_\sigma = P_0\, A^{[m]}_\sigma\, P_0 \;\oplus\; \cdots
        \]
        \vspace{-0.5em}
        {\small Each $P_u$-block of $\mathcal{E}^m$ is \textbf{primitive}
        $\Rightarrow$ reduce to normal case.}

        \column{0.42\textwidth}
        \begin{center}
        \begin{tikzpicture}[
            pnode/.style={circle, draw=leanblue, thick, fill=leanblue!15!black,
                          text=white, font=\scriptsize\bfseries, minimum size=0.7cm},

        ]
            \node[pnode] (p0) at (90:1.2) {$P_0$};

            \node[pnode] (p2) at (330:1.2) {$P_2$};

            \draw[parr] (p2) -- node[below,font=\tiny,text=dimwhite]{$A^i$} (p1);
            \draw[parr] (p0) -- node[right,font=\tiny,text=dimwhite]{$A^i$} (p2);

            \node[font=\tiny,text=dimwhite] at (0,-1.8) {Period $m=3$ example};
        \end{tikzpicture}
        \end{center}

        \vspace{0.1em}
        {\footnotesize The projectors $P_u$ are the
        \textbf{spectral projections}
        of the unitary eigenoperator
        $U$ with $\mathcal{E}(U) = e^{2\pi i/m}\,U$.}
    \end{columns}

    \vspace{0.15em}
    {\scriptsize\textcolor{dimwhite}{Finding the projectors $P_u$ requires
    the peripheral spectrum theory (Wolf Thm 6.6~\cite{Wolf2012Quantum}):
    $|\mu|=1$ eigenoperators are \emph{unitary}.}}
\end{frame}

\begin{frame}{Reducible Tensors: Block Decomposition}
    \vspace{-0.7em}

    {\small When $\mathcal{E}$ has a nontrivial invariant subspace: the
    tensor is reducible.}

    \vspace{0.2em}
    \textbf{Decomposition} \cite{Cirac2017Irreducible}:
    after a change of basis, every MPS tensor can be written as
    \vspace{-0.3em}
    \[
        A^i = \begin{pmatrix}
            \mu_1 A_1^i & * & * \\
            0 & \mu_2 A_2^i & * \\
            0 & 0 & \ddots
        \end{pmatrix}
        \qquad |\mu_j| \leq 1, \quad \text{each } \mathcal{E}_{A_j} \text{ irreducible}
    \]

    \vspace{-0.1em}
    \begin{columns}[T]
        \column{0.50\textwidth}
        \textbf{The irreducible form}:\\[2pt]
        {\small Discard off-diagonal (upper-triangular) parts ---
        they contribute to the state only at sub-leading order in $N$.}
        \vspace{-0.3em}
        \[
            |V_N(A)\rangle \;=\; \sum_{j:\,|\mu_j|=1}
            \mu_j^N |V_N(A_j)\rangle + O(c^N)
        \]
        \vspace{-0.5em}
        {\small for some $c < 1$.
        Only blocks with $|\mu_j| = 1$ survive.}

        \column{0.50\textwidth}
        \textbf{What the FT must handle}:
        \begin{itemize}\itemsep0.05em\small
            \item Multiple blocks $\to$ \textbf{block matching}:\\
                  {\scriptsize which blocks of $A$ correspond to which of $B$?}
            \item Phase freedom: $\mu_j$ can differ by a root of unity
            \item Permutation matrix $Z$: reorders blocks
            \item \textbf{Equal-or-orthogonal} decides matching:\\
                  $\rho(\mathcal{F}_{j,k}) < 1$ $\Rightarrow$ different,
                  $= 1$ $\Rightarrow$ same
        \end{itemize}
    \end{columns}

    \vspace{0.15em}
    {\scriptsize\textcolor{dimwhite}{Full canonical form:
    $B^i = Z\bigl(\bigoplus_j e^{i\theta_j} U_j A^i_{\pi(j)} U_j^\dagger\bigr)Z^\dagger$
    --- $Z$ permutation, $\pi$ block relabeling, $U_j$ unitaries,
    $\theta_j$ phases~\cite{Cirac2017Irreducible,Cirac2021Matrix}.}}
\end{frame}

\begin{frame}{Wolf Thm 6.6: The Peripheral Spectrum (the Axiom)}
    \vspace{-0.7em}

    {\small This single theorem is the \textcolor{softred}{key missing piece}.
    It characterizes what happens when $|\mu| = 1$ for a CP map.}

    \vspace{0.2em}
    \colorbox{cardbg}{\parbox{0.93\textwidth}{%
    \textbf{Theorem} (Wolf 6.6 \cite{Wolf2012Quantum}):
    {\small Let $\mathcal{E}$ be a unital CP map with
    $\mathcal{E}(X) = \mu X$, $|\mu| = 1$, $X \neq 0$. Then:}
    \begin{enumerate}\itemsep0.02em\small
        \item $X^\dagger X$ and $X X^\dagger$ are fixed points of $\mathcal{E}$
        \item If the fixed point is unique (up to scalar), then
            $X = c\, U$ for a \textbf{unitary} $U$
        \item The peripheral eigenvalues $\{\mu : |\mu|=1\}$
            form a \textbf{group} (closed under multiplication)
        \item For a primitive map: $\{1\}$ is the only peripheral eigenvalue
    \end{enumerate}}}

    \vspace{0.3em}
    \textbf{Proof chain} (each step nontrivial):

    \vspace{0.1em}
    \begin{center}

    \end{tikzpicture}
    \end{center}

    \vspace{0.1em}
    {\scriptsize\textcolor{dimwhite}{%
    \textbf{Kadison--Schwarz}: $\mathcal{E}(X)^\dagger \mathcal{E}(X) \leq \mathcal{E}(X^\dagger X)$ for CP unital $\mathcal{E}$.
    Equality when $|\mu|{=}1$: $\|X\|^2 = |\mu|^2 \|X\|^2 \leq \|\mathcal{E}(X^\dagger X)\| = \|X^\dagger X\| = \|X\|^2$.}}
\end{frame}

\begin{frame}{Moment Matching: How Multi-Block Separation Works}
    \vspace{-0.8em}

    {\small When both $A$ and $B$ are reducible with multiple blocks,
    we need to match them up.}

    \vspace{0.1em}
    \textbf{Setup}: $A^i = \bigoplus_{j=1}^r \mu_j A_j^i$, \;
    $B^i = \bigoplus_{k=1}^s \nu_k B_k^i$, \;
    $|V_N(A)\rangle = |V_N(B)\rangle$ for all $N$.

    \vspace{0.1em}
    \textbf{The overlap expansion}:
    \vspace{-0.5em}
    \[
        \langle V_N(A)\,|\,V_N(B)\rangle
        = \sum_{j,k} (\bar\mu_j \nu_k)^N \cdot
        \underbrace{\tr\bigl(\mathcal{F}_{j,k}^N(I)\bigr)}_{\text{mixed transfer}}
    \]

    \vspace{-0.4em}
    \begin{columns}[T]
        \column{0.52\textwidth}
        \textbf{Three cases for each $(j,k)$ pair}:
        \begin{itemize}\itemsep0em\small
            \item[\textcolor{leangreen}{$\checkmark$}] $\rho(\mathcal{F}_{j,k}) < 1$:
                overlap $\to 0$ exponentially\\
                {\tiny Block $j$ of $A$ and block $k$ of $B$ are unrelated}
            \item[\textcolor{leanblue}{$\circ$}] $\rho(\mathcal{F}_{j,k}) = 1$:
                overlap persists\\
                {\tiny Blocks are gauge-equivalent: $A_j \sim B_k$}
            \item[\textcolor{softred}{$\times$}] No other case possible
                ($\rho \leq 1$ always)
        \end{itemize}

        \column{0.48\textwidth}
        \textbf{Matching argument}:
        \begin{enumerate}\itemsep0em\small
            \item Phases $(\bar\mu_j\nu_k)^N$
                oscillate as $N$ varies
            \item \textbf{B\'ezout / Vandermonde}: distinct phases are
                linearly independent over $\mathbb{Z}$
            \item Surviving terms match individually
                $\Rightarrow$ bijection $\pi: j \mapsto k$
            \item Within each matched pair: apply the
                \textcolor{leangreen}{injective FT}
        \end{enumerate}
    \end{columns}

    \vspace{0.05em}
    {\scriptsize\textcolor{dimwhite}{%
    This is \texttt{sameMPV\textsubscript{2}\_implies\_perBlock\_sameMPV}:
    global state equality $\to$ per-block gauge equivalences.
    Needs spectral gap (axiom) + Vandermonde (algebraic).}}
\end{frame}
