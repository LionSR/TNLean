\begin{frame}{The Fundamental Theorem of Matrix Product States}
\vspace{-0.5em}

\begin{block}{Physical setup}
A \emph{matrix product state} (MPS) on a spin chain of length $N$ with
local dimension $d$ and bond dimension $D$ is a state whose coefficients are
\[
  c_{\sigma_1 \dots \sigma_N}
  = 
  \tr\!(A^{[\sigma_1]} A^{[\sigma_2]} \cdots A^{[\sigma_N]}),
  \qquad
  A^{[\sigma]} \in M_D(\mathbb{C}).
\]
\end{block}

\begin{block}{Fundamental Theorem \cite{Perez-Garcia2007Matrix}}
If two \textbf{injective} MPS tensors $A, B$ generate the same state family
for all system sizes $N$, then they are related by a \textbf{gauge transform}:
\[
  \boxed{ 
  \forall\,\sigma: 
  B^{[\sigma]} = X\, A^{[\sigma]} X^{-1},
  \qquad X \in \GL_D(\mathbb{C}).
  }
\]
\end{block}

\vspace{0.3em}
\textcolor{dimwhite}{\footnotesize%
This is the \emph{single-block} (injective) case.
The \emph{multi-block} case requires a canonical form decomposition.}
\end{frame}

\begin{frame}{Canonical Form and Block-Diagonal Structure}
\vspace{-0.5em}

Every MPS tensor can be brought into \textbf{canonical form}
\cite{Perez-Garcia2007Matrix,Cirac2021Matrix}:

\[
  A^{[\sigma]}
  = 
  \begin{pmatrix}
    \mu_1 A_1^{[\sigma]} & & \\
    & \mu_2 A_2^{[\sigma]} & \\
    & & \ddots \\
    & & & \mu_r A_r^{[\sigma]}
  \end{pmatrix}
\]

\vspace{-0.3em}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{itemize}
  \item $r$ blocks, bond dimensions $D_1, \dots, D_r$
  \item Each $A_k$ is \textbf{injective} (normal)
  \item $\mu_k \in \mathbb{C}^{\times}$ are \textbf{distinct} scalars
\end{itemize}

\column{0.5\textwidth}
\begin{center}

  \node[white,font=\footnotesize] at (4.25,0.75) {$\mu_3 A_3$};
\end{tikzpicture}
\end{center}
\end{columns}

\vspace{0.5em}
\textcolor{dimwhite}{\footnotesize%
In PGVWC, additionally: $\rho(\mathcal{E}_{A_k}) = 1$ (spectral radius normalization)
and $|\mu_1| > |\mu_2| > \cdots > |\mu_r|$ (strict ordering).}
\end{frame}

\begin{frame}{The Block Separation Question}
\vspace{-0.5em}

\begin{block}{Setup}
Two block-diagonal MPS tensors with the \emph{same} scaling factors $\mu_k$:
\begin{align*}
  A^{[\sigma]} &= \diag(\mu_1 A_1^{[\sigma]}, \dots, \mu_r A_r^{[\sigma]}), \\
  B^{[\sigma]} &= \diag(\mu_1 B_1^{[\sigma]}, \dots, \mu_r B_r^{[\sigma]}).
\end{align*}
\end{block}

\begin{block}{Question (\texttt{hSep} hypothesis)}
Does
\[
  \underbrace{%
    \forall\,N, \sigma: 
    \sum_{k=1}^r \mu_k^N \tr\!\left(\prod_j A_k^{[\sigma_j]}\right)
    = 
    \sum_{k=1}^r \mu_k^N \tr\!\left(\prod_j B_k^{[\sigma_j]}\right)
  }_{\text{SameMPV}_2 \text{ (global equality)}}
\]
imply \quad
$
  \underbrace{%
    \forall\,k,\,N, \sigma: 
    \tr\!\left(\prod_j A_k^{[\sigma_j]}\right) = \tr\!\left(\prod_j B_k^{[\sigma_j]}\right)
  }_{\text{per-block SameMPV}}
$ \textbf{?}
\end{block}

\vspace{0.3em}
\begin{center}
\textcolor{softred}{\large\textbf{No.}} \quad
\textcolor{dimwhite}{Not without additional hypotheses.}
\end{center}
\end{frame}

% в•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђ
\begin{frame}{Transfer Matrices and the Physical Meaning of $\mu_k$}
\vspace{-0.5em}

The \textbf{transfer map} (completely positive map) of block $k$:
\[
  \mathcal{E}_{A_k}(X)
  = 
  \sum_{\sigma=1}^{d} A_k^{[\sigma]}\, X\, (A_k^{[\sigma]})^{\!\dagger}
\]

\vspace{-0.3em}
\begin{columns}[T]
\column{0.55\textwidth}
\begin{itemize}
  \item $\mathcal{E}_{A_k}$ is a \textbf{quantum channel} (after normalization)
  \item Spectral radius $\rho(\mathcal{E}_{A_k})$ controls correlation decay
  \item The $\mu_k$ are eigenvalues of the \emph{full} transfer map
  \item $|\mu_k|^2 = \rho(\mathcal{E}_{A_k})$ in canonical form \cite{Cirac2021Matrix}
\end{itemize}

\column{0.45\textwidth}
\begin{block}{\small Correlation functions}
{\small
For large $N$, two-point correlations decay as
\[
  \langle O_i O_j \rangle - \langle O_i\rangle\langle O_j\rangle
  \sim 
  \sum_{k\ge 2} c_k \left|\frac{\mu_k}{\mu_1}\right|^{2|i-j|}
\]
Each $|\mu_k/\mu_1|^2 < 1$ gives a \textbf{correlation length} $\xi_k$.
}
\end{block}
\end{columns}

\vspace{0.5em}
\textcolor{amber}{\textbf{Key point:}} The $\mu_k$ encode the \emph{physical correlation structure}.
Block separation asks whether observing identical correlations at \emph{all} scales forces
per-block equality.
\end{frame}

% в•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђ
\begin{frame}[fragile]{What We Proved: The Repeated-Word Trace Identity}
\vspace{-0.5em}

\begin{block}{Theorem (\texttt{sameMPV\textsubscript{2}\_repeated\_word}, sorry-free)}
For any word $w$ of length $M$ and repetition count $L$:
\[
  \boxed{ 
  \sum_{k=1}^r (\mu_k)^{ML} \cdot
  \left[\tr(T_k^L) - \tr(U_k^L)\right] = 0
  }
\]
where $T_k = A_k^{[w_1]} \cdots A_k^{[w_M]}$ and
$U_k = B_k^{[w_1]} \cdots B_k^{[w_M]}$.
\end{block}

\vspace{-0.3em}
\begin{lstlisting}
theorem sameMPV2_repeated_word
    (*@$\mu$@*) : Fin r (*@$\to$@*) (*@$\mathbb{C}$@*))
    (A B : (k : Fin r) (*@$\to$@*) MPSTensor d (dim k))
    (hSame : SameMPV2 (toTensorFromBlocks (*@$\mu$@*) A)
                       (toTensorFromBlocks (*@$\mu$@*) B))
    (w : List (Fin d)) (L : (*@$\mathbb{N}$@*)) :
    (*@$\sum$@*) k, ((*@$\mu$@*) k) ^ (w.length * L) *
      (Matrix.trace ((evalWord (A k) w) ^ L) -
       Matrix.trace ((evalWord (B k) w) ^ L)) = 0
\end{lstlisting}

\vspace{0.2em}
\textcolor{dimwhite}{\footnotesize%
Proof: apply SameMPV\textsubscript{2} to the concatenated word $w^L = \underbrace{w \cdot w \cdots w}_{L}$,
then use \texttt{evalWord\_flatten\_replicate} to factor the product.}
\end{frame}

%  SLIDE 7 - Why Vandermonde Fails
% в•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђ
\begin{frame}{Why Na\"ive Vandermonde Separation Fails}
\vspace{-0.5em}

The repeated-word identity gives, for each fixed $w$ and varying $L$:
\[
  \sum_{k=1}^r \underbrace{(\mu_k^{|w|})^L}_{\text{Vandermonde node}^L}
  \cdot
  \underbrace{[\tr(T_k^L) - \tr(U_k^L)]}_{\Delta_k(L)}
  = 0
  \qquad \forall\, L \ge 0
\]

\vspace{0.3em}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{block}{\small If $\Delta_k$ were constant in $L$\ldots}
{\small
$\sum_k \lambda_k^L \cdot c_k = 0$ for all $L$
with distinct $\lambda_k = \mu_k^{|w|}$

$\Longrightarrow$ Vandermonde invertibility gives $c_k = 0$ for all $k$.

\textcolor{leangreen}{This is \texttt{Matrix.eq\_zero\_of\_forall\_pow\_sum\_mul\_pow\_eq\_zero} in Mathlib.}
}
\end{block}

\column{0.5\textwidth}
\begin{block}{\small But $\Delta_k$ depends on $L$!}
{\small
$\Delta_k(L) = \tr(T_k^L) - \tr(U_k^L)$

is a \textbf{polynomial in the eigenvalues} of $T_k$ and $U_k$, varying with $L$.

The ``coefficients'' are \emph{entangled} with the Vandermonde exponent.

\textcolor{softred}{Cannot separate.}
}
\end{block}
\end{columns}

\vspace{0.5em}
\textcolor{amber}{\textbf{The deeper issue:}} We have $r$ unknowns (blocks), but
$\tr(T_k^L)$ gives us power sums of eigenvalues of $T_k$ --- these are coupled
to $L$ through Newton's identities, not independent of it.
\end{frame}

\begin{frame}{The Counterexample: Construction}
\vspace{-0.5em}

\]

\]

  A^{[\sigma_2]} = \begin{pmatrix} 2 & 0 \\ 0 & 6 \end{pmatrix}

\end{block}

\]
\vspace{-0.5em}

\begin{frame}{The Counterexample: Verification}
\vspace{-0.5em}

\textbf{Claim 1: SameMPV$_2$ holds.}\quad
For any word $w = (\sigma_{j_1}, \dots, \sigma_{j_N})$:
\begin{align*}
  &\text{For } A: \quad
  \mu_1^N \prod_\ell A_1(\sigma_{j_\ell}) + \mu_2^N \prod_\ell A_2(\sigma_{j_\ell})
  = 1^N \cdot \prod_\ell A_1(\sigma_{j_\ell}) + 2^N \cdot \prod_\ell A_2(\sigma_{j_\ell}) \\
  &\text{For } B: \quad
  \mu_1^N \prod_\ell B_1(\sigma_{j_\ell}) + \mu_2^N \prod_\ell B_2(\sigma_{j_\ell})
\end{align*}

\vspace{-0.8em}
Since $\mu_1 B_1(\sigma_i) = \mu_2 A_2(\sigma_i)$ and
$\mu_2 B_2(\sigma_i) = \mu_1 A_1(\sigma_i)$:
\[
  \mu_1^N \prod_\ell B_1(\sigma_{j_\ell})
  = \prod_\ell [\mu_1 B_1(\sigma_{j_\ell})]
  = \prod_\ell [\mu_2 A_2(\sigma_{j_\ell})]
  = \mu_2^N \prod_\ell A_2(\sigma_{j_\ell})
\]

\vspace{-0.2em}
Similarly for the other term. So the $B$-sum equals
$\mu_2^N \prod A_2 + \mu_1^N \prod A_1$ --- same as $A$. \textcolor{leangreen}{\checkmark}

\vspace{0.5em}
\textbf{Claim 2: Per-block SameMPV fails.}\quad
\[
  \tr(A_1(\sigma_1)) = 4 \neq  2 = \tr(B_1(\sigma_1))
  \qquad \textcolor{softred}{\boldsymbol{\times}}
\]

\vspace{0.3em}
\begin{center}
\fcolorbox{softred}{cardbg}{\parbox{0.8\textwidth}{\centering
\textcolor{softred}{\textbf{Conclusion:}} \texttt{SameMPV\textsubscript{2}} with injective nonzero $\mu$
does \textbf{not} imply per-block \texttt{SameMPV}.}}
\end{center}
\end{frame}

% в•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђ
\begin{frame}{The Block Swap Symmetry}
\vspace{-0.5em}

\begin{block}{General construction}
Given any permutation $\pi \in S_r$ and block tensors $\{A_k\}$, define:
\[
  B_k^{[\sigma]}
  := 
  \frac{\mu_{\pi(k)}}{\mu_k}  A_{\pi(k)}^{[\sigma]}
  \qquad \forall\, k,\, \sigma.
\]
Then $\mu_k \cdot B_k^{[\sigma]} = \mu_{\pi(k)} \cdot A_{\pi(k)}^{[\sigma]}$, and
the block-diagonal matrices for $B$ are a \textbf{permutation} of those for $A$.
\end{block}

\vspace{0.3em}

\textbf{Why doesn't injectivity of $\mu$ help?}

\begin{itemize}
  \item Injectivity means $\mu_j \neq \mu_k$ for $j \neq k$
        --- prevents $\mu_k = \mu_{\pi(k)}$ for non-trivial $\pi$.
  \item But the \emph{scaled} products $\mu_k \cdot B_k(\sigma) = \mu_{\pi(k)} \cdot A_{\pi(k)}(\sigma)$
        can still match, because $B_k$ absorbs the ratio $\mu_{\pi(k)}/\mu_k$.
  \item The SameMPV$_2$ condition only sees $\sum_k \mu_k^N \cdot \text{(block trace)}_k$
        --- a \textbf{symmetric function} of the scaled blocks.
\end{itemize}

\vspace{0.3em}
\textcolor{amber}{\textbf{Analogy:}} Given $\{a + b = 5,  2a + 3b = 13\}$, we can solve
for $a, b$ uniquely. But $\{a + b,  a \cdot b\}$ only determines $\{a, b\}$
\emph{as a set} --- the assignment to names is lost.
\end{frame}

\begin{frame}{What PGVWC Actually Assumes}
\vspace{-0.5em}

In the original paper \cite{Perez-Garcia2007Matrix} and the review \cite{Cirac2021Matrix},
the canonical form has \textbf{two additional structural properties}:

\vspace{0.3em}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{block}{\small 1.\ Spectral radius normalization}
Each block tensor $A_k$ satisfies
\[
  \rho(\mathcal{E}_{A_k}) = 1
\]
i.e., the transfer map is normalized to have spectral radius 1.

\vspace{0.3em}
\textcolor{dimwhite}{\footnotesize%
This is the ``DS-gauge'' normalization:
$\sum_\sigma A_k^{[\sigma]} \Lambda_k (A_k^{[\sigma]})^\dagger = \Lambda_k$
for some positive definite $\Lambda_k$.
}
\end{block}

\column{0.5\textwidth}
\begin{block}{\small 2.\ Strict modulus ordering}
The scalars satisfy
\[
  |\mu_1| > |\mu_2| > \cdots > |\mu_r|
\]
In particular, $|\mu_j| \neq |\mu_k|$ for $j \neq k$.

\vspace{0.3em}
\textcolor{dimwhite}{\footnotesize%
This is stronger than injectivity of $\mu$:
it requires distinct \emph{moduli}, not just distinct values.
The $\mu_k$ are eigenvalues of the full transfer map,
ordered by magnitude.
}
\end{block}
\end{columns}

\vspace{0.5em}
\begin{center}
\fcolorbox{amber}{cardbg}{\parbox{0.85\textwidth}{\centering
\textcolor{amber}{\textbf{Our counterexample shows these are not mere conveniences ---}}\\
\textcolor{amber}{\textbf{they are necessary for block separation to hold.}}
}}
\end{center}
\end{frame}

\begin{frame}{How Spectral Normalization Prevents Block Swaps}
\vspace{-0.5em}

Suppose $B_k(\sigma) = (\mu_{\pi(k)}/\mu_k)\cdot A_{\pi(k)}(\sigma)$ for some non-trivial $\pi$.

\vspace{0.3em}
\textbf{Transfer map of $B_k$:}
\[
  \mathcal{E}_{B_k}(X)
  = \sum_\sigma B_k^{[\sigma]} X (B_k^{[\sigma]})^\dagger
  = \left|\frac{\mu_{\pi(k)}}{\mu_k}\right|^2
    \sum_\sigma A_{\pi(k)}^{[\sigma]} X (A_{\pi(k)}^{[\sigma]})^\dagger
  = \left|\frac{\mu_{\pi(k)}}{\mu_k}\right|^2 \mathcal{E}_{A_{\pi(k)}}(X)
\]

\vspace{0.3em}
\textbf{Spectral radius:}
\[
  \rho(\mathcal{E}_{B_k})
  = \left|\frac{\mu_{\pi(k)}}{\mu_k}\right|^2 \cdot
    \underbrace{\rho(\mathcal{E}_{A_{\pi(k)}})}_{= 1}
  = \left|\frac{\mu_{\pi(k)}}{\mu_k}\right|^2
\]

\vspace{0.3em}
For $B$ to also be in canonical form ($\rho(\mathcal{E}_{B_k}) = 1$), we need
$|\mu_{\pi(k)}| = |\mu_k|$ for all $k$.

\vspace{0.5em}
\begin{center}
\begin{tikzpicture}
  \node[draw=leangreen, thick, rounded corners, fill=cardbg, text=white,
        minimum width=12cm, minimum height=1.2cm, align=center] {
    With $|\mu_1| > |\mu_2| > \cdots > |\mu_r|$ (strict ordering),\\
    $|\mu_{\pi(k)}| = |\mu_k|$ forces $\pi = \mathrm{id}$. \quad
    \textcolor{leangreen}{\textbf{Block swap is impossible.}}
  };
\end{tikzpicture}
\end{center}
\end{frame}
