\begin{frame}{Step 1: The Transfer Operator}

For each block $k$ with matrices $\{A_k^\sigma\}_\sigma$, define:
\[
E_k(X) =  \sum_{\sigma=1}^{d} A_k^\sigma \,X\, (A_k^\sigma)^\dagger
\qquad\qquad E_k : M_{D_k}(\mathbb{C}) \to M_{D_k}(\mathbb{C})
\]

\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Key properties:}
\begin{itemize}
  \item \proved{Positive: $X \geq 0 \implies E_k(X) \geq 0$}
  \item \proved{Linear map on $D_k^2$-dimensional space}
  \item Completely positive (Kraus form)
  \item If $\sum_\sigma (A_k^\sigma)^\dagger A_k^\sigma = I$: trace-preserving
\end{itemize}

\vspace{0.2em}
\textbf{Why it matters:}
\begin{itemize}\small\itemsep0pt
  \item Powers of $E_k$ encode the MPV generating function
  \item $\tr(E_k^N(\rho))$ expands as a sum over all $\sigma$-configurations
\end{itemize}
\end{column}

\begin{frame}{Steps 2--3: Complete Positivity and Irreducibility}

\textbf{Step 2: Complete positivity.}\quad
$E_k$ has Kraus form $\{A_k^\sigma\}_\sigma$, so it is automatically \alert{completely positive (CP)}:
\[
E_k \otimes \mathrm{id}_n : M_{D_k \cdot n}(\mathbb{C}) \to M_{D_k \cdot n}(\mathbb{C}) \quad\text{preserves}\quad X \geq 0.
\]
Stronger than positivity: $E_k$ preserves entanglement structure.

\vspace{0.6em}
\textbf{Step 3: Irreducibility.}\quad
When the block tensor $A_k$ is \emph{injective} (its matrices span $M_{D_k}(\mathbb{C})$), the transfer operator $E_k$ is \alert{irreducible} as a positive map:

\begin{block}{Irreducibility of CP maps}
The only subspace $V \subseteq \mathbb{C}^{D_k}$ with $A_k^\sigma(V) \subseteq V$ for all $\sigma$ is $V = \{0\}$ or $V = \mathbb{C}^{D_k}$.

Equivalently: the only $E_k$-invariant faces of the PSD cone are $\{0\}$ and the full cone.
\end{block}

\vspace{0.3em}
\textcolor{dimwhite}{\footnotesize Mathlib has \texttt{Matrix.IsIrreducible} (definition only), not the eigenvalue consequences.}
\end{frame}

%======================================================================
% SLIDE 6 - Step 4: Quantum Perron-Frobenius
%======================================================================
\begin{frame}{Step 4: The Quantum Perron--Frobenius Theorem}

\begin{block}{Theorem (Evans--H{\o}egh-Krohn \cite{Evans1978Spectral}; Wolf \cite{Wolf2012Quantum})}
Let $E : M_D(\mathbb{C}) \to M_D(\mathbb{C})$ be an \alert{irreducible completely positive} map. Then:
\begin{enumerate}\itemsep0pt
  \item The spectral radius $\rho(E) > 0$ is an eigenvalue.
  \item The eigenspace for $\rho(E)$ is \textbf{one-dimensional}.
  \item The unique eigenvector (up to scaling) is \textbf{positive definite}: $\Lambda > 0$.
  \item No other eigenvalue of modulus $\rho(E)$ has a PSD eigenvector.
\end{enumerate}
\end{block}

\vspace{0.2em}
\textbf{For our transfer operator $E_k$:}
\begin{itemize}\itemsep0pt
  \item In canonical form, $E_k(I) = I$ (trace-preserving), so $\rho(E_k) = 1$
  \item The unique fixed point $\Lambda_k > 0$ with $E_k(\Lambda_k) = \Lambda_k$
  \item \alert{All other eigenvalues satisfy $|\lambda| < 1$} (spectral gap)
\end{itemize}

\vspace{0.2em}
\begin{center}
\textcolor{softred}{Not formally verified in any proof assistant (Lean, Isabelle, or Coq).}
\end{center}
\end{frame}

%======================================================================
% SLIDE 7 - Steps 5-6: Spectral Gap and Block Separation
%======================================================================
\begin{frame}{Steps 5--6: Spectral Gap $\implies$ Block Separation}

\textbf{Step 5: Spectral gap.}\quad After normalizing so $\rho(E_k) = 1$:
\[
E_k^N(X) \xrightarrow{N\to\infty}  \tr(X) \cdot \Lambda_k
\]
The convergence rate is $|\lambda_2|^N$ where $|\lambda_2| < 1$ is the second-largest eigenvalue modulus.

\vspace{0.5em}
\textbf{Step 6: Block separation.}\quad The SameMPV$_2$ equation says:
\[
\sum_k \mu_k^N \cdot \underbrace{[\tr(A_k^{\sigma_1}\cdots A_k^{\sigma_N}) - \tr(B_k^{\sigma_1}\cdots B_k^{\sigma_N})]}_{\text{involves } E_k^N}= 0 \quad \forall\, N, \sigma
\]

\begin{enumerate}\small
  \item For distinct $|\mu_k|$: divide by $\mu_1^N$, take $N \to \infty$ $\implies$ block 1 separates
  \item By induction on $k$: all blocks separate
  \item For equal $|\mu_k|$ with different phases: secondary Vandermonde on the unit circle
\end{enumerate}

\vspace{0.3em}
{\footnotesize \textcolor{dimwhite}{The spectral gap ensures convergence of $E_k^N$, providing fixed coefficients for Vandermonde despite the $N$--$\sigma$ coupling.}}
\end{frame}

\begin{frame}{The Key Insight: Choi--Jamio\l kowski Bridge}

\textbf{The vectorization identity} (already in Mathlib):\quad
$\mathrm{vec}(AXB^T) = (B \otimes A) \cdot \mathrm{vec}(X)$

\vspace{0.2em}
Applying to the transfer operator:
\vspace{-0.3em}
\[
\mathrm{vec}(E_A(X)) = \underbrace{\left(\sum_i \overline{A_i} \otimes A_i\right)}_{T_A =  \text{``Choi matrix''}} \cdot \mathrm{vec}(X)
\]

\vspace{-0.3em}
\begin{columns}[T]
\begin{column}{0.45\textwidth}
\textbf{Properties of $T_A$:}
\begin{itemize}\itemsep0pt
  \item $D^2 \times D^2$ matrix
  \item $T_A$ is \alert{positive semidefinite} (sum of rank-1 PSD terms)
  \item Eigenvalues of $T_A$ = eigenvalues of $E_A$
  \item $T_A$ irreducible $\iff$ $E_A$ irreducible
\end{itemize}
\end{column}

\begin{frame}{The Hardest Step: The Irreducibility Bridge}

\textbf{We need:}\quad Injectivity of $A_k$ $\implies$ $T_A = \sum_i \overline{A_i} \otimes A_i$ is an irreducible nonneg matrix.

\vspace{0.3em}
\textbf{Why this is hard:}
\begin{itemize}\itemsep1pt
  \item ``Irreducible'' for CP maps: no nontrivial $E_A$-invariant face of the PSD cone
  \item ``Irreducible'' for nonneg matrices: no nontrivial invariant index subset (graph-theoretic)
  \item These are \emph{different concepts} --- connecting them requires careful analysis
\end{itemize}

\vspace{0.2em}
\textbf{Two approaches:}

\begin{columns}[T]
\begin{column}{0.47\textwidth}
\textbf{A: Via Choi matrix structure}
\begin{itemize}\small\itemsep1pt
  \item $T_A$ is PSD $\implies$ entries $\geq 0$?
  \item \textcolor{softred}{Not automatic!} PSD $\not\Rightarrow$ nonneg
  \item Need nonneg in a \emph{suitable} basis (e.g.\ rank-1 PSD matrices)
  \item Estimated: \textbf{2--4 weeks}
\end{itemize}
\end{column}
\begin{column}{0.47\textwidth}
\textbf{B: Direct quantum PF (bypass)}
\begin{itemize}\small\itemsep1pt
  \item Prove quantum PF directly using:
    \begin{itemize}\scriptsize\itemsep0pt
      \item Compactness of state space
      \item Brouwer FPT (in Mathlib)
      \item Irreducibility $\implies$ interior fixed pt
    \end{itemize}
  \item \textcolor{leangreen}{Avoids classical PF bridge entirely}
  \item Estimated: \textbf{3--4 weeks}
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Approach B: Direct Quantum PF (Bypass Classical)}

Skip the classical $\to$ quantum bridge entirely. Prove quantum PF from first principles:

\vspace{0.3em}
\begin{columns}[T]

\begin{enumerate}\small\itemsep2pt
  \item \textbf{Fixed point existence} (1 wk, 150--250 LoC)
    \begin{itemize}\scriptsize\itemsep0pt
      \item $\mathcal{S} = \{\rho \geq 0 : \tr(\rho)=1\}$ is compact convex
      \item Brouwer FPT $\implies \exists\,\Lambda$ with $E_A(\Lambda)=\Lambda$
    \end{itemize}
  \item \textbf{Positive definiteness} (1 wk, 200--300 LoC)
    \begin{itemize}\scriptsize\itemsep0pt
      \item Irreducibility $\implies$ $\Lambda > 0$ (strictly PD)
      \item If $\ker(\Lambda) \neq \{0\}$, it would be $E_A$-invariant
    \end{itemize}
  \item \textbf{Uniqueness} (1 wk, 150--300 LoC)
    \begin{itemize}\scriptsize\itemsep0pt
      \item Hilbert projective metric: $E_A$ is contraction
      \item $\implies$ unique fixed point
    \end{itemize}
  \item \textbf{Spectral gap} (1 wk, 100--200 LoC)
    \begin{itemize}\scriptsize\itemsep0pt
      \item Gelfand formula $+$ uniqueness $\implies |\lambda| < 1$
    \end{itemize}
\end{enumerate}
