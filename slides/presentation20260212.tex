\begin{frame}{The Fundamental Theorem of MPS}
    \vspace{-0.5em}
    \begin{columns}[T]
        \column{0.55\textwidth}
        \textbf{Setup.} Physical index $i\in\{1,\dots,d\}$, bond dimension~$D$.
        Assign a $D\times D$ matrix $A^i$ to each physical index.
        \vspace{0.3em}

        \textbf{The state} (periodic boundary conditions):
        \[
            \ket{V^{(N)}} = \sum_{i_1,\dots,i_N} \tr\!(A^{i_1}\!\cdots A^{i_N}) \ket{i_1\cdots i_N}
        \]

        \textbf{Injectivity}: $\spn\{A^i\}_{i=1}^d = \MD$.

        \vspace{0.3em}
        \textbf{Gauge freedom:}
        $B^i = X A^i X^{-1}$ with $X\in\GL(D,\mathbb{C})$
        leaves every coefficient invariant (trace cyclicity).

        \column{0.43\textwidth}
        \begin{center}

    \vspace{0.3em}
    \begin{block}{Fundamental Theorem \cite{Perez-Garcia2007Matrix} \cite{Cirac2021Matrix}}
        If $A$ is injective and $A,B$ generate the \emph{same state family} for all $N$,
        then $B^i = X\, A^i\, X^{-1}$ for some $X\in\GL(D,\mathbb{C})$.
    \end{block}

abbrev MPSTensor (d D : ℕ) :=
  Fin d → Matrix (Fin D) (Fin D) ℂ
\end{lstlisting}
        \vspace{-0.3em}
        \begin{lstlisting}
def IsInjective (A : MPSTensor d D) :=
  Submodule.span ℂ (Set.range A) = ⊤
\end{lstlisting}
        \vspace{-0.3em}
        \begin{lstlisting}
def SameMPV (A B : MPSTensor d D) :=
  ∀ N σ, mpv A σ = mpv B σ
\end{lstlisting}
        \vspace{-0.3em}
        \begin{lstlisting}
def GaugeEquiv (A B : MPSTensor d D) :=
  ∃ X : GL (Fin D) ℂ,
    ∀ i, B i = X * A i * X⁻¹
\end{lstlisting}
    \end{columns}

\begin{frame}{Single-Block Proof\quad\textcolor{leangreen}{FULLY PROVED \checkmark}}
    \vspace{-0.4em}

    \textbf{The key object: $T\colon \MD \to \MD$.}\quad
    {\small Injectivity means $\{A^i\}$ span all of~$\MD$.
        Define $T$ by $T(A^i):= B^i$ and extend linearly.
        This is well-defined and unique \emph{because} $\{A^i\}$ span.}

    \vspace{0.2em}
    \begin{center}
        \begin{tikzpicture}[xscale=0.88,
            sbox/.style={rectangle,rounded corners,draw=white,text=white,
                    font=\scriptsize\bfseries,minimum width=1.8cm,minimum height=0.95cm,align=center},
            sarr/.style={-{Stealth[scale=0.8]},white,thick},
            slab/.style={font=\tiny,text=amber,align=center,above,yshift=-1pt}
            ]
            \node[sbox,fill=gray!50!black] (s0) at (0,0) {Same state\\$\forall\,N,\sigma$};
            \node[sbox,fill=leanblue!40!black] (s1) at (2.7,0) {Unique\\linear $T$};
            \node[sbox,fill=leanblue!60!black] (s2) at (5.4,0) {$T$ is\\multiplicative};
            \node[sbox,fill=leangreen!40!black] (s3) at (8.1,0) {$T$ is an\\automorphism};
            \node[sbox,fill=leangreen!60!black] (s4) at (10.8,0) {$B^i\!=\!XA^iX^{\!-\!1}$};
            \draw[sarr] (s0) -- node[slab]{spanning} (s1);
            \draw[sarr] (s1) -- node[slab]{trace ids} (s2);
            \draw[sarr] (s2) -- node[slab]{simplicity} (s3);
            \draw[sarr] (s3) -- node[slab]{Skolem--N.} (s4);
        \end{tikzpicture}
    \end{center}

    \vspace{0.1em}
    \begin{enumerate}\itemsep0.1em
        \item {\small \textbf{Spanning $\boldsymbol{\Rightarrow}$ linear map.}
              $T(A^i):=B^i$, extended to all of~$\MD$ by linearity.
              SameMPV ensures consistency (via the trace pairing).}
        \item {\small \textbf{Trace identities $\boldsymbol{\Rightarrow}$ multiplicativity.}
              $\tr(A^i A^j A^k) = \tr(B^i B^j B^k)$ forces
              $T(MN)=T(M)\,T(N)$.}
        \item {\small \textbf{Simplicity $\boldsymbol{\Rightarrow}$ bijectivity.}
              $\ker T$ is a two-sided ideal of $\MD$, which is simple, so $\ker T = 0$.}
        \item {\small \textbf{Skolem--Noether $\boldsymbol{\Rightarrow}$ conjugation.}
              Every automorphism of $\MD$ is inner: $T(M) = XMX^{-1}$.}
    \end{enumerate}

    \vspace{0.1em}
    \begin{center}
        {\small \textcolor{leangreen}{\textbf{Zero sorry.}} Pure algebra --- our own proof strategy, chosen because
            Mathlib already has simplicity and Skolem--Noether.}
    \end{center}
\end{frame}

\begin{frame}{Multi-Block: Canonical Form}
    \vspace{-0.5em}
    \begin{columns}[T]
        \column{0.45\textwidth}
        \begin{center}
            \begin{tikzpicture}[scale=0.85]
                \draw[white,thick] (0,0) rectangle (4,4);
                \fill[leangreen!30!black] (0.1,2.9) rectangle (1.4,3.9);
                \fill[leanblue!30!black] (1.5,1.6) rectangle (2.5,2.8);
                \fill[amber!30!black] (2.6,0.1) rectangle (3.9,1.5);
                \node[white,font=\small\bfseries] at (0.75,3.4) {$\mu_1 A_1^i$};
                \node[white,font=\small\bfseries] at (2.0,2.2) {$\mu_2 A_2^i$};
                \node[white,font=\small\bfseries] at (3.25,0.8) {$\mu_3 A_3^i$};
                \node[gray,font=\scriptsize] at (2.5,3.4) {$0$};
                \node[gray,font=\scriptsize] at (0.75,0.8) {$0$};
                \node[white,font=\bfseries] at (2,-0.5) {Block-diagonal};
            \end{tikzpicture}
        \end{center}

        \column{0.52\textwidth}
        \textbf{Canonical form} \cite{Cirac2021Matrix}:
        $T^{(i)} = \bigoplus_k \mu_k\, A_k^{(i)}$

        \vspace{0.2em}
        Each block $A_k$ is injective, $\mu_k\in\mathbb{C}$ distinct.

        \vspace{0.3em}
        Block MPVs contribute independently:
        \begin{equation*}
            V^{(N)} = \textstyle\sum_k \mu_k^N \cdot V_k^{(N)}
        \end{equation*}

        \vspace{0.3em}
        \textbf{Multi-block theorem} \cite{Cirac2017Annals} \cite{Cirac2017Irreducible}:
        Same state $\Rightarrow$ blocks match pairwise up to gauge and phase.
    \end{columns}

    \vspace{0.4em}
    {\small\textbf{Informal proof outline:}}
    \begin{enumerate}\itemsep0.1em
        \item {\small \textbf{Decompose}: MPV is a sum over blocks, weighted by $\mu_k^N$.}
        \item {\small \textbf{Separate}: distinct $\mu_k$ $\Rightarrow$ blocks decouple (the hard step --- see later slides).}
        \item {\small \textbf{Match}: per-block SameMPV; apply single-block theorem to each.}
        \item {\small \textbf{Assemble}: per-block gauge transforms $\Rightarrow$ global block-diagonal conjugation.}
    \end{enumerate}
\end{frame}

% SLIDE 10 - The Physics Proof of Block Separation
%======================================================================
\begin{frame}{The Physics Proof of Block Separation \cite{Cirac2017Annals} \cite{Cirac2021Matrix}}
    \vspace{-0.3em}
    \textbf{Step 1.}  Define transfer operators:
    $ \mathcal{E}_A(X) = \sum_i A^{(i)} X A^{(i)\dagger}$,\quad
    $\mathcal{F}_{AB}(X) = \sum_i A^{(i)} X B^{(i)\dagger}$.

    \vspace{0.2em}
    \textbf{Step 2.}  Injectivity of $A$ $\Rightarrow$ $\mathcal{E}_A$ is \emph{irreducible} CP:
    no nontrivial projection $P$ satisfies $P\,\mathcal{E}_A(PXP)\,P = \mathcal{E}_A(PXP)$.

    \vspace{0.2em}
    \textbf{Step 3.}  \textcolor{amber}{Quantum Perron--Frobenius} \cite{Evans1978Spectral}:
    irreducible CP $\Rightarrow$ primitive: unique positive definite fixed point, exponential convergence.

    \vspace{0.2em}
    \textbf{Step 4.}  For the \emph{mixed transfer operator} $\mathcal{F}_{AB}(X)=\sum_i A^i X {B^i}^\dagger$ across different blocks:
    no rank-1 fixed point $\Rightarrow$ spectral radius $< 1$.

    \vspace{0.2em}
    \textbf{Step 5.}  $\rho < 1$ $\Rightarrow$ $\mathcal{F}_{AB}^n \to 0$ $\Rightarrow$ cross-correlations vanish.

    \vspace{0.2em}
    \textbf{Step 6.}  MPV identity $+$ diagonal terms persist $+$ off-diagonal decay
    $\Rightarrow$ blocks separate.

    \vspace{0.4em}
    \begin{center}
        {\small Steps 1--2: fully proved \textcolor{leangreen}{\checkmark}\quad
            Steps 3--4: \textcolor{softred}{the core gap}\quad
            Steps 5--6: fully proved \textcolor{leangreen}{\checkmark}}
    \end{center}
\end{frame}

\begin{frame}{The Remaining Gap --- Quantum Perron--Frobenius}
    \vspace{-0.3em}

    \begin{columns}[T]
        \column{0.48\textwidth}
        \small
        \textbf{Quantum PF} \cite{Evans1978Spectral}: for an irreducible CP map $\mathcal{E}$,
        \begin{itemize}\itemsep0.12em
            \item unique positive definite fixed point $\Lambda$
            \item spectral gap: all other eigenvalues $|\lambda|<1$
        \end{itemize}

        \vspace{0.3em}
        This gives Steps 3--4: spectral gap for self-transfer operators
        and $\rho(\mathcal{F}_{AB})<1$ for mixed transfer operators \cite{Fernandez2015Frustration}.

        \vspace{0.4em}
        \textbf{\textcolor{softred}{Not formalized in any proof assistant.}}

        \vspace{0.2em}
        Classical PF (real nonneg matrices) exists in Lean \cite{Cipollina2025Formalized},
        but does not apply: the transfer matrix is complex and not Hermitian.

        \column{0.48\textwidth}
        \small
        \textbf{What we have built toward it:}
        \begin{itemize}\itemsep0.1em
            \item[\textcolor{leangreen}{\checkmark}] CP map definitions, irreducibility
            \item[\textcolor{leangreen}{\checkmark}] Injectivity $\Rightarrow$ irreducible CP
            \item[\textcolor{leangreen}{\checkmark}] $\rho<1 \Rightarrow \mathcal{F}^n \to 0$
            \item[\textcolor{leangreen}{\checkmark}] Full chain from spectral gap to block separation
            \item[\textcolor{softred}{$\times$}] Quantum PF itself (Steps 3--4)
        \end{itemize}

        \vspace{0.4em}
        Everything above and below the gap is proved.
        Estimated $\sim$2,000--3,000 additional lines to close it.
    \end{columns}
\end{frame}

% SLIDE 13 - Key Result: Injectivity ⟹ Irreducible CP

\begin{frame}{Key Result: Injectivity $\Rightarrow$ Irreducible CP\quad\textcolor{leangreen}{\checkmark}}
    \vspace{-0.3em}
    {\small Step 2 of the physics proof, fully proved ($\sim$80 lines, no sorry).}

    \vspace{0.3em}
    \textbf{Theorem.}  If $\spn\{A^{(i)}\} = \MD$, then $\mathcal{E}_A$ is irreducible.

    \vspace{0.3em}
    \textbf{Proof.}  Suppose projection $P$ is invariant: $P\,\mathcal{E}_A(PXP)\,P = \mathcal{E}_A(PXP)\ \forall\, X$.
    \begin{enumerate}\itemsep0.15em
        \item {\small Expand: $\sum_i (PA^{(i)}P - A^{(i)}P)\,X\,(PA^{(i)}P - A^{(i)}P)^\dagger = 0$ for all $X$.}
        \item {\small Each summand is PSD $\Rightarrow$ each is zero $\Rightarrow$ $A^{(i)}P = PA^{(i)}P$.}
        \item {\small Adjoints: $PA^{(i)} = A^{(i)}P$ $\Rightarrow$ $P$ commutes with $\spn\{A^{(i)}\} = \MD$.}
        \item {\small $P \in Z(\MD) = \mathbb{C}\cdot I$ and $P^2 = P = P^\dagger$ $\Rightarrow$ $P = 0$ or $I$. \qed}
    \end{enumerate}
\end{frame}

\begin{frame}[fragile]{Backup: Lean Code --- Single-Block FT}
    \vspace{-0.6em}
    \begin{lstlisting}
theorem fundamentalTheorem_singleBlock {A B : MPSTensor d D}
    (hA : IsInjective A) (hAB : SameMPV A B) : GaugeEquiv A B := by
  classical
  cases D with
  | zero => exact ⟨1, fun i => by simpa using Subsingleton.elim (B i) (A i)⟩
  | succ D' =>
      -- Step 1: linear map T with T(A_i) = B_i
      rcases linearExtension_exists_unique hA hAB with ⟨T, hT, -⟩
      -- Step 2: T is multiplicative
      have hMul := linearExtension_mul hA hAB hT
      -- Steps 3-4: T is bijective (simplicity of M_D)
      have hBij := linear_mul_endomorphism_bijective T hMul
                     (linearExtension_nonzero hA hAB hT)
      -- Step 5: T as algebra equivalence
      let fHom := linearMapToAlgHom T hMul hBij.surjective
      let f := AlgEquiv.ofBijective fHom hBij
      -- Step 6: Skolem-Noether gives conjugation matrix
      rcases skolemNoether_matrix f with ⟨X, hX⟩
      refine ⟨X, fun i => ?_⟩
      have hfi : f (A i) = B i := by simpa [f, fHom] using hT i
      simpa [hfi] using hX (A i)
\end{lstlisting}

    \vspace{-0.2em}
    {\scriptsize The proof mirrors the mathematical argument 1:1.
        Each \texttt{have}/\texttt{rcases} corresponds to one step in the chain from Slide~6.}
\end{frame}
