\begin{frame}{Matrix Product States in one slide}
    \begin{columns}[T]
        \column{0.52\textwidth}
        \begin{block}{Definition (MPS tensor)~[1]}
            Given $A = (A^i)_{i=0}^{d-1}$, $A^i\in M_D(\mathbb{C})$:
            \[
                \ket{V^{(N)}\!(A)} =  \sum_{i_1,\ldots,i_N}
                  \tr(A^{i_1}\!\cdots A^{i_N})\, \ket{i_1\!\cdots i_N}.
            \]
        \end{block}
        \begin{block}{Gauge invariance}
            $A^i \mapsto X A^i X^{-1}$ ($X \in \GL_D$) leaves every $\ket{V^{(N)}}$ unchanged.
        \end{block}
        \textcolor{dimwhite}{\footnotesize\textbf{Key question:} Is gauge the \emph{only} ambiguity?}

\begin{frame}{The Fundamental Theorem of MPS}
    \begin{block}{Theorem (P\'erez-Garc\'ia et al.~2007~[2], Cirac et al.~2017~[3])}
      If two MPS tensors $A$, $B$ generate the same family of quantum states for all chain lengths $N$, then $B$ is related to $A$ by a block-diagonal gauge transform --- up to a canonical form reduction.
    \end{block}
    \vspace{0.5em}
    The full theorem involves:
    \begin{enumerate}
        \item Decomposition into \textbf{irreducible blocks}
        \item \textbf{Spectral normalization} (Perron--Frobenius fixed point)
        \item \textbf{Periodicity removal} via blocking
        \item \textbf{Spectral gap} $\Rightarrow$ block separation
        \item \textbf{Per-block gauge equivalence} (Skolem--Noether)
        \item \textbf{Coefficient convergence} for the proportional case
    \end{enumerate}
    \vspace{0.4em}
    \textcolor{dimwhite}{\footnotesize Paper proofs span $\sim$30 pages across multiple sources. Lean formalization of key components: \textbf{$\sim$57K lines} across 172 files. Full end-to-end pipeline: not yet complete.}
\end{frame}

\begin{frame}[fragile]{Single-block Fundamental Theorem}
    \vspace{-0.4em}
    \begin{block}{The injective case: Skolem--Noether in Lean}
    \end{block}
    \vspace{-0.4em}
\begin{lstlisting}
theorem fundamentalTheorem_singleBlock
    {A B : MPSTensor d D}
    (hA : IsInjective A) (hAB : SameMPV A B) :
    GaugeEquiv A B
\end{lstlisting}
    \vspace{0.3em}
    \begin{itemize}
        \item If $A$ is \emph{injective} ($\{A^i\}$ span $M_D(\mathbb{C})$) and $B$ generates the same MPV family, then $B^i = X A^i X^{-1}$ for some invertible $X$.
        \item Proof route: word evaluation $\to$ algebra homomorphism $\to$ Skolem--Noether.
        \item \proved{Sorry-free, $\sim$200 lines total}
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Multi-block assembly and spectral gap}

theorem fundamentalTheorem_multiBlock_full
    ((*@$\mu$@*) : Fin r (*@$\to$@*) (*@$\mathbb{C}$@*))
    (A B : (k : Fin r) (*@$\to$@*) MPSTensor d (dim k))

    \xrightarrow{ \text{injectivity} }
    \underbrace{\rho(\cF_{AB}) < 1}_{\text{strict gap}}
    \xrightarrow{   }
    \underbrace{\cF_{AB}^N \to 0}_{\text{decay}}
    \]
\end{frame}

\begin{frame}[fragile]{Wielandt bound: $D^2$ cumulative span}
    \vspace{-0.4em}

\begin{frame}{Errata discovered during formalization}

        \item \textcolor{softred}{\textbf{Per-block trace equality} (attempted lemma):} SameMPV$_2$ + injective $\mu$ does \textbf{not} imply per-block traces agree. Counterexample: $r\!=\!2$, $\mu\!=\!(1,2)$, $A\!=\!([[4]],[[1]])$, $B\!=\!([[2]],[[2]])$.

        \item \textcolor{amber}{\textbf{Wolf citation numbering:}} Four separate counters (Thm/Prop/Cor/Lem). Multiple blueprint citations referenced the wrong result. {\scriptsize(Caught by audit agent.)}
    \end{enumerate}

\begin{frame}{The Proof Agent: \texttt{lean}}

                \item[$\circ$] \texttt{lean\_inspect}: goal state at cursor

        \end{itemize}

        \begin{enumerate}

            \item Read current file state
            \item Write proof attempt

            \item \texttt{lean\_inspect} $\to$ check goal state
            \item Fix type mismatches, missing imports
            \item Repeat 4--6 until \textcolor{leangreen}{0 errors}
            \item Write memory: what worked, what didn't
        \end{enumerate}

        \item \textbf{Day 12}: Proof agent attempts to show SameMPV$_2$ + injective $\mu$ $\Rightarrow$ per-block trace equality. After 2 hours, finds a counterexample. Writes \texttt{counterexample\_found.md} to memory.
        \item \textbf{Day 18}: New proof agent is asked to work on block separation. Reads memory. Sees the counterexample. \emph{Immediately} takes the correct route (peeling argument).
        \item \textbf{Day 25}: Blueprint agent is updating Ch\,9. Reads memory. Notes that the blueprint's proof sketch assumes trace equality. Corrects the sketch.
    \end{enumerate}
    \vspace{0.3em}
    \textcolor{dimwhite}{\footnotesize Without memory, each of these would have been an independent 2-hour dead end. With memory, the total cost of the counterexample discovery was amortized across all subsequent sessions.}
\end{frame}

\begin{frame}{What worked}
    \vspace{-0.4em}
    \begin{enumerate}
        \item \textcolor{leangreen}{\textbf{Scout-before-prove}} saved enormous time.\\ {\footnotesize Finding \texttt{monotone\_stabilizes\_iff\_noetherian} $\to$ 50 lines instead of 200.}
        \item \textcolor{leangreen}{\textbf{Disjoint file assignment}} prevented all merge conflicts after Phase 1.
        \item \textcolor{leangreen}{\textbf{Memory system}} prevented re-discovering known counterexamples.
        \item \textcolor{leangreen}{\textbf{Blueprint sync}} caught statement mismatches before they propagated.
        \item \textcolor{leangreen}{\textbf{Cleanup waves}} (batch, not continuous) prevented technical debt accumulation.
        \item \textcolor{leangreen}{\textbf{Errata tracking}} caught paper errors \emph{early}: the DS gauge $\neq$ TP discovery changed the proof route on Day 3.
    \end{enumerate}
\end{frame}

\begin{frame}{What didn't work}
    \vspace{-0.4em}
    \begin{enumerate}
        \item \textcolor{softred}{\textbf{False completion claims}} frequently for complex proofs.\\ {\footnotesize Agent says ``done'' $\to$ diagnostics shows 4 errors. Mitigation: always verify.}
        \item \textcolor{softred}{\textbf{Plausible but wrong math.}}\\ {\footnotesize Agents produce convincing proof sketches for false statements. The per-block trace equality attempt ran 2 hours before a \emph{human} found the counterexample.}
        \item \textcolor{softred}{\textbf{Multi-file refactoring}} is risky.\\ {\footnotesize Renaming across 15 files broke imports in 3. Keep changes local.}
        \item \textcolor{amber}{\textbf{Sorry closure needs mathematical strategy, not brute force.}}\\ {\footnotesize Throwing an expensive agent at a hard sorry without a plan wastes time.}
        \item \textcolor{amber}{\textbf{Heartbeat explosions}} from large proof terms.\\ {\footnotesize Some proofs still require up to 16M heartbeats. Factoring helps but doesn't solve the underlying complexity.}
    \end{enumerate}
\end{frame}

\begin{frame}{Scaling to larger projects}
    \textcolor{leangreen}{\textbf{This project:}} 57K lines, 172 files, 39 days, 1 human + 5 agent roles.
    \vspace{0.3em}

    \textcolor{amber}{\textbf{What scales:}}
    \begin{itemize}
        \item Scout $\to$ implement pattern {\scriptsize(works for any Mathlib-based project)}
        \item Disjoint file assignment {\scriptsize(scales with number of files)}
        \item Memory system {\scriptsize(more files = more value from persistent knowledge)}
        \item Blueprint sync {\scriptsize(independent chapters = linear parallelism)}
    \end{itemize}
    \vspace{0.3em}

    \textcolor{softred}{\textbf{What doesn't scale:}}
    \begin{itemize}
        \item Human mathematical oversight {\scriptsize(bottleneck for hard proofs)}
        \item Memory navigation at 1000+ files {\scriptsize(needs semantic search)}
        \item Integration complexity {\scriptsize(N agents $\to$ $O(N^2)$ potential conflicts)}
        \item Mathlib dependency updates {\scriptsize(breaking changes cascade)}
    \end{itemize}
\end{frame}

\begin{frame}{Lessons for the formalization community}
    \vspace{-0.4em}
    \begin{enumerate}
        \item \textcolor{leangreen}{\textbf{Formalization finds bugs.}} Every project of this scale discovers errors in published papers. This is a \emph{feature}, not a cost.
        \item \textcolor{leangreen}{\textbf{Search before you prove.}} The single highest-ROI activity is searching Mathlib for existing lemmas. Invest in search tooling.
        \item \textcolor{leangreen}{\textbf{Specialize your agents.}} \texttt{leanSearch} searches, \texttt{lean} proves, \texttt{leanSimplifier} cleans. One agent that does everything does nothing well.
        \item \textcolor{leangreen}{\textbf{Memory is multiplicative.}} Knowledge persisted from Day 1 saves time on Day 40.
        \item \textcolor{amber}{\textbf{Agents are amplifiers, not replacements.}} They amplify a human mathematician's ability to manage complexity.
        \item \textcolor{amber}{\textbf{The formalization gap is real.}} ``By standard arguments'' in a paper becomes 800 lines of Lean. Plan for $100\times$--$1000\times$ expansion.
    \end{enumerate}
\end{frame}
