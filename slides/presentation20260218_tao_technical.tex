\section{The theorem and its context}
%======================================================================

\begin{frame}{Matrix Product Vectors: the basic object}
\vspace{-0.6em}

\begin{block}{Definition (MPV family) \cite{Cirac2021Matrix}}
Given matrices $A = (A^i)_{i=1}^d$ with each $A^i\in \Mat_D(\mathbb{C})$, set
\[
  \ket{V^{(N)}(A)} \;:=\; \sum_{i_1,\dots,i_N=1}^d \tr(A^{i_1}\cdots A^{i_N})\, \ket{i_1\cdots i_N} \;\in\; (\mathbb{C}^d)^{\otimes N}.
\]
The \emph{MPV family} is $\mathcal{V}(A):=\bigl\{\ket{V^{(N)}(A)}\bigr\}_{N\ge 1}$.
\end{block}

\begin{block}{Gauge invariance}
Conjugating every matrix $A^i \mapsto X A^i X^{-1}$ ($X$ invertible)
leaves every $\ket{V^{(N)}}$ unchanged.
\end{block}

\textcolor{dimwhite}{\footnotesize\textbf{Key question:} Is gauge the \emph{only} ambiguity?}
\end{frame}

\begin{frame}{Fundamental Theorem: the injective (single-block) case}
\vspace{-0.6em}

\begin{block}{Injectivity}
$A$ is \emph{injective} if $\Span_{\mathbb{C}}\{A^i\}=\Mat_D(\mathbb{C})$.
\end{block}

\begin{block}{Theorem (FT, injective) \cite{Perez-Garcia2007Matrix,Cirac2021Matrix}}
If $A$ is injective and $\mathcal{V}(A)=\mathcal{V}(B)$ for all~$N$, then
\[
  \exists\,X\in \GLC_D(\mathbb{C}),\quad B^i = X A^i X^{-1}\quad (\forall\, i).
\]
\end{block}

\textcolor{dimwhite}{\footnotesize\textbf{Lean status:} \proved{Fully proved.}
\texttt{MPS/FundamentalTheorem.lean} (stable).}
\end{frame}

\begin{frame}{The multi-block target: canonical form}
\vspace{-0.6em}

\begin{block}{Canonical form \cite{Cirac2017Annals,Cirac2021Matrix}}
After blocking, a translation-invariant tensor decomposes as
\[
  A^i = \bigoplus_{k=1}^r \mu_k\,A_k^i,
\]
where each $A_k$ is \emph{primitive} (hence injective after blocking)
and the $\mu_k\in\mathbb{C}^\times$ are pinned by normalization.
\end{block}

\begin{block}{Fundamental Theorem in canonical form (informal)}
$\mathcal{V}(A)=\mathcal{V}(B)$ implies that the blocks of $A$ and $B$ agree
up to permutation and per-block gauge equivalence.
\end{block}
\end{frame}

\begin{frame}{The standard two-step derivation \cite{Cirac2021Matrix}}
\vspace{-0.6em}

\begin{block}{Structure (RMP Section IV.1)}
\begin{enumerate}\itemsep0.2em
  \item \textbf{Canonical form reduction.} 
  Remove nonphysical off-diagonal blocks; normalize each block
  via its transfer operator (blocking removes periodicity).
  \item \textbf{FT for canonical tensors.} 
  Equal MPV families imply blockwise gauge-phase equivalence
  plus a block permutation.
\end{enumerate}
\end{block}

\begin{block}{Where this talk focuses}
The technical heart is \emph{block separation} (first half of Step~2):
turning a global weighted-sum identity into per-block equalities.
\end{block}
\end{frame}

\begin{frame}{Step~(i): canonical form reduction \cite{Cirac2017Annals,Cirac2021Matrix}}
\vspace{-0.7em}

\begin{columns}[T]
\column{0.62\textwidth}
\begin{block}{Key moves}
\begin{itemize}\itemsep0.25em
  \item Decompose via invariant subspaces:
  $B^i \mapsto \bigoplus_k \mu_k A_k^i$.
  \item Discard upper-triangular/off-diagonal blocks
  (they never contribute to $\tr(A^{i_1}\cdots A^{i_N})$).
  \item Normalize each block by transfer-operator data:
  spectral radius $\rho(\cE_{A_k})=1$, full-rank fixed points.
\end{itemize}
\end{block}

\begin{block}{Lean status}
Canonical forms exist as records and constructors.
Existence theorem (arbitrary tensor $\to$ canonical form) is not yet formalized.
\end{block}

\begin{frame}{Quantum Wielandt: injectivity after blocking}
\vspace{-0.6em}

\begin{block}{Idea \cite{Sanz2010Quantum,Cirac2017Annals}}
For a primitive transfer operator, there is a finite blocking length~$L$
after which products of length~$L$ span the full matrix algebra.
A coarse bound:
\[
  L \le D^4 \quad \text{(quantum Wielandt)};\qquad
  \text{block-injectivity after }\le 3D^5.
\]
\end{block}

\begin{block}{Lean status}
The library works with explicit hypotheses
(\texttt{IsNBlkInjective}, \texttt{IsNormal}, \texttt{IsInjective})
rather than quantitative bounds.
\end{block}
\end{frame}

\begin{frame}{Equal-or-orthogonal: the mixed transfer test}
\vspace{-0.6em}

\begin{block}{Mixed transfer operator \cite{Perez-Garcia2007Matrix,Cirac2021Matrix}}
For injective blocks, define
\[
  \cF_{jk}(X) := \sum_{i=1}^d A_j^i\,X\,(A_k^i)^\dagger.
\]
\end{block}

\begin{block}{Key dichotomy}
\begin{itemize}\itemsep0.2em
  \item $\rho(\cF_{jk})=1$ $\Longrightarrow$ 
  blocks are gauge-phase equivalent.
  \item $\rho(\cF_{jk})<1$ $\Longrightarrow$ 
  cross-terms decay exponentially (asymptotic orthogonality).
\end{itemize}
\end{block}

\textcolor{dimwhite}{\footnotesize This is the engine behind block separation: inequivalent blocks decouple at large~$N$.}
\end{frame}

\begin{frame}{Step~(ii): FT in canonical form}
\vspace{-0.6em}

\begin{block}{Theorem (proportional MPVs; simplified) \cite{Cirac2017Annals,Cirac2021Matrix}}
Let $A$ and~$B$ be in canonical form.
If $\ket{V^{(N)}(A)}$ and $\ket{V^{(N)}(B)}$ are proportional for all~$N$,
then (after relabeling) each block $B_j$ is gauge-phase equivalent
to some~$A_{\pi(j)}$.
\end{block}

\begin{block}{Corollary (equal MPVs)}
If $\mathcal{V}(A)=\mathcal{V}(B)$ exactly, phases cancel
and one obtains genuine per-block gauge equivalence plus a block permutation.
\end{block}
\end{frame}

\begin{frame}{Why canonical form is essential}
\vspace{-0.6em}

\begin{block}{Ambiguity without normalization}
Rescaling $A_k^i \mapsto c\, A_k^i$ while compensating $\mu_k \mapsto c^{-1}\mu_k$
preserves the block-diagonal tensor. This freedom breaks naive separation.
\end{block}

\begin{block}{How normalization helps \cite{Cirac2017Annals,Cirac2021Matrix}}
Canonical form ties block scale to transfer-operator data
(e.g.\ $\rho(\cE_{A_k})=1$), so the only allowed rescalings are phases.
A strict $|\mu_k|$ ordering then pins down a unique labeling.
\end{block}
\end{frame}

\begin{frame}{Injectivity: spanning vs.\ inverse tensor}
\vspace{-0.6em}

\begin{block}{Literature formulation \cite{Cirac2017Annals,Perez-Garcia2007Matrix}}
$A$ is injective when products $\{A^{i_1}\cdots A^{i_L}\}$ span $\Mat_D(\mathbb{C})$
for some~$L$.
Equivalently, an ``inverse tensor'' $A^{-1}$ exists satisfying
\[
  \sum_{i_1,\dots,i_L} (A^{i_1}\cdots A^{i_L})_{\alpha\beta}\,(A^{-1})^{i_1\cdots i_L}_{\alpha'\beta'}
  = \delta_{\alpha\alpha'}\delta_{\beta\beta'}.
\]
\end{block}

\begin{block}{Lean formulation}
\texttt{IsInjective A}: the span of $\{A^i\}$ equals $\Mat_D(\mathbb{C})$.\\
\texttt{IsNBlkInjective}: the $L$-step spanning condition.
\end{block}
\end{frame}

\begin{frame}{Step 1: intertwiner $\to$ linear extension}
\vspace{-0.6em}

\begin{block}{Idea (literature) \cite{Perez-Garcia2007Matrix,Cirac2021Matrix}}
$\mathcal{V}(A)=\mathcal{V}(B)$ with $A$ injective implies
$\exists\, X\neq 0$ such that $B^i X = X A^i$ for all~$i$.
\end{block}

\begin{block}{Lean route}
From injectivity, construct a unique $\mathbb{C}$-linear map
$T : \Mat_D(\mathbb{C})\to \Mat_D(\mathbb{C})$ with $T(A^i)=B^i$.

\textcolor{dimwhite}{\footnotesize If $T(M)=XMX^{-1}$, then $T(A^i)=B^i$ is exactly $B^i X = X A^i$.
The two viewpoints are mathematically equivalent.}
\end{block}

\textcolor{dimwhite}{\footnotesize Source: \texttt{MPS/LinearExtension.lean}.}
\end{frame}

\begin{frame}{Step 2: trace pairing and uniqueness}
\vspace{-0.6em}

\begin{block}{Key construction}
Define
$\Phi_A : \Mat_D(\mathbb{C}) \to \mathbb{C}^d$, \quad $(\Phi_A(M))_i := \tr(M A^i)$.

Injectivity of $A$ forces $\ker(\Phi_A)=\{0\}$ (nondegenerate trace pairing).
SameMPV supplies enough trace identities to determine~$T$ uniquely via~$\Phi_A$.
\end{block}

\textcolor{dimwhite}{\footnotesize Source: \texttt{Algebra/TracePairing.lean}, \texttt{MPS/LinearExtension.lean}.}
\end{frame}

\begin{frame}{Step 3: multiplicativity from length-3 words}
\vspace{-0.6em}

\begin{block}{Key lemma}
From $\tr(A^i A^j A^k)=\tr(B^i B^j B^k)$ for all $i,j,k$, conclude
\[
  T(MN)=T(M)\,T(N)\qquad \forall\, M,N\in \Mat_D(\mathbb{C}).
\]
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{linearExtension\_mul} in \texttt{MPS/LinearExtension.lean}.}
\end{frame}

\begin{frame}{Step 4: simplicity + Skolem--Noether}
\vspace{-0.6em}

\begin{block}{Simplicity $\Rightarrow$ bijectivity}
$T$ is linear and multiplicative, so $\ker(T)$ is a two-sided ideal of $\Mat_D(\mathbb{C})$.
Since this algebra is simple, $\ker(T)=0$ (a trace argument rules out $T=0$).
Finite dimension gives surjectivity.
\end{block}

\begin{block}{Skolem--Noether $\Rightarrow$ inner automorphism}
Every $\mathbb{C}$-algebra automorphism of $\Mat_D(\mathbb{C})$
has the form $M\mapsto XMX^{-1}$ for some invertible~$X$.
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{Algebra/SkolemNoether.lean} (wraps Mathlib's Skolem--Noether).}
\end{frame}

\section{Block separation: the main open problem}
%======================================================================

\begin{frame}{The block separation problem}
\vspace{-0.6em}

\begin{block}{Setup}
For $A^i=\bigoplus_k \mu_k A_k^i$, the MPV decomposes as
\[
  \mathrm{mpv}(A,\sigma)=\sum_{k=1}^r \mu_k^N\,\mathrm{mpv}(A_k,\sigma).
\]
Global equality $\mathcal{V}(A)=\mathcal{V}(B)$ becomes
\[
  \sum_{k=1}^r \mu_k^N\,\Delta_k(N,\sigma)=0,
  \qquad \Delta_k:=\mathrm{mpv}(A_k,\sigma)-\mathrm{mpv}(B_k,\sigma).
\]
\end{block}

\begin{block}{Goal}
Prove $\Delta_k=0$ for every block~$k$ (per-block SameMPV).
\end{block}
\end{frame}

\begin{frame}{Why naive Vandermonde fails}
\vspace{-0.6em}

\begin{block}{Two obstructions}
\begin{enumerate}\itemsep0.25em
  \item \textbf{Type-varying coefficients.} 
  The ``coefficient'' $\Delta_k(N,\sigma)$ depends on~$N$,
  so there is no fixed vector to which classical Vandermonde applies.
  \item \textbf{Block swap + rescale.} 
  Even with distinct $\mu_k$, swapping blocks and compensating scales
  can preserve the global MPV.
\end{enumerate}
\end{block}

\begin{block}{Minimal counterexample (scalar blocks)}
$r=2$, $D=1$, $\mu=(1,2)$: set $B_0^i=\tfrac{\mu_1}{\mu_0}A_1^i$, 
$B_1^i=\tfrac{\mu_0}{\mu_1}A_0^i$.
Global MPVs agree, but $A_0\neq B_0$.
\end{block}

\textcolor{dimwhite}{\footnotesize Documented in \texttt{PiAlgebra/BlockSeparationProof.lean} (Experimental).}
\end{frame}

\begin{frame}{How canonical form defeats the counterexample}
\vspace{-0.6em}

\begin{block}{Key insight}
In canonical form, $\mu_k$ are fixed by transfer-operator normalization
($\rho(\cE_{A_k})=1$).
Rescaling $A_k\mapsto cA_k$ is only allowed when $|c|=1$.
The swap-and-rescale trick requires $|\mu_0|=|\mu_1|$,
which strict ordering forbids.
\end{block}

\begin{block}{Lean encoding}
\texttt{IsCanonicalForm} enforces DS-gauge normalization and
strict ordering of~$\|\mu_k\|$.
\end{block}
\end{frame}

\begin{frame}{Separation strategy: dominance + peeling}
\vspace{-0.6em}

\begin{block}{Step 1: asymptotic dominance}
From $\|\mu_1\|>\|\mu_2\|\ge\cdots$ and
$\sum_k \mu_k^N\Delta_k=0$:
\[
  \Delta_1(N,\sigma) = -\sum_{k\ge 2} \left(\frac{\mu_k}{\mu_1}\right)^{\!N} \Delta_k(N,\sigma).
\]
If $|\Delta_k|$ is uniformly bounded, then $|\Delta_1|$ decays exponentially.
\end{block}

\begin{block}{Step 2: uniform bounds from DS gauge}
Under $\sum_i (A^i)^\dagger A^i=I$:
\[
  \bigl|\tr(\mathrm{evalWord}(A,w)^L)\bigr| \le D
  \qquad (\forall\, w, \forall\, L).
\]
\end{block}

\textcolor{dimwhite}{\footnotesize Both steps are proved in Lean (Experimental).}
\end{frame}

\begin{frame}{From decay to exact equality}
\vspace{-0.6em}

\begin{block}{What the peeling gives}
For suitable test words:
\[
  \tr(T_1^L)=\tr(U_1^L)+O(\rho^L),
\]
where $T_1=\mathrm{evalWord}(A_1,w)$, $U_1=\mathrm{evalWord}(B_1,w)$.
\end{block}

\begin{block}{What remains (literature route)}
\begin{enumerate}\itemsep0.2em
  \item Use primitivity to upgrade asymptotic identities to exact constraints.
  \item Use injectivity (spanning) to vary~$w$ and force equality of all word traces.
  \item Induct: subtract the leading block and recurse.
\end{enumerate}
\end{block}
\end{frame}

\begin{frame}{Formal gap: \texttt{block\_separation\_core} for $r\ge 2$}
\vspace{-0.6em}

\begin{block}{Open statement}
Under \texttt{IsCanonicalForm} (injective blocks, DS gauge,
strict $\|\mu\|$ ordering), prove:
\[
  \Bigl(\forall\, N,\;\forall\, \sigma,\;\sum_k \mu_k^N\,\Delta_k(N,\sigma)=0\Bigr)
  \;\Longrightarrow\; \forall\, k,\; \mathrm{SameMPV}(A_k,B_k).
\]
\end{block}

\begin{block}{Lean status}
\texttt{PiAlgebra/CanonicalFormSep.lean} (Experimental):
$r=0,1$ proved; $r\ge 2$ is the remaining \texttt{sorry}.
\end{block}
\end{frame}

\begin{frame}{Per-block SameMPV $\Rightarrow$ per-block gauge}
\vspace{-0.6em}

\begin{block}{Reuse of the single-block FT}
Once separation gives $\mathrm{SameMPV}(A_k,B_k)$ for each~$k$,
and each $A_k$ is injective:
\[
  \forall\, k,\;\exists\, X_k\in\GLC_{D_k}(\mathbb{C}),\quad B_k^i=X_k A_k^i X_k^{-1}.
\]
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: apply \texttt{fundamentalTheorem\_singleBlock} blockwise.}
\end{frame}

\begin{frame}{\texorpdfstring{The product algebra $\Pi$ and its automorphisms}{The product algebra Pi and its automorphisms}}
\vspace{-0.6em}

\begin{block}{Object}
$\Pi := \prod_{k=1}^r \Mat_{D_k}(\mathbb{C})$.
\end{block}

\begin{block}{Theorem (proved in Lean)}
Every $\mathbb{C}$-algebra automorphism of~$\Pi$ decomposes into:
\begin{enumerate}\itemsep0.2em
  \item a block permutation $\pi$ (preserving dimensions),
  \item inner automorphisms per block: $M\mapsto X_k M X_k^{-1}$.
\end{enumerate}
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{algEquiv\_pi\_matrix\_decomposition} in \texttt{MPS/BlockPermutationMPS.lean}.}
\end{frame}

\begin{frame}{Multi-block FT: complete modulo separation}
\vspace{-0.6em}

\begin{block}{Lean theorem}
\begin{itemize}\itemsep0.2em
  \item \textbf{Input:} block-diagonal SameMPV plus a separation hypothesis\\
  \texttt{hSep :\ forall k, SameMPV (A k) (B k)}.
  \item \textbf{Output:} blockwise gauges $X_k$, global block-diagonal gauge,
  and the $\Pi$-automorphism decomposition.
\end{itemize}
\end{block}

\begin{block}{Agreement with the literature}

\end{block}

\begin{block}{Irreducible/periodic generalization \cite{Cirac2017Irreducible}}
If the transfer operator has period $p>1$, canonical form is replaced by
\emph{irreducible form}, and the FT allows an extra diagonal unitary~$Z$:
\[
  Z A^i = Y B^i Y^{-1},\quad [Z,A^i]=0,\quad \mathcal{V}(A)=\mathcal{V}(ZA).
\]
\end{block}

\begin{block}{Quantitative blocking bounds \cite{Sanz2010Quantum,Fernandez2015Frustration}}
Physics applications (parent Hamiltonians, uncle Hamiltonians, RG fixed points)
use explicit blocking lengths.
The library currently works with qualitative hypotheses only.
\end{block}
\end{frame}
