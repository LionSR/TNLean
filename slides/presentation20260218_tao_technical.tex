\section{Project goal and theorem statements}
%======================================================================

\begin{frame}{Matrix Product Vectors (MPVs): coefficient family}
\vspace{-0.6em}

\begin{block}{Definition (MPV family) \cite{Cirac2021Matrix}}
For matrices $A = (A^i)_{i=1}^d$ with $A^i\in \Mat_D(\mathbb{C})$, define
\[
  \ket{V^{(N)}(A)} := \sum_{i_1,\dots,i_N=1}^d \tr(A^{i_1}\cdots A^{i_N}) \ket{i_1\cdots i_N} \in (\mathbb{C}^d)^{\otimes N}.
\]
The MPV family is $\mathcal{V}(A):=\{\ket{V^{(N)}(A)}\}_{N\in\mathbb{N}}$.
\end{block}

\begin{block}{Gauge invariance}
If $B^i = X A^i X^{-1}$ with $X\in \GLC_D(\mathbb{C})$, then $\mathcal{V}(A)=\mathcal{V}(B)$.
\end{block}
\end{frame}

\begin{frame}{Fundamental Theorem: injective (single-block) case}
\vspace{-0.6em}

\begin{block}{Injectivity (spanning formulation)}
$A$ is \emph{injective} if $\Span_{\mathbb{C}}\{A^i : i=1,\dots,d\}=\Mat_D(\mathbb{C})$.
\end{block}

\begin{block}{Theorem (Fundamental Theorem, injective) \cite{Perez-Garcia2007Matrix,Cirac2021Matrix}}
Assume $A$ is injective and $\mathcal{V}(A)=\mathcal{V}(B)$ (equality for all $N$).
Then $A$ and $B$ are gauge equivalent:
\[
  \exists\,X\in \GLC_D(\mathbb{C}),\quad \forall i,\qquad B^i = X A^i X^{-1}.
\]
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{MPSLean/MPS/FundamentalTheorem.lean} (stable).}
\end{frame}

\begin{frame}{Canonical-form multi-block target (physics statement)}
\vspace{-0.6em}

\begin{block}{Canonical form shape \cite{Cirac2017Annals,Cirac2021Matrix}}
After blocking, a translation-invariant tensor can be represented as
\[
  A^i = \bigoplus_{k=1}^r \mu_k\,A_k^i,
\]
where each block $A_k$ is \emph{normal/primitive} (hence injective after blocking) and
$\mu_k\in\mathbb{C}^\times$ are fixed by a normalization convention.
\end{block}

\begin{block}{Fundamental Theorem in canonical form (informal)}
If $A$ and $B$ are in canonical form and $\mathcal{V}(A)=\mathcal{V}(B)$, then their blocks
match up to permutation, gauge, and (in the proportional case) phases.
\end{block}
\end{frame}

\section{Literature roadmap (Cirac RMP IV.1)}

\begin{frame}{Two-step derivation in Cirac RMP \cite{Cirac2021Matrix}}
\vspace{-0.6em}

\begin{block}{Standard structure (Section IV.1: Fundamental Theorem of MPVs)}
\begin{enumerate}\itemsep0.2em
  \item \textbf{Canonical form reduction:}
  remove nonphysical off-diagonal blocks, normalize each block by its transfer operator
  (blocking removes periodicity).
  \item \textbf{Fundamental Theorem for canonical tensors:}
  equal (or proportional) MPV families imply blockwise gauge-phase equivalence and a block permutation.
\end{enumerate}
\end{block}

\begin{block}{Where this talk places emphasis}
The technical heart is \emph{block separation} (Step 2, first half):
turning a global weighted-sum identity into per-block equalities.
\end{block}
\end{frame}

\begin{frame}{Step (i): canonical form reduction \cite{Cirac2017Annals,Cirac2021Matrix}}
\vspace{-0.7em}

\begin{columns}[T]
\column{0.62\textwidth}
\begin{block}{Key moves}
\begin{itemize}\itemsep0.25em
  \item Decompose using invariant subspaces:
  $B^i \mapsto \bigoplus_k \mu_k A_k^i$.
  \item Fix \emph{physical irrelevance} of upper-triangular/off-diagonal blocks:
  they never contribute to $\tr(A^{i_1}\cdots A^{i_N})$.
  \item Normalize each block by transfer-operator data:
  spectral radius $\rho(\cE_{A_k})=1$ and fixed points full rank.
\end{itemize}
\end{block}

\begin{block}{Lean status}
Canonical forms exist as \emph{records} and constructors (no existence theorem yet).
\end{block}

\begin{frame}{Quantum Wielandt: why ``injective after blocking'' is standard}
\vspace{-0.6em}

\begin{block}{Quantitative blocking bound \cite{Sanz2010Quantum,Cirac2017Annals}}
For a \emph{normal} tensor (primitive transfer operator), there exists a blocking length $L$
so that products of length $L$ span the full matrix algebra (injectivity).
A coarse bound in the literature is
\[
  L \le D^4 \quad \text{(quantum Wielandt), and block-injectivity after } \le 3D^5.
\]
\end{block}

\begin{block}{Lean status}
Blocking-length bounds are not formalized; Lean works with explicit hypotheses
\texttt{IsNBlkInjective} / \texttt{IsNormal} / \texttt{IsInjective}.
\end{block}
\end{frame}

\begin{frame}{Basis of normal tensors and the mixed transfer test}
\vspace{-0.6em}

\begin{block}{Basis of normal tensors (BNT) \cite{Cirac2017Annals,Cirac2021Matrix}}
Canonical form blocks can repeat up to gauge and phase:
$\tilde A = e^{\mathrm{i}\phi} X A X^{-1}$.
A BNT selects representatives $\{A_j\}_{j=1}^g$ so that the families
$\{\mathcal{V}(A_j)\}$ become linearly independent for large $N$.
\end{block}

\begin{block}{Mixed transfer operator detects equivalence vs orthogonality}
For injective blocks define
\[
  \cF_{jk}(X) := \sum_{i=1}^d A_j^i\,X\,(A_k^i)^\dagger.
\]
Then
\begin{itemize}\itemsep0.2em
  \item $\rho(\cF_{jk})=1$ $\Rightarrow$ blocks are gauge-phase equivalent.
  \item $\rho(\cF_{jk})<1$ $\Rightarrow$ cross-terms decay, giving asymptotic orthogonality.
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Step (ii): Fundamental Theorem in canonical form}
\vspace{-0.6em}

\begin{block}{Theorem (proportional MPVs; simplified) \cite{Cirac2017Annals,Cirac2021Matrix}}
Let $A$ and $B$ be in canonical form with BNTs $\{A_j\}$ and $\{B_j\}$.
If $\ket{V^{(N)}(A)}$ and $\ket{V^{(N)}(B)}$ are proportional for all $N$, then
(after relabeling blocks) each $B_j$ is gauge-phase equivalent to some $A_{\pi(j)}$.
\end{block}

\begin{block}{Corollary (equal MPVs)}
If $\mathcal{V}(A)=\mathcal{V}(B)$ exactly (no $N$-dependent phase), then phases cancel
and one obtains a genuine gauge equivalence, together with a block permutation.
\end{block}
\end{frame}

\begin{frame}{Canonical form prevents ``hidden'' rescalings}
\vspace{-0.6em}

\begin{block}{Ambiguity without normalization}
Rescaling a block $A_k^i \mapsto c A_k^i$ and compensating $\mu_k \mapsto c^{-1}\mu_k$
leaves $\mu_k A_k^i$ unchanged, hence preserves the block-diagonal tensor.
This breaks naive separation arguments.
\end{block}

\begin{block}{Normalization principle \cite{Cirac2017Annals,Cirac2021Matrix}}
Canonical form ties block scale to transfer-operator data
(e.g. $\rho(\cE_{A_k})=1$ and fixed points), so rescalings are only allowed by phases.
Ordering conventions (e.g. strict $|\mu_k|$) then pin down a unique block labeling.
\end{block}
\end{frame}

\end{frame}

\begin{frame}{Injectivity: spanning vs inverse tensor}
\vspace{-0.6em}

\begin{block}{Literature formulation \cite{Cirac2017Annals,Perez-Garcia2007Matrix}}
Injectivity is often stated via an inverse map $\Gamma_L^{-1}$:
for some $L$, the matrices $\{A^{i_1}\cdots A^{i_L}\}$ span $\Mat_D(\mathbb{C})$.
Equivalently, there exists an ``inverse tensor'' $A^{-1}$ with
\[
  \sum_{i_1,\dots,i_L} (A^{i_1}\cdots A^{i_L})_{\alpha\beta}\,(A^{-1})^{i_1\cdots i_L}_{\alpha'\beta'}
  = \delta_{\alpha\alpha'}\delta_{\beta\beta'}.
\]
\end{block}

\begin{block}{Lean formulation}
\texttt{IsInjective A} means $\Span\{A^i\}=\Mat_D(\mathbb{C})$.
Blocking versions \texttt{IsNBlkInjective} encode the $L$-step spanning condition.
\end{block}
\end{frame}

\begin{frame}{Literature step: ``intertwiner implies gauge''}
\vspace{-0.6em}

\begin{block}{Typical statement (injective case) \cite{Perez-Garcia2007Matrix,Cirac2021Matrix}}
If $\mathcal{V}(A)=\mathcal{V}(B)$ and $A$ is injective, then there exists a matrix
$X\neq 0$ such that
\[
  B^i X = X A^i\qquad (\forall i).
\]
Injectivity forces $X$ to be invertible, hence $B^i = X A^i X^{-1}$.
\end{block}

\begin{block}{Interpretation}
The map $X$ is an \emph{intertwiner} between two representations of the tensor algebra
generated by the letters $i\in\{1,\dots,d\}$.
\end{block}
\end{frame}

\begin{frame}{Lean route: build a linear extension $T$ (algebraic avatar of the intertwiner)}
\vspace{-0.6em}

\begin{block}{Key construction}
From injectivity, define the unique $\mathbb{C}$-linear map
$T : \Mat_D(\mathbb{C})\to \Mat_D(\mathbb{C})$ satisfying $T(A^i)=B^i$.
\end{block}

\begin{block}{Why it matches the literature}
If $T(M)=XMX^{-1}$, then $T(A^i)=B^i$ is equivalent to the intertwining relations
$B^i X = X A^i$.
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{MPS/LinearExtension.lean}.}
\end{frame}

\begin{frame}{Trace pairing and uniqueness of the extension}
\vspace{-0.6em}

\begin{block}{Trace pairing map}
Define
\[
  \Phi_A : \Mat_D(\mathbb{C}) \to \mathbb{C}^d,\qquad (\Phi_A(M))_i := \tr(M A^i).
\]
Injectivity of $A$ implies $\ker(\Phi_A)=\{0\}$ (nondegenerate trace pairing).
\end{block}

\begin{block}{Extension lemma (Lean)}
SameMPV provides identities among traces of words, giving enough constraints to solve for
$T$ uniquely via $\Phi_A$.
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{Algebra/TracePairing.lean} and \texttt{MPS/LinearExtension.lean}.}
\end{frame}

\begin{frame}{Multiplicativity: using length-3 word identities}
\vspace{-0.6em}

\begin{block}{Input from SameMPV}
For all $i,j,k$,
\[
  \tr(A^i A^j A^k)=\tr(B^i B^j B^k).
\]
\end{block}

\begin{block}{Conclusion (Lean lemma)}
The extension $T$ satisfies $T(MN)=T(M)T(N)$ for all $M,N\in \Mat_D(\mathbb{C})$.
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{linearExtension\_mul} in \texttt{MPS/LinearExtension.lean}.}
\end{frame}

\begin{frame}{Matrix algebras are simple: nonzero multiplicative maps are bijective}
\vspace{-0.6em}

\begin{block}{Simplicity argument}
If $T$ is $\mathbb{C}$-linear and multiplicative then $\ker(T)$ is a two-sided ideal.
Since $\Mat_D(\mathbb{C})$ is simple, $\ker(T)$ is either $0$ or all of $\Mat_D$.
A short trace argument rules out $T=0$, hence $T$ is injective; finite-dimensionality gives surjectivity.
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{Algebra/SkolemNoether.lean}.}
\end{frame}

\begin{frame}{Skolem--Noether: automorphisms of $\Mat_D(\mathbb{C})$ are inner}
\vspace{-0.6em}

\begin{block}{Theorem (Skolem--Noether)}
Every $\mathbb{C}$-algebra automorphism $f: \Mat_D(\mathbb{C})\to\Mat_D(\mathbb{C})$ is of the form
\[
  f(M)=XMX^{-1}\qquad (\exists X\in\GLC_D(\mathbb{C})).
\]
\end{block}

\begin{block}{Lean usage}
The proof packages $T$ as an algebra equivalence and invokes Mathlib's Skolem--Noether.
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{skolemNoether\_matrix} in \texttt{Algebra/SkolemNoether.lean}.}
\end{frame}

\begin{frame}{Equivalence to the intertwiner formulation}
\vspace{-0.6em}

\begin{block}{From Lean to the literature}
If $T(M)=XMX^{-1}$ and $T(A^i)=B^i$, then
\[
  X A^i = B^i X \quad \Leftrightarrow \quad B^i = X A^i X^{-1}.
\]
Thus the Lean proof is a mathematically equivalent encoding of
``intertwiner implies gauge'', with $T$ representing conjugation by $X$.
\end{block}

\begin{block}{Why this is a good formalization boundary}
The only nontrivial external input is Skolem--Noether, already present in Mathlib.
No analytic/spectral assumptions are required in the single-block FT.
\end{block}
\end{frame}

\begin{frame}{Transfer and mixed transfer operators}
\vspace{-0.6em}

\begin{block}{Transfer operator (channel viewpoint)}
\[
  \cE_A(X) := \sum_{i=1}^d A^i X (A^i)^\dagger.
\]
\end{block}

\begin{block}{Mixed transfer operator}
\[
  \cF_{A,B}(X) := \sum_{i=1}^d A^i X (B^i)^\dagger.
\]
Iterates expand as sums over words:
\[
  \cF_{A,B}^N(X)=\sum_{\sigma:\mathrm{Fin}\,N\to\{1,\dots,d\}} \mathrm{evalWord}(A,\sigma)\,X\,\mathrm{evalWord}(B,\sigma)^\dagger.
\]
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{Spectral/MixedTransfer.lean} (\texttt{mixedTransferMap\_pow\_apply}).}
\end{frame}

\begin{frame}{Irreducible and primitive channels (Wolf framework)}
\vspace{-0.6em}

\begin{block}{Channel-theoretic taxonomy \cite{Evans1978Spectral,Wolf2012Quantum,Cirac2021Matrix}}
\begin{itemize}\itemsep0.25em
  \item \textbf{Irreducible}: unique full-rank fixed points; no nontrivial invariant subspaces.
  \item \textbf{Primitive}: unique peripheral eigenvalue (no periodicity); implies a spectral gap.
  \item \textbf{Blocking}: powers $\cE_A^p$ remove peripheral phases $\mathrm{e}^{2\pi\mathrm{i}q/p}$.
\end{itemize}
\end{block}

\begin{block}{Connection to MPS}
Primitivity is the analytic engine behind exponential convergence and correlation decay.
\end{block}
\end{frame}

\begin{frame}{Canonical form normalization and DS gauge in Lean}
\vspace{-0.6em}

\begin{block}{Literature normalization \cite{Cirac2017Annals,Cirac2021Matrix}}
Blocks are normalized so that $\rho(\cE_{A_k})=1$ and fixed points are full rank.
This pins down the magnitude of the scaling factors $\mu_k$.
\end{block}

\begin{block}{Lean canonical-form predicate (used for separation)}
A convenient gauge choice is a (one-sided) doubly-stochastic normalization,
\[
  \sum_i (A_k^i)^\dagger A_k^i = I,\qquad \|\mu_1\|>\cdots>\|\mu_r\|,
\]
packaged as \texttt{IsCanonicalForm} in \texttt{PiAlgebra/CanonicalFormSep.lean}.
\end{block}
\end{frame}

\begin{frame}{Equal-or-orthogonal lemma via mixed-transfer spectral radius}
\vspace{-0.6em}

\begin{block}{Literature principle \cite{Perez-Garcia2007Matrix,Cirac2021Matrix}}
For injective normalized blocks $A,B$:
\begin{itemize}\itemsep0.2em
  \item Either $B$ is gauge-phase equivalent to $A$:
  $B^i = \mathrm{e}^{\mathrm{i}\phi} X A^i X^{-1}$,
  \item or the mixed transfer has strict contraction:
  $\rho(\cF_{A,B})<1$ and hence $\cF_{A,B}^N\to 0$.
\end{itemize}
\end{block}

\begin{block}{Lean theorem (stable)}
\texttt{spectralRadius\_mixedTransfer\_lt\_one} and \texttt{mixedTransfer\_pow\_tendsto\_zero}
implement the ``orthogonal'' branch.
\end{block}
\end{frame}

\begin{frame}{Correlation length and spectral gap (transfer-operator perspective)}
\vspace{-0.6em}

\begin{block}{Correlation length heuristic \cite{Lancien2021Correlation}}
For a primitive transfer operator with eigenvalues $1=\lambda_1$, $|\lambda_2|\ge|\lambda_3|\ge\cdots$,
connected correlations decay as $|\lambda_2|^n$ and
\[
  \xi \approx -\frac{1}{\log |\lambda_2|}.
\]
\end{block}

\begin{block}{Why it matters for block separation}
The same spectral-gap mechanism controls decay of mixed-transfer cross-terms
between inequivalent blocks.
\end{block}
\end{frame}

\section{Block separation: literature-aligned strategy and Lean status}
%======================================================================

\begin{frame}{Block separation: the formal problem}
\vspace{-0.6em}

\begin{block}{Weighted-sum identity (block-diagonal MPV)}
For $A^i=\oplus_k \mu_k A_k^i$,
\[
  \mathrm{mpv}(A,\sigma)=\sum_{k=1}^r \mu_k^N\,\mathrm{mpv}(A_k,\sigma)
  \qquad (\sigma:\mathrm{Fin}\,N\to\{1,\dots,d\}).
\]
Global equality $\mathcal{V}(A)=\mathcal{V}(B)$ becomes
\[
  \sum_{k=1}^r \mu_k^N\,\Delta_k(N,\sigma)=0,\qquad
  \Delta_k:=\mathrm{mpv}(A_k,\sigma)-\mathrm{mpv}(B_k,\sigma).
\]
\end{block}

\begin{block}{Goal}
Deduce $\Delta_k(N,\sigma)=0$ for each block $k$ (per-block SameMPV).
\end{block}
\end{frame}

\begin{frame}{Why naive Vandermonde separation is false}
\vspace{-0.6em}

\begin{block}{Two distinct obstructions}
\begin{itemize}\itemsep0.25em
  \item \textbf{Type-varying coefficients}: $\sigma$ depends on $N$, so there is no fixed coefficient
  family $c_k$ to which classical Vandermonde applies.
  \item \textbf{Block swap + rescale}: even with injective nonzero $\mu_k$, global MPV equality does
  not force per-block equality.
\end{itemize}
\end{block}

\begin{block}{Minimal counterexample (scalar blocks)}
$r=2$, $D_1=D_2=1$, $\mu=(1,2)$, and swap/rescale the blocks:
$B_0(i)=\tfrac{\mu_1}{\mu_0}A_1(i)$, $B_1(i)=\tfrac{\mu_0}{\mu_1}A_0(i)$.
Then global MPVs agree, but $A_0\neq B_0$.
\end{block}

\textcolor{dimwhite}{\footnotesize Documented in \texttt{PiAlgebra/BlockSeparationProof.lean} (Experimental).}
\end{frame}

\begin{frame}{How canonical form rules out the counterexample}
\vspace{-0.6em}

\begin{block}{Core point: $\mu_k$ are not free parameters in canonical form}
In the literature, $\mu_k$ are fixed by transfer-operator normalization
(e.g. $\rho(\cE_{A_k})=1$), so scaling $A_k\mapsto cA_k$ is only allowed when $|c|=1$.
The block-swap rescaling requires $|\mu_0|=|\mu_1|$, which is forbidden by strict ordering.
\end{block}

\begin{block}{Lean canonical-form hypothesis}
\texttt{IsCanonicalForm} explicitly enforces (i) DS-gauge normalization and
(ii) strict anti ordering of $\|\mu_k\|$.
\end{block}
\end{frame}

\begin{frame}{Literature-aligned separation strategy: equal-or-orthogonal}
\vspace{-0.6em}

\begin{block}{Step 1: reduce to a BNT / injective blocks}
Use canonical form and blocking to ensure each block is injective.
Then inequivalent blocks satisfy a mixed-transfer contraction $\rho(\cF_{jk})<1$.
\end{block}

\begin{block}{Step 2: asymptotic orthogonality}
If $\rho(\cF_{jk})<1$, then $\cF_{jk}^N\to 0$.
Cross-correlations (mixed traces) vanish exponentially, while diagonal blocks persist.
\end{block}

\textcolor{dimwhite}{\footnotesize Implemented in Lean: \texttt{Spectral/SpectralGap.lean}.}
\end{frame}

\begin{frame}{Asymptotic dominance under strict $|\mu|$ ordering}
\vspace{-0.6em}

\begin{block}{Peeling identity}
Assume $\|\mu_1\|>\|\mu_2\|\ge\cdots$ and
$\sum_k \mu_k^N\Delta_k(N,\sigma)=0$.
Then
\[
  \Delta_1(N,\sigma) = -\sum_{k\ge 2} \left(\frac{\mu_k}{\mu_1}\right)^N \Delta_k(N,\sigma).
\]
If $|\Delta_k|$ are uniformly bounded, $|\Delta_1|$ decays exponentially.
\end{block}

\begin{block}{Interpretation}
The leading block cannot be cancelled by subleading blocks for large $N$.
\end{block}
\end{frame}

\begin{frame}{Uniform bounds from DS gauge and trace inequalities}
\vspace{-0.6em}

\begin{block}{DS gauge trace bound (proved in Lean; Experimental file)}
Under $\sum_i (A^i)^\dagger A^i=I$,
\[
  \bigl|\tr((\mathrm{evalWord}(A,w))^L)\bigr| \le D
  \qquad (\forall w,\forall L).
\]
\end{block}

\begin{block}{Peeling lemma (proved in Lean; Experimental file)}
A dominance gap $\|\mu_k\|\le \|\mu_1\|\rho$ with $\rho<1$ plus uniform bounds
implies exponential decay for the leading coefficient.
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{PiAlgebra/CanonicalFormSep.lean} (behind \texttt{MPSLean.Experimental}).}
\end{frame}

\begin{frame}{From decay to exact equality: where injectivity and spectral theory enter}
\vspace{-0.6em}

\begin{block}{What the peeling step gives}
For suitable families of test words, one gets
\[
  \tr(T_1^L)=\tr(U_1^L)+O(\rho^L),
\]
where $T_1=\mathrm{evalWord}(A_1,w)$ and $U_1=\mathrm{evalWord}(B_1,w)$.
\end{block}

\begin{block}{What must be added (literature route)}
\begin{itemize}\itemsep0.25em
  \item Use primitivity to upgrade asymptotic identities to exact constraints.
  \item Use injectivity (spanning) to vary $w$ and obtain enough equations to force
  equality of all word traces.
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Inductive block removal}
\vspace{-0.6em}

\begin{block}{Induction template}
\begin{enumerate}\itemsep0.25em
  \item Identify the leading block(s) using $|\mu|$ dominance and mixed-transfer decay.
  \item Prove per-block SameMPV for those blocks.
  \item Subtract their contributions from the global identity and recurse on the remaining blocks.
\end{enumerate}
\end{block}

\begin{block}{Why this matches Cirac RMP}
This is the canonical-form block separation step in the proof of the
Fundamental Theorem of MPVs \cite{Cirac2017Annals,Cirac2021Matrix}.
\end{block}
\end{frame}

\section{Multi-block FT after separation: $\Pi$-algebra and block permutations}
%======================================================================

\begin{frame}{Once separated: per-block SameMPV $\Rightarrow$ per-block gauge}
\vspace{-0.6em}

\begin{block}{Reuse of the single-block FT}
If $\forall k,\ \mathrm{SameMPV}(A_k,B_k)$ and each $A_k$ is injective, then
\[
  \forall k,\ \exists X_k\in\GLC_{D_k}(\mathbb{C}),\quad B_k^i=X_k A_k^i X_k^{-1}.
\]
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: apply \texttt{fundamentalTheorem\_singleBlock} blockwise.}
\end{frame}

\begin{frame}{Global block-diagonal tensor and MPV sum formula (Lean)}
\vspace{-0.6em}

\begin{block}{Constructor}
\[
  (\mathrm{toTensorFromBlocks}(\mu,A))^i := \bigoplus_{k=1}^r \mu_k A_k^i.
\]
\end{block}

\begin{block}{MPV coefficient decomposes as a block sum}
\[
  \mathrm{mpv}(\mathrm{toTensorFromBlocks}(\mu,A),\sigma)
  = \sum_{k=1}^r \mu_k^N\,\mathrm{mpv}(A_k,\sigma).
\]
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{MPS/FundamentalTheoremMulti.lean}.}
\end{frame}

\begin{frame}{Why the product algebra $\Pi=\prod_k \Mat_{D_k}(\mathbb{C})$ appears}
\vspace{-0.6em}

\begin{block}{Object}
\[
  \Pi := \prod_{k=1}^r \Mat_{D_k}(\mathbb{C}).
\]
\end{block}

\begin{block}{Interpretation}
Multi-block gauge is an automorphism of $\Pi$:
blockwise conjugations plus the possibility of permuting isomorphic blocks.
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{PiAlgebra/Construction.lean}.}
\end{frame}

\begin{frame}{Decomposition of $\Pi$-automorphisms: permutation $+\,$inner conjugations}
\vspace{-0.6em}

\begin{block}{Theorem (proved in Lean)}
Any $\mathbb{C}$-algebra automorphism of $\Pi$ decomposes into:
\begin{enumerate}\itemsep0.2em
  \item a block permutation $\pi$ (preserving dimensions),
  \item inner automorphisms on each block: $M\mapsto X_k M X_k^{-1}$.
\end{enumerate}
\end{block}

\textcolor{dimwhite}{\footnotesize Lean: \texttt{algEquiv\_pi\_matrix\_decomposition} in \texttt{MPS/BlockPermutationMPS.lean}.}
\end{frame}

  \item Input: \texttt{SameMPV}\textsubscript{2} for block-diagonal tensors, plus a separation hypothesis

\section{Extensions and current repository status}

\begin{frame}{Blocking-length bounds (motivation rather than formalization)}
\vspace{-0.6em}

\begin{block}{Why bounds matter in physics proofs}
Quantitative results (e.g. parent Hamiltonians, correlation length bounds)
use explicit blocking lengths to guarantee injectivity/primivity at finite scale.
Examples include RG fixed points \cite{Cirac2017Annals} and uncle Hamiltonians \cite{Fernandez2015Frustration}.
\end{block}

\begin{block}{Lean status}
The library formalizes the qualitative definitions (injective / normal / primitive),
but does not yet track explicit best-known bounds such as quantum Wielandt \cite{Sanz2010Quantum}.
\end{block}
\end{frame}

\begin{frame}{Irreducible / periodic generalization (not covered yet)}
\vspace{-0.6em}

\begin{block}{What changes beyond primitivity \cite{Cirac2017Irreducible}}
If the transfer operator has period $p>1$, canonical form is replaced by
\emph{irreducible form}. The fundamental theorem then allows an extra diagonal/unitary
matrix $Z$ commuting with the tensor:
\[
  Z A^i = Y B^i Y^{-1},\qquad [Z,A^i]=0,\qquad \mathcal{V}(A)=\mathcal{V}(ZA).
\]
\end{block}

\begin{block}{Lean status}
Periodic/irreducible forms and the diagonal $Z$ freedom are not yet formalized.
\end{block}
\end{frame}
