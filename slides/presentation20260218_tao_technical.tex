\begin{frame}{Fundamental Theorem: gauge is the only ambiguity}
    \vspace{-0.6em}

    \begin{block}{Injectivity}
        $A$ is \emph{injective} if $\Span_{\mathbb{C}}\{A^i\}=\Mat_D(\mathbb{C})$.

\begin{frame}{Precise Lean theorem statement}
    \vspace{-0.6em}

    \begin{block}{Theorem (\texttt{fundamentalTheorem\_canonicalForm\_explicit})}
        \textbf{Given:}
        \begin{itemize}\itemsep0.05em
            \item Weights $\mu_0,\ldots,\mu_{r-1}\in\mathbb{C}^\times$ with
                  \alert{$\|\mu_0\|>\|\mu_1\|>\cdots>\|\mu_{r-1}\|$}
                  \textcolor{dimwhite}{\footnotesize(simplifying; \cite{Cirac2017Annals} allows repeated magnitudes via BNT).}.
            \item Block tensors $A_k, B_k$ ($k=0,\ldots,r{-}1$), each $A_k^i\in\Mat_{D_k}(\mathbb{C})$.
            \item \alert{Injectivity}: $\Span\{A_k^i\}=\Mat_{D_k}$ and $\Span\{B_k^i\}=\Mat_{D_k}$ for all $k$.
            \item \alert{DS gauge}: $\sum_i (A_k^i)^\dagger A_k^i = I$ and $\sum_i (B_k^i)^\dagger B_k^i = I$.
            \item \alert{Primitivity}: $\Ov(A_k,A_k,N)\xrightarrow{N\to\infty} 1$ for all~$k$.
            \item \textbf{Global MPV equality}:
                  $\mathrm{mpv}\!(\bigoplus_k\mu_k A_k, \sigma)
                      =\mathrm{mpv}\!(\bigoplus_k\mu_k B_k, \sigma)$
                  for all~$N,\sigma$.
        \end{itemize}
        \textbf{Then:} $\exists\, X_k\in\GL_{D_k}(\mathbb{C}),  B_k^i = X_k A_k^i X_k^{-1}   \forall\, k,i$.
    \end{block}

    \textcolor{dimwhite}{\footnotesize Canonical form is \emph{assumed}, not constructed.
        Both $A$ and $B$ share the same block structure.}
\end{frame}

\begin{frame}{Fixed point $\Rightarrow$ invariant projection (Lean: verified)}
    \vspace{-0.5em}

    \begin{block}{Missing link: how to \emph{find} the invariant projection}
        The two Lean lemmas above require an \emph{input} projection $P$
        satisfying $(1{-}P)A^i P = 0$.
        The canonical-form iteration must \alert{construct} such $P$ automatically.
    \end{block}

    \begin{columns}[T]
        \column{0.52\textwidth}
        \begin{block}{Key lemma (Lean: verified)}
            \textbf{Statement:}
            Let $\mathcal{E}_A(\rho) = \sum_i A^i\,\rho\,(A^i)^\dagger$ be the transfer map.
            Suppose $\rho \ge 0$, $\mathcal{E}_A(\rho) = \rho$, and $\rho$ is
            \emph{not} full-rank. Then the support projection
            $P := \mathrm{supp}(\rho)$ satisfies
            \[
                (1 - P)\,A^i\,P = 0 \quad \forall\,i,
            \]
            i.e.\ $P\mathcal{H}$ is a nontrivial $A$-invariant subspace.
        \end{block}

        \begin{block}{Proof sketch}
            The positivity $\rho \ge 0$ and channel equation give
            $P\rho P = \rho$.
            Each summand $A^i\rho(A^i)^\dagger \ge 0$ has support $\subseteq P\mathcal{H}$,
            forcing $(1{-}P)A^i P = 0$ for every~$i$
            (a Kraus-rank argument; see~\cite{Perez-Garcia2007Matrix} Thm.~3,
            TeX lines 765--770).
        \end{block}

        \column{0.46\textwidth}
        \begin{block}{Literature attribution}
            \begin{itemize}\itemsep0.18em
                \item \textbf{\cite{Perez-Garcia2007Matrix} Thm.~3}
                      (quant-ph/0608197, TeX lines 765--770):
                      first explicit proof; fixed point $\rho_R$ (non-invertible)
                      $\Rightarrow$ $P_R = \mathrm{supp}(\rho_R)$ satisfies
                      $A^i P_R = P_R A^i P_R$.
                \item \textbf{\cite{Cirac2017Annals} \S2.3}
                      (arXiv:1606.00608):
                      same construction; the existence of such $\rho$ follows
                      from quantum Perron--Frobenius when the transfer map
                      is reducible.
            \end{itemize}
        \end{block}

\begin{frame}{Quantum Wielandt: injectivity after blocking}
    \vspace{-0.6em}

    \begin{block}{Idea~\cite{Sanz2010Quantum,Cirac2017Annals}}
        For a primitive transfer operator, there is a finite blocking length~$L$
        after which products of length~$L$ span the full matrix algebra.
        A coarse bound:
        \[
            L \le D^4 \quad \text{(quantum Wielandt)};\qquad
            \text{block-injectivity after }\le 3D^5.
        \]
    \end{block}

    \begin{block}{Lean status}
        The library works with explicit hypotheses
        (\texttt{IsNBlkInjective}, \texttt{IsNormal}, \texttt{IsInjective})
        rather than quantitative bounds.
    \end{block}
\end{frame}

\begin{frame}{Equal-or-orthogonal: the mixed transfer dichotomy}
    \vspace{-0.6em}

    \begin{block}{Mixed transfer operator~\cite{Perez-Garcia2007Matrix,Cirac2021Matrix}}
        For injective blocks, define
        \[
            \cF_{jk}(X) := \sum_{i=1}^d A_j^i\,X\,(A_k^i)^\dagger.
        \]
    \end{block}

    \begin{block}{Key dichotomy}
        \begin{itemize}\itemsep0.2em
            \item $\rho(\cF_{jk})=1$ $\Longrightarrow$
                  blocks are gauge-phase equivalent.
            \item $\rho(\cF_{jk})<1$ $\Longrightarrow$
                  cross-terms decay exponentially (asymptotic orthogonality).
        \end{itemize}
    \end{block}

    \textcolor{dimwhite}{\footnotesize This is the engine behind block separation: inequivalent blocks decouple at large~$N$.}
\end{frame}

\begin{frame}{Step~(ii): FT in canonical form}
    \vspace{-0.6em}

    \begin{block}{Theorem (proportional MPVs; simplified)~\cite{Cirac2017Annals,Cirac2021Matrix}}
        Let $A$ and~$B$ be in canonical form.
        If $\ket{V^{(N)}(A)}$ and $\ket{V^{(N)}(B)}$ are proportional for all~$N$,
        then (after relabeling) each block $B_j$ is gauge-phase equivalent
        to some~$A_{\pi(j)}$.
    \end{block}

    \begin{block}{Corollary (equal MPVs)}
        If $\mathcal{V}(A)=\mathcal{V}(B)$ exactly, phases cancel
        and one obtains per-block gauge equivalence plus a block permutation.
    \end{block}

    \textcolor{dimwhite}{\footnotesize Our Lean formalization proves the \emph{equal} case (Corollary) with canonical form assumed.}
\end{frame}

\begin{frame}{Why canonical form is essential}
    \vspace{-0.6em}

    \begin{block}{Ambiguity without normalization}
        Rescaling $A_k^i \mapsto c\, A_k^i$ while compensating $\mu_k \mapsto c^{-1}\mu_k$
        preserves the block-diagonal tensor. This freedom breaks naive separation.
    \end{block}

    \begin{block}{How normalization helps~\cite{Cirac2017Annals,Cirac2021Matrix}}
        Canonical form ties block scale to transfer-operator data
        (e.g.\ $\rho(\cE_{A_k})=1$), so the only allowed rescalings are phases.
        A strict $|\mu_k|$ ordering then pins a unique labeling.
    \end{block}
\end{frame}

\begin{frame}{Injectivity: spanning vs.\ inverse tensor}
    \vspace{-0.6em}

    \begin{block}{Literature formulation~\cite{Cirac2017Annals,Perez-Garcia2007Matrix}}
        $A$ is injective when products $\{A^{i_1}\cdots A^{i_L}\}$ span $\Mat_D(\mathbb{C})$
        for some~$L$.
        Equivalently, an ``inverse tensor'' $A^{-1}$ exists satisfying
        \[
            \sum_{i_1,\ldots,i_L} (A^{i_1}\cdots A^{i_L})_{\alpha\beta}\,(A^{-1})^{i_1\cdots i_L}_{\alpha'\beta'}
            = \delta_{\alpha\alpha'}\delta_{\beta\beta'}.
        \]
    \end{block}

    \begin{block}{Lean formulation}
        \texttt{IsInjective A}: the span of $\{A^i\}$ equals $\Mat_D(\mathbb{C})$.\\
        \texttt{IsNBlkInjective}: the $L$-step spanning condition.
    \end{block}
\end{frame}

\begin{frame}{Step 1: intertwiner from equal MPVs}
    \vspace{-0.6em}

    \begin{block}{Idea~\cite{Perez-Garcia2007Matrix,Cirac2021Matrix}}
        $\mathcal{V}(A)=\mathcal{V}(B)$ with $A$ injective implies
        $\exists\, X\neq 0$ such that $B^i X = X A^i$ for all~$i$.
    \end{block}

    \begin{block}{Lean route}
        From injectivity, construct a unique $\mathbb{C}$-linear map
        $T : \Mat_D(\mathbb{C})\to \Mat_D(\mathbb{C})$ with $T(A^i)=B^i$.

        \textcolor{dimwhite}{\footnotesize If $T(M)=XMX^{-1}$, then $T(A^i)=B^i$ is exactly $B^i X = X A^i$.}
    \end{block}

    \textcolor{dimwhite}{\footnotesize Source: \texttt{MPS/LinearExtension.lean}.}
\end{frame}

\begin{frame}{Step 2: trace pairing forces uniqueness}
    \vspace{-0.6em}

    \begin{block}{Key construction}
        Define
        $\Phi_A : \Mat_D(\mathbb{C}) \to \mathbb{C}^d$, \quad $(\Phi_A(M))_i := \tr(M A^i)$.

        Injectivity of $A$ forces $\ker(\Phi_A)=\{0\}$ (nondegenerate trace pairing).
        SameMPV supplies enough trace identities to determine~$T$ uniquely via~$\Phi_A$.
    \end{block}

    \textcolor{dimwhite}{\footnotesize Source: \texttt{Algebra/TracePairing.lean}, \texttt{MPS/LinearExtension.lean}.}
\end{frame}

\begin{frame}{Step 3: multiplicativity from length-3 words}
    \vspace{-0.6em}

    \begin{block}{Key lemma}
        From $\tr(A^i A^j A^k)=\tr(B^i B^j B^k)$ for all $i,j,k$, conclude
        \[
            T(MN)=T(M)\,T(N)\qquad \forall\, M,N\in \Mat_D(\mathbb{C}).
        \]
    \end{block}

    \begin{block}{Proof idea}
        Write $M = \sum_i \alpha_i A^i$ and $N = \sum_j \beta_j A^j$ using injectivity (spanning).
        Then $T(MN) = T\!(\sum_{i,j} \alpha_i \beta_j A^i A^j) = \sum_{i,j} \alpha_i \beta_j T(A^i A^j)$.
        The length-3 trace identity shows $T(A^i A^j) = B^i B^j = T(A^i)T(A^j)$.
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: \texttt{linearExtension\_mul} in \texttt{MPS/LinearExtension.lean}.}
\end{frame}

\begin{frame}{Step 4: simplicity + Skolem--Noether closes the proof}
    \vspace{-0.6em}

    \begin{block}{Simplicity $\Rightarrow$ bijectivity}
        $T$ is linear and multiplicative, so $\ker(T)$ is a two-sided ideal of $\Mat_D(\mathbb{C})$.
        Since this algebra is simple, $\ker(T)=0$ (a trace argument rules out $T=0$).
        Finite dimension gives surjectivity.
    \end{block}

    \begin{block}{Skolem--Noether $\Rightarrow$ inner automorphism}
        Every $\mathbb{C}$-algebra automorphism of $\Mat_D(\mathbb{C})$
        has the form $M\mapsto XMX^{-1}$ for some invertible~$X$.
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: \texttt{Algebra/SkolemNoether.lean} (wraps Mathlib's Skolem--Noether).
        The only nontrivial external input is Skolem--Noether (already in Mathlib).
        No analytic or spectral assumptions needed for the single-block case.\\
        \textbf{Deviation from~\cite{Cirac2017Annals}:} Appendix~A derives the same conclusion via a CFII + Cauchy--Schwarz argument (Lemma~equalMPS), without Skolem--Noether.}
\end{frame}

\begin{frame}{Kadison--Schwarz inequality~\cite{Wolf2012Quantum}}
    \vspace{-0.6em}

    \begin{block}{Theorem (Operator Schwarz / Kadison--Schwarz inequality; Wolf Ch.~5, Thm.~5.3)}
        For a unital completely positive map $\cE$:
        \[
            \cE(X^\dagger X) \ge  \cE(X)^\dagger\,\cE(X)
            \qquad \text{(L\"{o}wner order)}.
        \]
    \end{block}

    \begin{block}{Proof idea (channel-theoretic)}
        Form the $2\times 2$ block matrix
        $[\begin{smallmatrix} X^\dagger X & X^\dagger \\ X & I \end{smallmatrix}] =
            [\begin{smallmatrix} X^\dagger \\ I \end{smallmatrix}]
            [\begin{smallmatrix} X & I \end{smallmatrix}] \ge 0$.
        Apply $\cE$ blockwise (2-positivity). The Schur complement of the $(2,2)$-block
        gives $\cE(X^\dagger X) - \cE(X)^\dagger \cE(X) \ge 0$.
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: \texttt{kadison\_schwarz} in \texttt{Channel/KadisonSchwarz.lean} (sorry-free).\\
        \textbf{Deviation from~\cite{Cirac2017Annals}:} Appendix~A proves the same rigidity step by a direct Cauchy--Schwarz argument weighted by the CFII fixed point $\Lambda$ (no explicit KS).}
\end{frame}

\begin{frame}{KS equality: peripheral eigenvectors are multiplicative}
    \vspace{-0.6em}

    \begin{block}{Multiplicative domain~\cite{Wolf2012Quantum}}
        For a unital CP map $\cE$ with Kraus operators $\{K_i\}$, KS equality
        \[
            \cE(X^\dagger X) = \cE(X)^\dagger\,\cE(X)
        \]
        holds iff $K_i X = \cE(X)\, K_i$ for all~$i$ (\alert{intertwining with Kraus operators}).
    \end{block}

    \begin{block}{Weighted KS equality: peripheral eigenvectors~\cite{Cirac2017Annals}}
        Assume $\cE = \sum_i K_i(\cdot)K_i^\dagger$ is \alert{unital} and
        $\exists\,\rho\succ 0$ with $\cE^\dagger(\rho)=\rho$ (adjoint PF fixed point).
        If $\cE(M)=\mu M$, $|\mu|=1$:
        \begin{enumerate}\itemsep0.1em
            \item KS gives $\cE(M^\dagger M) - \cE(M)^\dagger\cE(M) \ge 0$ (PSD gap).
            \item Weighted trace: $\tr\!\bigl(\rho\,[\cE(M^\dagger M){-}\cE(M)^\dagger\cE(M)]\bigr)=0$
                  \;(uses $\cE^\dagger(\rho){=}\rho$, $\bar\mu\mu{=}1$).
            \item Faithfulness: PSD gap $+$ zero $\rho$-trace $\Rightarrow$ gap $= 0$.
        \end{enumerate}
        \textcolor{leangreen}{Peripheral eigenvalues force intertwining with Kraus operators.}\\
        \textcolor{dimwhite}{\footnotesize DS special case: $\rho=I$ (TP gauge). Generic injective MPS: $\rho={}$PF right fixed pt ($\neq I$ in general).%
        % SOURCE: Kraus.ks_equality_of_peripheral_eigenvector_of_fixedPoint in Channel/Schwarz.lean
        \\Lean: \texttt{ks\_equality\_of\_peripheral\_eigenvector\_of\_fixedPoint} (\texttt{Channel/Schwarz.lean}).}
    \end{block}
\end{frame}

\begin{frame}{HS contraction and eigenvalue bound}
    \vspace{-0.6em}

    \begin{block}{$\rho$-weighted HS contraction (from weighted KS equality)}
        If $\cE$ is unital and $\cE^\dagger(\rho)=\rho$ with $\rho\succ 0$:
        \[
            \tr\!\bigl(\rho\,\cE(X)^\dagger\cE(X)\bigr) \le  \tr\!\bigl(\rho\, X^\dagger X\bigr)
            \quad (\rho\text{-weighted HS contraction}).
        \]
        \textcolor{dimwhite}{\footnotesize DS special case: $\rho=I$ (TP gauge) gives $\|\cE(X)\|_{\HS}\le\|X\|_{\HS}$.
        \cite{Cirac2017Annals} App.~A uses the PF fixed point $\Lambda$ in exactly this role.}
    \end{block}

    \begin{block}{Eigenvalue bound for the mixed transfer}
        From the iterated word expansion and Cauchy--Schwarz:
        $\|\cF_{A,B}^n(X)\|_{\HS}^2 \le D^2\,\|X\|_{\HS}^2$ for all~$n$.
        If $\cF_{A,B}(X)=\mu X$ with $X\neq 0$, then
        $|\mu|^{2n} \|X\|_{\HS}^2 \le D^2\,\|X\|_{\HS}^2$.
        Taking $n\to\infty$ forces $|\mu|\le 1$.
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: \texttt{eigenvalue\_norm\_le\_one} in \texttt{Spectral/SpectralGap.lean}.}
\end{frame}

\begin{frame}{Eigenvector rigidity: $|\mu|=1$ implies gauge equivalence}
    \vspace{-0.6em}

    \begin{block}{Theorem (the hard algebraic step)~\cite{Wolf2012Quantum}}
        If $A,B$ are injective and normalized, and $\cF_{A,B}(X) = \mu X$ with $|\mu|=1$,
        $X\neq 0$, then $\exists\,Y\in\GLC_D(\mathbb{C})$ such that $B^i = \bar\mu\, Y A^i Y^{-1}$.
    \end{block}

    \begin{block}{Proof outline (channel-theoretic route)}
        \begin{enumerate}\itemsep0.1em
            \item \textbf{PF/unital gauge}: $A'_i = S_A^{-1}A_i S_A$ with $S_A S_A^\dagger = \rho_A$
                  (PF right fixed pt), making $\cE_{A'}$ unital and $\cE_{A'}^\dagger$ having PD fixed pt
                  $\rho_A$; likewise $B'_i$. (Not generally TP; DS iff $\rho=I$.)
            \item \textbf{Block-embed}: form $K_i = \diag(A'_i, B'_i)$ on $\mathbb{C}^D \oplus \mathbb{C}^D$.
            \item \textbf{Weighted KS equality} for peripheral eigenvector $\Rightarrow$ intertwining:
                  $A'_i X' = \mu\, X' B'_i$.
            \item \textbf{Kernel invariance}: $\ker(X')$ is invariant under all $(B'_i)^\dagger$.
                  Injectivity forces $\ker(X') = 0$, so $X'$ is invertible.
            \item \textbf{Conclude}: $B'_i = \bar\mu\,(X')^{-1}A'_i X'$; gauge back.
        \end{enumerate}
    \end{block}
    \textcolor{dimwhite}{\footnotesize Lean gauge: \texttt{gauged\_unital} (\texttt{Channel/DSGauge.lean});
        weighted KS $+$ Kraus commutation: \texttt{Channel/Schwarz.lean}.}
\end{frame}

\begin{frame}{The block-embedding construction in detail}
    \vspace{-0.6em}

    \begin{block}{The key construction}
        Given $\cF_{A,B}(X) = \mu X$ with $|\mu|=1$, pass to PF/unital-gauged tensors
        $A'_i, B'_i$ ($\cE_{A'},\cE_{B'}$ unital via PF similarity; \texttt{Channel/DSGauge.lean})
        and define the \emph{block channel}:
        \[
            \mathcal{K}(Y) := \sum_{i} K_i\, Y\, K_i^\dagger, \qquad
            K_i := \begin{pmatrix} A'_i & 0 \\ 0 & B'_i \end{pmatrix}.
        \]
        Then $\mathcal{K}$ is a \alert{unital} CP map; the adjoint fixed point $\rho_A\oplus\rho_B\succ 0$
        enables the weighted KS argument.\\
        \textcolor{dimwhite}{\footnotesize (\cite{Cirac2017Annals} App.~A uses the PF fixed point $\Lambda$ analogously; not TP in general.)}
    \end{block}

    \begin{block}{Embedding the eigenvector}
        Set $M := (\begin{smallmatrix} 0 & X' \\ 0 & 0 \end{smallmatrix})$
        where $X' = S_A^{-1} X S_B^{-1}$.
        Then $\mathcal{K}(M) = \mu M$: a \emph{peripheral eigenvector}.
        By KS equality, $M$ is in the multiplicative domain, yielding:
        \[
            A'_i X' = \mu\, X' B'_i \quad \text{and} \quad
            X' (B'_i)^\dagger = \mu\, (A'_i)^\dagger X'.
        \]
    \end{block}
\end{frame}

\begin{frame}{From intertwining to invertibility: the kernel argument}
    \vspace{-0.6em}

    \begin{block}{Kernel invariance}
        From $X' (B'_k)^\dagger = \mu\,(A'_k)^\dagger X'$:
        if $X' v = 0$ then $X' (B'_k)^\dagger v = \mu\,(A'_k)^\dagger (X' v) = 0$.
        So \alert{$\ker(X')$ is invariant under every $(B'_k)^\dagger$}.
    \end{block}

    \begin{block}{Injectivity closes the argument}
        $B$ injective $\Rightarrow$ $\Span\{(B'_k)^\dagger\}$ generates $\Mat_D(\mathbb{C})$.
        An invariant subspace for the full matrix algebra is $\{0\}$ or~$\mathbb{C}^D$.
        Since $X'\neq 0$, we get $\ker(X')=\{0\}$, so $X'$ is invertible:
        \[
            B'_i = \bar\mu\,(X')^{-1}A'_i X'.
        \]
        Gauging back: $B_i = \bar\mu\, Y A_i Y^{-1}$ with $Y = S_B\,(X')^{-1} S_A^{-1}$.
    \end{block}
\end{frame}

\begin{frame}{Equal-or-orthogonal: the complete dichotomy}
    \vspace{-0.6em}

    \begin{block}{Key dichotomy~\cite{Perez-Garcia2007Matrix,Cirac2021Matrix}}
        For injective, normalized blocks $A,B$:
        \begin{itemize}\itemsep0.2em
            \item \textbf{Equivalent:}
                  $\rho(\cF_{A,B})=1$ implies $B^i = e^{i\phi}\, X A^i X^{-1}$.
            \item \textbf{Orthogonal:}
                  $\rho(\cF_{A,B})<1$ implies $\cF_{A,B}^N\to 0$ exponentially.
        \end{itemize}
    \end{block}

    \begin{block}{The logical chain}
        \[
            \text{KS ineq.} \Rightarrow  \|\cF\|_{\HS}\le 1
            \Rightarrow  \rho(\cF)\le 1
            \underset{|\mu|=1}{\Longrightarrow}  \text{gauge-phase equiv}
            \Rightarrow  \rho<1 \text{ if } \lnot\text{equiv}.
        \]
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: \texttt{spectralRadius\_mixedTransfer\_lt\_one} and
        \texttt{mixedTransfer\_pow\_tendsto\_zero} in \texttt{Spectral/SpectralGap.lean}.}
\end{frame}

\begin{frame}{The overlap--trace identity: bridging spectra and combinatorics}
    \vspace{-0.6em}

    \begin{block}{MPV overlap}
        \vspace{-0.5em}
        \[
            \Ov(A,B,N) :=  \sum_{\sigma:\mathrm{Fin}\,N\to[d]} \mathrm{mpv}(A,\sigma)\, \overline{\mathrm{mpv}(B,\sigma)}
            = \braket{V^{(N)}\!(A)|V^{(N)}\!(B)}.
        \]
    \end{block}

    \begin{block}{Key identity (proved in Lean)}
        \[
            \boxed{ \Ov(A,B,N) =  \tr(\cF_{A,B}^N) }
        \]
        \vspace{-0.8em}
        \begin{itemize}\itemsep0.15em
            \item \textbf{Spectral side:} $\tr(\cF^N)$ is controlled by eigenvalues of $\cF$.
            \item \textbf{Combinatorial side:} $\Ov(A,B,N)$ is the inner product in $\mathbb{C}^{d^N}$.
            \item Works for \alert{heterogeneous bond dimensions} $D_j \neq D_k$.
        \end{itemize}
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: \texttt{trace\_mixedTransferMap\_pow\_eq\_mpvOverlap}
        in \texttt{Spectral/MPVOverlapTraceRect.lean} (180 lines, sorry-free).}
\end{frame}

\begin{frame}{Proof of the overlap--trace identity}
    \vspace{-0.6em}

    \begin{block}{Proof idea}
        \begin{enumerate}\itemsep0.2em
            \item Expand the operator trace via matrix units:
                  $\tr(T) = \sum_{p,q} (T(e_{pq}))_{pq}$.
            \item Use the iterated word expansion:
                  $\cF^N(e_{pq}) = \sum_\sigma \mathrm{evalWord}(A,\sigma)\,e_{pq}\, \mathrm{evalWord}(B,\sigma)^\dagger$.
            \item Extract diagonal entries:
                  $(M\,e_{pq}\,N)_{pq} = M_{pp}\,N_{qq}$.
            \item Exchange summation order ($\sigma$ outermost) and factor:
                  \[
                      \sum_\sigma \left(\sum_p \mathrm{ev}(A,\sigma)_{pp}\right)
                      \left(\sum_q \overline{\mathrm{ev}(B,\sigma)_{qq}}\right)
                      = \sum_\sigma \mathrm{mpv}(A,\sigma)\, \overline{\mathrm{mpv}(B,\sigma)}.
                  \]
        \end{enumerate}
    \end{block}

    \textcolor{dimwhite}{\footnotesize Key helper: \texttt{entry\_mul\_single\_mul\_2} isolates individual matrix entries
        from the rank-1 matrix unit sandwich.}
\end{frame}

\begin{frame}{Overlap decay: spectral gap implies orthogonality}
    \vspace{-0.6em}

    \begin{block}{Theorem (proved in Lean)}
        If $A,B$ are injective, normalized, and $\lnot\,\mathrm{GaugePhaseEquiv}(A,B)$:
        \[
            \Ov(A,B,N) \xrightarrow{N\to\infty}  0.
        \]
    \end{block}

    \begin{block}{Proof chain}
        \begin{enumerate}\itemsep0.15em
            \item Rewrite: $\Ov(A,B,N) = \tr(\cF_{A,B}^N)$ (overlap--trace identity).
            \item Expand: $\tr(\cF^N) = \sum_{p,q} (\cF^N(e_{pq}))_{pq}$.
            \item Each entry $\to 0$: by $\cF^N(X)\to 0$ (spectral gap) composed with the
                  continuous evaluation functional $M\mapsto M_{pq}$.
            \item Finite sum of limits: $\mathrm{tendsto\_finset\_sum}$.
        \end{enumerate}
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: \texttt{mpvOverlap\_tendsto\_zero} in \texttt{Spectral/MPVOverlapDecay.lean} (sorry-free).}
\end{frame}

\section{Block separation: the proof}

\begin{frame}{The block separation problem}
    \vspace{-0.6em}

    \begin{block}{Setup}
        For $A^i=\bigoplus_{k=0}^{r-1} \mu_k A_k^i$, the MPV decomposes as
        \[
            \mathrm{mpv}(A,\sigma)=\sum_{k=0}^{r-1} \mu_k^N\,\mathrm{mpv}(A_k,\sigma).
        \]
        Global equality $\mathcal{V}(A)=\mathcal{V}(B)$ becomes
        \[
            \sum_{k=0}^{r-1} \mu_k^N\,\Delta_k(N,\sigma)=0,
            \qquad \Delta_k:=\mathrm{mpv}(A_k,\sigma)-\mathrm{mpv}(B_k,\sigma).
        \]
    \end{block}

    \begin{block}{Goal (\textcolor{leangreen}{proved})}
        Under canonical form hypotheses, $\Delta_k=0$ for every block~$k$.
    \end{block}
\end{frame}

\begin{frame}{Why naive Vandermonde fails}
    \vspace{-0.6em}

    \begin{block}{Two obstructions}
        \begin{enumerate}\itemsep0.25em
            \item \textbf{Type-varying coefficients.}
                  $\Delta_k(N,\sigma)$ depends on~$N$,
                  so there is no fixed vector for classical Vandermonde.
            \item \textbf{Block swap + rescale.}
                  Even with distinct $\mu_k$, swapping blocks and compensating scales
                  preserves the global MPV.
        \end{enumerate}
    \end{block}

    \begin{block}{Minimal counterexample (scalar blocks)}
        $r=2$, $D=1$, $\mu=(1,2)$: set $B_0^i=\tfrac{\mu_1}{\mu_0}A_1^i$,
        $B_1^i=\tfrac{\mu_0}{\mu_1}A_0^i$.
        Global MPVs agree, but $A_0\neq B_0$.
    \end{block}

    \textcolor{dimwhite}{\footnotesize Documented in \texttt{PiAlgebra/BlockSeparationProof.lean} (Experimental).
        The naive Vandermonde lemma there is \emph{known-false} (counterexample included).}
\end{frame}

\begin{frame}{How canonical form defeats the counterexample}
    \vspace{-0.6em}

    \begin{block}{Key insight}
        In canonical form, $\mu_k$ are fixed by transfer-operator normalization
        ($\rho(\cE_{A_k})=1$).
        Rescaling $A_k\mapsto cA_k$ is only allowed when $|c|=1$.
        The swap-and-rescale trick requires $|\mu_0|=|\mu_1|$,
        which \alert{strict ordering forbids}.
    \end{block}

    \begin{block}{Lean encoding (simplifying hypotheses)}
        \texttt{IsCanonicalForm} enforces DS-gauge normalization and a strict $\|\mu\|$ ordering
        \textcolor{dimwhite}{\footnotesize(sufficient for our peeling proof; \cite{Cirac2017Annals} uses BNT for the general case).}
    \end{block}
\end{frame}

\begin{frame}{Step 1: asymptotic dominance via peeling}
    \vspace{-0.6em}

    \begin{block}{Setup}
        From $\|\mu_0\|>\|\mu_1\|>\cdots>\|\mu_{r-1}\|$ and
        $\sum_{k=0}^{r-1} \mu_k^N\Delta_k=0$:
        \[
            \Delta_0(N,\sigma) = -\sum_{k=1}^{r-1} \left(\frac{\mu_k}{\mu_0}\right)^{\!N} \Delta_k(N,\sigma).
        \]
    \end{block}

    \begin{block}{Uniform bound from DS gauge}
        $\sum_i (A^i)^\dagger A^i=I$ implies
        $|\mathrm{mpv}(A,\sigma)| = |\tr(\mathrm{evalWord}(A,w))| \le D$.
    \end{block}

    \begin{block}{Exponential decay}
        Set $\rho := \max_{k\ge 1}|\mu_k/\mu_0|<1$. Then $|\Delta_0(N,\sigma)| \le C\,\rho^N \to 0$.
    \end{block}

    \textcolor{dimwhite}{\footnotesize \textbf{Deviation from~\cite{Cirac2017Annals}:} Appendix~A matches blocks via BNT + eventual linear independence (Cor.~Lem1), without strict $\|\mu\|$ ordering or peeling.}
\end{frame}

\begin{frame}{Step 2: the overlap test}
    \vspace{-0.6em}

    \begin{block}{Multiplying by test functions}
        Multiply $\sum_{k=0}^{r-1} \mu_k^N \Delta_k(N,\sigma) = 0$
        by $\overline{\mathrm{mpv}(A_0,\sigma)}$ and sum over $\sigma$:
        \[
            \sum_{k=0}^{r-1} (\bar\mu_k)^N [\Ov(A_0,A_k,N) - \Ov(A_0,B_k,N)] = 0.
        \]
    \end{block}

    \begin{block}{Apply peeling to the overlap identity}
        Same dominance: $|\bar\mu_0|>|\bar\mu_1|>\cdots$, so
        \[
            \bigl|\Ov(A_0,A_0,N) - \Ov(A_0,B_0,N)\bigr| \le  C\,\rho^N \to  0.
        \]
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: \texttt{peeling\_exponential\_bound} in \texttt{PiAlgebra/CanonicalFormSep.lean}.}
\end{frame}

\begin{frame}{Step 3: equal-or-orthogonal forces gauge equivalence}
    \vspace{-0.6em}

    \begin{block}{The two limits collide}
        \begin{itemize}\itemsep0.2em
            \item \textbf{Primitivity:}
                  $\Ov(A_0,A_0,N) \to 1$ \quad (hypothesis).
            \item \textbf{Peeling:}
                  $|\Ov(A_0,A_0,N) - \Ov(A_0,B_0,N)| \to 0$.
            \item \textbf{Therefore:} $\Ov(A_0,B_0,N) \to 1$.
        \end{itemize}
    \end{block}

    \begin{block}{Contradiction if $\lnot\,\mathrm{GaugePhaseEquiv}(A_0,B_0)$}
        Overlap decay gives $\Ov(A_0,B_0,N)\to 0$,
        but we just showed $\Ov(A_0,B_0,N)\to 1$.
        \alert{Uniqueness of limits}: $0 = 1$---contradiction
    \end{block}

    \begin{block}{Conclusion}
        $\mathrm{GaugePhaseEquiv}(A_0,B_0)$: $\exists\, \zeta$ with $|\zeta|=1$
        and invertible $X$ such that $B_0^i = \zeta\,X\,A_0^i\,X^{-1}$.
    \end{block}
\end{frame}

\begin{frame}{Step 4: phase cancellation ($\zeta = 1$)}
    \vspace{-0.6em}

    \begin{block}{Setup}
        We have $B_0^i = \zeta\,X\,A_0^i\,X^{-1}$ with $|\zeta|=1$.
        Gauge invariance of the MPV gives:
        \[
            \Ov(A_0,B_0,N) = \bar\zeta^N \cdot \Ov(A_0,A_0,N).
        \]
    \end{block}

    \begin{block}{Forcing $\zeta = 1$}
        \begin{itemize}\itemsep0.15em
            \item From primitivity: $\Ov(A_0,A_0,N) \to 1$.
            \item From peeling: $\Ov(A_0,B_0,N) \to 1$.
            \item Therefore $\bar\zeta^N \to 1$.
            \item But $|\bar\zeta^{N+1}-\bar\zeta^N|=|\bar\zeta-1|$ is constant.
                  Convergence forces $|\bar\zeta-1|=0$, so \alert{$\zeta = 1$}.
        \end{itemize}
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: uses \texttt{Filter.Tendsto.unique} (uniqueness of limits in $T_2$ spaces).}
\end{frame}

\begin{frame}{Newton--Girard: an algebraic alternative route}
    \vspace{-0.6em}

    \begin{block}{Theorem (proved in Lean, 316 lines)}
        For matrices $A, B$ over a characteristic-zero integral domain:
        \[
            (\forall\, k\ge 1,\  \tr(A^k)=\tr(B^k))
            \Longrightarrow
            \mathrm{charpoly}(A) = \mathrm{charpoly}(B).
        \]
    \end{block}

    \begin{block}{Proof via Jacobi's formula}
        \begin{enumerate}\itemsep0.1em
            \item Derive $P'(X) = -\tr(M\cdot\mathrm{adj}(I-XM))$
                  by differentiating $\det$ column-by-column.
            \item Extract the recursion: $(m{+}1)\cdot p_{m+1} = -\sum_{j=0}^m p_j\cdot\tr(M^{m+1-j})$.
            \item Strong induction on coefficients: equal power sums $\Rightarrow$
                  equal characteristic polynomial.
        \end{enumerate}
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: \texttt{charpoly\_eq\_of\_forall\_trace\_pow\_eq} in \texttt{Algebra/NewtonGirard.lean}.
        Not used in the main proof but provides an independent algebraic route.}
\end{frame}

\begin{frame}{Gram matrix linear independence}
    \vspace{-0.6em}

    \begin{block}{Theorem (proved in Lean, 73 lines)}
        If vectors $v_1(N),\ldots,v_r(N)$ in a Hilbert space satisfy
        \[
            \langle v_i(N),v_j(N)\rangle \xrightarrow{N\to\infty}  \delta_{ij},
        \]
        then for all sufficiently large~$N$, $\{v_1(N),\ldots,v_r(N)\}$ are linearly independent.
    \end{block}

    \begin{block}{Proof idea}
        The Gram matrix $G(N)_{ij} = \langle v_i, v_j\rangle \to I$.
        By continuity of $\det$: $\det G(N) \to 1 \neq 0$.
        Eventually $\det G(N) \neq 0$, which is equivalent to linear independence.
    \end{block}

    \textcolor{dimwhite}{\footnotesize Lean: \texttt{eventually\_linearIndependent\_of\_gram\_tendsto\_id}
        in \texttt{Algebra/GramMatrixLI.lean}.
        Corresponds to Corollary~(Lem1) in Appendix~A of~\cite{Cirac2017Annals} (BNT eventual linear independence).}
\end{frame}

\begin{frame}{Backup E: the counterexample that killed naive Vandermonde}
    \vspace{-0.6em}

    \begin{block}{Setup: $r=2$ blocks, $D=1$ (scalar), $\mu=(1,2)$}
        Let $A_0, A_1$ be any pair of scalar blocks ($A_k^i \in \mathbb{C}$).
        Define $B$ by swapping and rescaling:
        \[
            B_0^i := \frac{\mu_1}{\mu_0}\,A_1^i = 2\,A_1^i,
            \qquad
            B_1^i := \frac{\mu_0}{\mu_1}\,A_0^i = \tfrac{1}{2}\,A_0^i.
        \]
    \end{block}

    \begin{block}{Verification: global MPVs agree}
        \vspace{-0.5em}
        \[
            \sum_{k=0}^{1} \mu_k^N \mathrm{mpv}(B_k,\sigma)
            = 1^N \cdot 2^N A_1^{i_1}\cdots A_1^{i_N}
            + 2^N \cdot \tfrac{1}{2^N} A_0^{i_1}\cdots A_0^{i_N}
            = \sum_{k} \mu_k^N \mathrm{mpv}(A_k,\sigma).
        \]
        But $A_0 \neq B_0$ in general. The naive Vandermonde lemma is \textbf{false}.
    \end{block}

    \begin{block}{Why canonical form blocks this}
        DS normalization ($\rho(\cE_{A_k})=1$) fixes block scale.
        Swap-and-rescale requires $|\mu_0|=|\mu_1|$, contradicting strict ordering.
    \end{block}
\end{frame}
