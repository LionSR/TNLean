\begin{frame}{The Main Theorem: Fundamental Theorem of MPS}
    \vspace{-0.5em}

    \textbf{Theorem} \cite{Perez-Garcia2007Matrix}:
    {\small Let $A = (A_1,\ldots,A_d)$ and $B = (B_1,\ldots,B_d)$ be \emph{injective}
    MPS tensors of bond dimension $D$.
    If they generate the same matrix product vectors for all $N$:}
    \[
        \sum_{\sigma_1,\ldots,\sigma_N} \tr(A_{\sigma_1}\cdots A_{\sigma_N})
        |\sigma_1\cdots\sigma_N\rangle
        = 
        \sum_{\sigma_1,\ldots,\sigma_N} \tr(B_{\sigma_1}\cdots B_{\sigma_N})
        |\sigma_1\cdots\sigma_N\rangle
    \]
    {\small then they are \textbf{gauge equivalent}: $\exists\, X \in \GL(D,\mathbb{C})$
    such that $B_i = X A_i X^{-1}$ for all $i$.}

    \vspace{0.3em}
    \textbf{\textcolor{leangreen}{Single-block}}: proved, 0 sorry.\\
    \textbf{\textcolor{amber}{Multi-block version}}: adds block permutation
    (different blocks can be reordered); proved modulo 2 sorry in spectral theory.

% SLIDE 14 - Kadison--Schwarz Inequality

\begin{frame}{Kadison--Schwarz Inequality \textcolor{leangreen}{(209 lines, 0 sorry)}}
    \vspace{-0.5em}

    \textbf{Theorem} (\texttt{kadison\_schwarz}):
    {\small For a unital completely positive map $\Phi$:}
    \[
        \Phi(X)^\dagger\,\Phi(X) \leq  \Phi(X^\dagger X)
    \]

    \vspace{0.15em}
    \begin{columns}[T]
        \column{0.50\textwidth}
        \textbf{\textcolor{leangreen}{Proof strategy:}}
        \begin{enumerate}\itemsep0.1em\small
            \item Construct $2\times 2$ block matrix:
                $M = \begin{pmatrix} X^\dagger X & X^\dagger \\ X & I \end{pmatrix}$
            \item $M \geq 0$ by \textbf{Schur complement}:\\
                $I - X X^\dagger X^{-\dagger} \geq 0$?
                No --- use $M = N^\dagger N$.
            \item Apply $\Phi$ block-wise (CP $\Rightarrow$ positivity preserved).
            \item Extract $(2,2)$-Schur complement of result.
            \item Get $\Phi(X^\dagger X) - \Phi(X)^\dagger\Phi(X) \geq 0$.
        \end{enumerate}

        \column{0.46\textwidth}
        \textbf{\textcolor{leangreen}{Corollary} (HS contraction):}\\
        {\small For unital, trace-preserving CP maps:}
        \[
            \|\Phi(X)\|_{\mathrm{HS}} \leq  \|X\|_{\mathrm{HS}}
        \]
        {\scriptsize Apply KS with $\tr$:
        $\tr(\Phi(X)^\dagger\Phi(X)) \leq \tr(\Phi(X^\dagger X)) = \tr(X^\dagger X)$.}

        \vspace{0.3em}
        {\small\textbf{Status in Mathlib:}}
        \begin{itemize}\itemsep0.05em\scriptsize
            \item[\textcolor{softred}{$\times$}] \textbf{Not currently in Mathlib}
            \item[\textcolor{leangreen}{\checkmark}] Proved in TNLean
            \item[\textcolor{leangreen}{\checkmark}] Candidate for upstreaming
        \end{itemize}
    \end{columns}
\end{frame}
