% SLIDE 4 - Existence: Ces\`aro mean (our proof vs Wolf's Brouwer)

\begin{frame}{QPF Existence: Ces\`aro Mean \textnormal{(cf.\ \cite{Wolf2012Quantum} Thm 6.11)}}
    \vspace{-0.3em}

    \begin{columns}[T]
        \column{0.48\textwidth}
        \textbf{\textcolor{leangreen}{Our proof} (0 sorry):}
        \begin{enumerate}\itemsep0.1em\small
            \item Start with any density matrix $\rho_0 = \tfrac{1}{D}I$
            \item Ces\`aro means:
                $\sigma_N = \tfrac{1}{N}\sum_{n=0}^{N-1} \mathcal{E}^n(\rho_0)$
            \item Each $\sigma_N$ is a density matrix\\
                {\tiny PSD: convex comb of PSD. \quad Tr = 1: $\mathcal{E}$ is TP.}
            \item Compact $\Rightarrow$ convergent subsequence $\sigma_{N_k} \to \sigma$
            \item \textbf{Telescoping}: $\mathcal{E}(\sigma_N) - \sigma_N = \frac{\mathcal{E}^N(\rho_0) - \rho_0}{N} \to 0$
            \item $\Rightarrow$ $\mathcal{E}(\sigma) = \sigma$ \quad {\tiny (limit uniqueness)}
        \end{enumerate}

        \column{0.48\textwidth}
        \textbf{\textcolor{dimwhite}{Wolf's proof} (\cite{Wolf2012Quantum} Thm 6.11):}
        \begin{enumerate}\itemsep0.1em\small\color{dimwhite}
            \item $\mathcal{E}$ maps density matrices to density matrices\\
                {\tiny (positive + trace-preserving)}
            \item Density matrices form a compact convex set
            \item $\mathcal{E}$ is continuous
            \item Apply \textbf{Brouwer fixed-point theorem}:\\
                continuous self-map of compact convex set has a fixed point
        \end{enumerate}

        \vspace{0.3em}
        \textbf{\textcolor{amber}{Why we diverge:}}
        \begin{itemize}\itemsep0.1em\small
            \item Mathlib has Schauder--Tychonoff, but
                applying it to density matrices requires nontrivial
                compactness/convexity setup
            \item Ces\`aro argument is self-contained\\
                {\tiny Uses only: compactness, linearity, sequential limits}
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{QPF Positive Definiteness: Kernel Argument \textnormal{(cf.\ \cite{Wolf2012Quantum} Thm 6.3)}}
    \vspace{-0.3em}

    \begin{columns}[T]
        \column{0.48\textwidth}
        \textbf{\textcolor{leangreen}{Our proof} (0 sorry):}
        \begin{enumerate}\itemsep0.12em\small
            \item Suppose $\rho \geq 0$, $\rho \neq 0$, $\mathcal{E}(\rho) = \rho$
            \item If $\rho x = 0$, then from FP equation:\\
                $0 = x^* \rho\, x = \sum_i (A_i^\dagger x)^* \rho (A_i^\dagger x)$
            \item Each term $\geq 0$ (PSD), sum $= 0$\\
                $\Rightarrow$ $\rho\,(A_i^\dagger x) = 0$ for all $i$
            \item \textbf{Key}: $\{A_i\}$ spans $M_D$ (injectivity \cite{Perez-Garcia2007Matrix})\\
                $\Rightarrow$ $\{A_i^\dagger\}$ spans $M_D$\\
                $\Rightarrow$ $\rho \cdot (M x) = 0$ for all $M$
            \item Since $x \neq 0$: $M \mapsto Mx$ is surjective\\
                $\Rightarrow$ $\rho = 0$, contradicting $\rho \neq 0$
        \end{enumerate}

        \column{0.48\textwidth}
        \textbf{\textcolor{dimwhite}{Wolf's proof} (\cite{Wolf2012Quantum} Thm 6.3):}
        \begin{enumerate}\itemsep0.12em\small\color{dimwhite}
            \item Define $r(X) = \lambda_{\max}(X) / \lambda_{\min}(X)$\\
                {\tiny (ratio of extreme eigenvalues)}
            \item Show $r(\mathcal{E}(X)) \leq r(X)$ for PD $X$
            \item Apply $(I + \mathcal{E})^{d^2-1}$ to a PSD eigenvector
            \item Conclude PD via the ratio bound
        \end{enumerate}

        \vspace{0.3em}
        \textbf{\textcolor{amber}{Why we diverge:}}
        \begin{itemize}\itemsep0.1em\small
            \item Wolf's proof works for general \emph{irreducible} maps;
                ours only handles MPS injectivity
            \item Our kernel argument is shorter ($\sim$200 lines)
                but \emph{less general}
            \item \textbf{Also proved}: irreducibility version\\
                {\tiny via support projection $Q$ (closer to Wolf)}
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{QPF Uniqueness: Critical Scalar \textnormal{($\approx$ \cite{Wolf2012Quantum} Thm 6.15)}}
    \vspace{-0.5em}

    \textbf{Theorem}: Any two nonzero PSD fixed points of $\mathcal{E}_A$
    (with $A$ injective) are proportional.

    \vspace{0.15em}
    \textbf{\textcolor{leangreen}{Proof}} (same strategy as \cite{Wolf2012Quantum}):
    \begin{enumerate}\itemsep0.08em\small
        \item Both $\rho$ and $\sigma$ are PD (by the kernel argument above).
        \item Conjugate: $H := \rho^{-1/2}\,\sigma\,\rho^{-1/2}$ is PD.
            \quad {\tiny Via spectral decomp: $S = U\,\mathrm{diag}(\sqrt{\lambda})$, $B = U\,\mathrm{diag}(1/\sqrt{\lambda})$.}
        \item \textbf{Critical scalar}: $c_0 = \lambda_{\min}(H) > 0$.
        \item Define $\tau = \sigma - c_0\,\rho$.
            \quad {\tiny Spectral shift: $\tau = S(H - c_0 I)S^\dagger \geq 0$ but $\lambda_{\min} = 0$ $\Rightarrow$ not PD.}
        \item $\tau$ is a fixed point (linearity).
            \quad If $\tau \neq 0$: PD theorem forces PD --- \textbf{contradiction}!
        \item $\therefore$ $\tau = 0$, i.e., $\sigma = c_0\,\rho$. \qed
    \end{enumerate}

    \vspace{0.15em}
    \begin{center}

\begin{frame}{Spectral Bound: $\rho(\mathcal{F}_{AB}) \leq 1$ via HS Contraction
    \textnormal{(beyond Wolf)}}
    \vspace{-0.5em}

    \textbf{Setting}: the \emph{mixed} transfer $\mathcal{F}_{AB}(X) = \sum_i A_i X B_i^\dagger$
    for two (possibly different) MPS tensors. \textcolor{amber}{Not covered by \cite{Wolf2012Quantum} Ch.~6; argument follows \cite{Perez-Garcia2007Matrix}.}

    \vspace{0.15em}
    \textbf{\textcolor{leangreen}{Theorem}} (\texttt{eigenvalue\_norm\_le\_one}):
    Every eigenvalue $\mu$ of $\mathcal{F}_{AB}$ has $|\mu| \leq 1$.

    \vspace{0.15em}
    \textbf{Proof} ($\sim$200 new lines in \texttt{TransferSpectral.lean}):
    \begin{enumerate}\itemsep0.06em\small
        \item Embed $M_D(\mathbb{C}) \hookrightarrow \mathrm{EuclideanSpace} \mathbb{C} 
            (\mathrm{Fin} D \times \mathrm{Fin} D)$
            via \texttt{toES}, with $\|\texttt{toES}(M)\|^2 = \mathrm{frobSq}(M)$
        \item Expand: $\mathcal{F}^n(X) = \sum_\sigma w_A(\sigma) \cdot X \cdot w_B(\sigma)^\dagger$
            \quad {\tiny (\texttt{mixedTransferMap\_pow\_apply})}
        \item Triangle + Frob.\ submult: $\|\mathcal{F}^n(X)\|_F \leq \sum_\sigma \|w_A(\sigma)\|_F \cdot \|X\,w_B(\sigma)^\dagger\|_F$
        \item Cauchy--Schwarz on the word sum:
            $(\sum f_A \cdot f_B)^2 \leq (\sum f_A^2)(\sum f_B^2)$
        \item TP condition $\sum A_i^\dagger A_i = I$ iterated over words gives:
            \vspace{-0.2em}
            \[
                \textstyle\sum_\sigma \mathrm{frobSq}(w_A(\sigma)) = D, \qquad
                \sum_\sigma \mathrm{frobSq}(X \cdot w_B(\sigma)^\dagger) = \mathrm{frobSq}(X)
            \]
            \vspace{-1em}
        \item \textbf{Result}: $\mathrm{frobSq}(\mathcal{F}^n(X)) \leq D^2 \cdot \mathrm{frobSq}(X)$
            for all $n$. \quad {\small If $|\mu| > 1$: $|\mu|^{2n} \to \infty$, contradiction.}
    \end{enumerate}

    \vspace{0.15em}
    {\footnotesize\textcolor{amber}{Note}: this gives $|\mu| \leq 1$ (non-strict).
    The strict bound $\rho < 1$ requires the axiom (next slide).}
\end{frame}

\begin{frame}{Proof Walk-Through: Why $\mathcal{E}$ Has a Fixed Point}
    \vspace{-0.5em}

    {\small\textcolor{dimwhite}{Goal: find $\rho \geq 0$, $\rho \neq 0$, with
    $\mathcal{E}(\rho) = \rho$.
    \quad(\texttt{exists\_posSemidef\_fixedPoint}, 130 lines, 0 sorry)}}

    \vspace{0.4em}
    \textbf{Idea}: average the orbits and extract a limit.

    \begin{enumerate}\itemsep0.2em\small
        \item \textbf{Seed}: pick $\rho_0 = \frac{1}{D}I$
            \quad (any density matrix works).

        \item \textbf{Cesàro means}: $\displaystyle\sigma_N
            = \frac{1}{N}\sum_{n=0}^{N-1}\mathcal{E}^n(\rho_0)$.\\
            Each $\sigma_N$ is a density matrix:
            {\footnotesize PSD (convex combo of PSD), trace 1 ($\mathcal{E}$ is TP).}

        \item \textbf{Compactness}: density matrices live in a compact set
            (finite-dim, bounded trace).\\
            $\Rightarrow$ some subsequence $\sigma_{N_k}\to\sigma$.

        \item \textbf{Telescoping trick}: $\mathcal{E}(\sigma_N) - \sigma_N
            = \dfrac{\mathcal{E}^N(\rho_0) - \rho_0}{N}$.\\[3pt]
            Both terms in the numerator have bounded norm, so the
            difference $\to 0$.

        \item \textbf{Conclusion}: passing to the limit,
            $\mathcal{E}(\sigma) - \sigma = 0$, i.e.\ $\mathcal{E}(\sigma) = \sigma$.
    \end{enumerate}

    \vspace{0.3em}
    \begin{center}
    \begin{tikzpicture}
        \node[draw=leangreen,thick,rounded corners=4pt,fill=cardbg,text=white,
            text width=0.88\textwidth,align=center,inner sep=4pt,font=\small] {
            Self-contained argument: only uses compactness + linearity.\\
            {\scriptsize No Brouwer/Schauder FPT invoked (avoids nontrivial Mathlib plumbing).}
        };
    \end{tikzpicture}
    \end{center}
\end{frame}

\begin{frame}{Proof Walk-Through: Why the Fixed Point Is Positive Definite}
    \vspace{-0.7em}

    {\small\textcolor{dimwhite}{Goal: $\rho \geq 0$, $\mathcal{E}(\rho) = \rho$,
    $\rho \neq 0$ $\Longrightarrow$ $\rho > 0$ (all eigenvalues $> 0$).
    \quad(\texttt{posSemidef\_fixedPoint\_isPosDef}, 200 lines)}}

    \vspace{0.3em}
    \textbf{Idea}: if $\rho$ has a kernel, injectivity forces $\rho = 0$ --- contradiction.

    \begin{enumerate}\itemsep0.05em\small
        \item Suppose $\rho x = 0$ for some $x \neq 0$.

        \item From the fixed-point equation $\rho = \sum_i A_i\,\rho\, A_i^\dagger$:
            \vspace{-0.3em}
            \[
                0 = x^*\rho\, x = \sum_i \underbrace{(A_i^\dagger x)^*\;\rho\; (A_i^\dagger x)}_{\geq\, 0\;\text{(PSD)}}
            \]
            \vspace{-0.6em}
            Each term $\geq 0$ and sum $= 0$
            $\;\Rightarrow\;$ $\rho\,(A_i^\dagger x) = 0$ for \emph{every} $i$.

        \item \textbf{Injectivity} \cite{Perez-Garcia2007Matrix}: $\{A_i\}$ spans $\MD$
            $\;\Rightarrow\;$ $\{A_i^\dagger\}$ spans $\MD$.
            So $\rho\,(M x) = 0$ for \emph{every} matrix $M$.

        \item Since $x \neq 0$: $M \mapsto Mx$ is surjective on $\mathbb{C}^D$
            $\;\Rightarrow\;$ $\rho\, v = 0$ for all $v$
            $\;\Rightarrow\;$ $\rho = 0$.  \textbf{Contradiction.}
    \end{enumerate}

    \vspace{0.1em}
    \begin{center}
    \begin{tikzpicture}
        \node[draw=amber,thick,rounded corners=4pt,fill=cardbg,text=white,
            text width=0.88\textwidth,align=center,inner sep=3pt,font=\small] {
            This is where the \textbf{MPS injectivity} assumption pays
            off: it ``closes'' the kernel propagation in one step.
        };
    \end{tikzpicture}
    \end{center}
\end{frame}

\begin{frame}{Proof Walk-Through: Why the Fixed Point Is Unique}
    \vspace{-0.7em}

    {\small\textcolor{dimwhite}{Goal: any two nonzero PSD fixed points are proportional.
    \quad(\texttt{posSemidef\_fixedPoint\_unique}, 350 lines)}}

    \vspace{0.2em}
    \textbf{Idea}: construct a PSD-but-not-PD fixed point, contradicting the
    previous theorem.

    \begin{enumerate}\itemsep0em\small
        \item Let $\rho, \sigma$ both be nonzero PSD fixed points of $\mathcal{E}$.
            By the PD theorem, both are actually \textbf{positive definite}.

        \item Conjugate:
            $H = \rho^{-1/2}\,\sigma\,\rho^{-1/2}$.
            $H$ is PD.
            \textbf{Critical scalar}: $c_0 = \lambda_{\min}(H) > 0$.

        \item Define $\tau = \sigma - c_0\,\rho$.
            Since $\tau = \rho^{1/2}(H - c_0 I)\rho^{1/2}$:
            $\tau$ is \textbf{PSD} ($H - c_0 I \geq 0$) but \textbf{not PD}
            ($\lambda_{\min} = 0$).

        \item $\tau$ is a fixed point by linearity:
            $\mathcal{E}(\tau) = \mathcal{E}(\sigma) - c_0\,\mathcal{E}(\rho) = \sigma - c_0\,\rho = \tau$.

        \item If $\tau \neq 0$: the PD theorem says any nonzero PSD FP is PD.
            But $\tau$ is not PD --- \textbf{contradiction!}
            Therefore $\tau = 0$, i.e.\ $\sigma = c_0\,\rho$.\qed
    \end{enumerate}

    \vspace{0.1em}
    {\scriptsize\textcolor{dimwhite}{Bulk of 350 lines: bookkeeping around
    $\rho^{1/2}$, $\rho^{-1/2}$, and spectral decomposition
    ($S\,\mathrm{diag}(\lambda)\,S^\dagger$ round-trips).}}
\end{frame}

\begin{frame}{Proof Walk-Through: Every Eigenvalue of $\mathcal{F}_{AB}$ Has $|\mu| \leq 1$}
    \vspace{-0.5em}

    {\small\textcolor{dimwhite}{Goal: $\rho(\mathcal{F}_{AB}) \leq 1$.
    \quad(\texttt{eigenvalue\_norm\_le\_one}, 200 lines, 0 sorry)}}

    \vspace{0.3em}
    \textbf{Strategy}: show $\|\mathcal{F}^n(X)\|_F$ is uniformly bounded, then
    Gelfand says $\rho \leq 1$.

    \begin{enumerate}\itemsep0.12em\small
        \item Expand: $\mathcal{F}^n_{AB}(X) = \sum_{\sigma \in [d]^n}
            w_A(\sigma)\, X\, w_B(\sigma)^\dagger$
            \quad{\tiny where $w_A(\sigma) = A_{\sigma_1}\cdots A_{\sigma_n}$}

        \item \textbf{HS Cauchy--Schwarz}:
            \vspace{-0.3em}
            \[
                \|\mathcal{F}^n(X)\|_F^2
                \;\leq\;
                \underbrace{\biggl(\sum_\sigma \|w_A(\sigma)\|_F^2\biggr)}_{%
                    = \;\tr(\mathcal{E}_A^n(I))\;=\;D}
                \;\cdot\;
                \underbrace{\biggl(\sum_\sigma \|X\, w_B(\sigma)^\dagger\|_F^2\biggr)}_{%
                    \leq\; D \cdot \|X\|_F^2}
            \]
            \vspace{-0.6em}
            \begin{itemize}\footnotesize
                \item Left factor: iterated TP gives $\tr(\mathcal{E}^n(I)) = D$.
                \item Right factor: unital property
                    $\sum B_i^\dagger B_i = I$ iterated over words.
            \end{itemize}

        \item \textbf{Result}: $\|\mathcal{F}^n(X)\|_F \leq D \cdot \|X\|_F$
            for \emph{all} $n$.

        \item If $\mu$ is an eigenvalue with $|\mu| > 1$: pick eigenvector $X$,
            then $\|\mathcal{F}^n(X)\|_F = |\mu|^n\|X\|_F \to \infty$.
            Contradicts uniform bound.
    \end{enumerate}

    \vspace{0.2em}
    {\footnotesize\textcolor{dimwhite}{Technical: Lean uses \texttt{EuclideanSpace}
    embedding to access Frobenius norm, then norm-equivalence back to
    operator norm for Gelfand formula.
    Original idea: \cite{Perez-Garcia2007Matrix} Lemma 5.}}
\end{frame}

    \vspace{-0.5em}

    {\small\textcolor{dimwhite}{Goal: $|V_N(A)\rangle = |V_N(B)\rangle$ for all $N$
    $\Longrightarrow$ $B_i = U A_i U^\dagger$.
    \quad(\texttt{gauge\_equiv\_of\_sameMPV}, 0 sorry)}}

    \vspace{0.3em}
    \textbf{Idea}: extract intertwining matrices from boundary freedom,
    then use unique fixed point.

    \begin{enumerate}\itemsep0.12em\small
        \item \textbf{Same state $\Rightarrow$ intertwiner}:
            from $\sum_\sigma \tr(A_{\sigma_1}\cdots A_{\sigma_N}\,\alpha)
            = \sum_\sigma \tr(B_{\sigma_1}\cdots B_{\sigma_N}\,\beta)$
            for all $N$,\\
            extract an invertible $R$ with
            $R\,A_i = \frac{1}{x}\,B_i\,R$ for all $i$.

        \item \textbf{Trace-preserving $\Rightarrow$ $|x|^2 = 1$}:
            apply $\sum_i (\cdot)^\dagger (\cdot) = I$ to both sides.

        \item \textbf{Fixed point uniqueness $\Rightarrow$ unitarity}:
            $R R^\dagger$ is a PSD fixed point of $\mathcal{E}_B$.\\
            By QPF uniqueness: $R R^\dagger = c \cdot \rho_B$.
            With proper normalization, $R R^\dagger = I$.\\
            Set $U = R$. Then $B_i = x\, U A_i U^\dagger$
            with $|x| = 1$.

        \item \textbf{Algebra backbone}: the intertwiner extraction
            (step 1) uses
            \emph{Skolem--Noether}:\\
            {\footnotesize any algebra isomorphism $M_D \xrightarrow{\sim} M_D$ is inner,
            i.e.\ conjugation by an invertible.}\\
            {\footnotesize Formalized via Lean's \texttt{piAlgEquiv} machinery.}
    \end{enumerate}

            This is a \textbf{purely algebraic} proof --- no analysis, no limits,
            no spectral theory.\\
            {\scriptsize QPF is used only for the uniqueness step (``$RR^\dagger$
            is the unique fixed point'').}
