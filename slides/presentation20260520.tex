%---------- Frame 4: Why formalize ----------
\begin{frame}{Why formalize?}
    \begin{itemize}
        \item A proof assistant (\textbf{Lean~4}) checks every logical step against a tiny, trusted kernel.
        \item Recent landmark formalisations:
            \begin{itemize}
                \item[$\circ$] \textbf{Kepler conjecture}~\cite{Hales2017Formal}
                \item[$\circ$] \textbf{Liquid Tensor Experiment}~\cite{Commelin2023Liquid}
                \item[$\circ$] \textbf{Strong Prime Number Theorem}~\cite{MathInc2025Strong}
            \end{itemize}
        \item For research physics: PhysLean and a Lean verification of generalised quantum Stein's lemma~\cite{Meiburg2025Formalization}.
    \end{itemize}
    \vspace{0.6em}
    \textcolor{dimwhite}{\footnotesize The kernel verifies \emph{logic}, not \emph{intent}. That gap is the punchline of this talk.}
\end{frame}

%---------- Frame 5: physics is harder ----------
\begin{frame}{Why physics is harder than pure math, for now}
    \textcolor{leangreen}{\textbf{Mathlib is built primarily for pure mathematics so far.}}
    \vspace{0.6em}
    \begin{itemize}
        \item Mathlib lacks \textbf{quantum channel theory} as a bundled subject:
            \begin{itemize}
                \item[$\circ$] no irreducibility / primitivity predicates for CP maps;
                \item[$\circ$] no Perron--Frobenius for completely positive maps;
                \item[$\circ$] no Kadison--Schwarz / multiplicative-domain calculus.
            \end{itemize}
        \item Each of these must be built from scratch \emph{before} a literature step that takes one line can be formalised.
    \end{itemize}
    \vspace{0.6em}
    \textcolor{dimwhite}{\footnotesize Plus: implicit hypotheses in the paper that formalisation must lift into the statement.}
\end{frame}

\section{The fundamental theorem of MPS}
%======================================================================



%---------- Frame 8: Transfer operator ----------
\begin{frame}{The transfer operator $E_A$}
    \begin{block}{Definition}
        $E_A : \MD \to \MD$, \quad $E_A(X) \;=\; \sum_{i=0}^{d-1} A^i\, X\, (A^i)^\dagger$.
        \\[2pt]
        A CP map with Kraus operators $\{A^i\}$.
    \end{block}
    \vspace{0.5em}
    \begin{itemize}
        \item Long-distance correlations of $\{\ket{\psi_A^{(N)}}\}$ are controlled by the spectrum of $E_A$ --- spectral radius, peripheral eigenvalues, spectral gap.
        \item \textbf{Injective} and \textbf{normal} are technical regularity conditions on $\{A^i\}$ that we will need later.
    \end{itemize}
    \vspace{0.6em}
    \textcolor{dimwhite}{\footnotesize Quantum Wielandt~\cite{Sanz2010Quantum}: every normal $A$ becomes injective after blocking $L \leq D^2 - d + 1$ sites.}
\end{frame}

%---------- Frame 9: FT statement (absorbs why-FT-MPS motivation) ----------
\begin{frame}{The fundamental theorem (informal)}
    Two MPS tensors can describe identical states for many reasons --- gauge, embedding, multiplicities, block redundancy. FT-MPS classifies all of them.
    \vspace{0.4em}
    \begin{block}{Theorem~\cite{Perez-Garcia2007Matrix,Cirac2017Annals,Cirac2017Irreducible}}
        Let $A, B$ be MPS tensors with $\ket{\psi_A^{(N)}} = \ket{\psi_B^{(N)}}$ for all $N$. Then, after canonical-form reduction:
        \begin{itemize}
            \item the \textbf{block structure} of $A$ matches that of $B$ --- same number of blocks, same bond dimensions, same multiplicities;
            \item each pair of corresponding blocks is \textbf{gauge-equivalent} via some $X_k \in \GL_{D_k}(\mathbb{C})$.
        \end{itemize}
    \end{block}
    \vspace{0.3em}
    \begin{itemize}
        \item Special case (\textbf{injective} $A$): $B^i = X A^i X^{-1}$ for a single $X \in \GL_D(\mathbb{C})$.
        \item Downstream: parent Hamiltonians~\cite{Fannes1992Finitely}, 1D SPT classification.
    \end{itemize}
\end{frame}

%---------- Frame 10: First autonomous attempt ----------
\begin{frame}{First autonomous attempt: Skolem--Noether}
    \textbf{Test.} Ask the agent pool to prove FT-MPS with \emph{no} literature route specified.
    \vspace{0.4em}
    \begin{itemize}
        \item First non-trivial result returned: the \textbf{injective case}.
        \item Route chosen: \textbf{Skolem--Noether} --- not canonical forms.
        \item \emph{Every algebra automorphism of $\MD$ is inner} $\Rightarrow$ $\exists\, X \in \GL_D(\mathbb{C})$ with $\varphi(M) = X M X^{-1}$, hence
            \[ B^i \;=\; X A^i X^{-1}. \]
    \end{itemize}
    \vspace{0.4em}
    \textcolor{dimwhite}{\footnotesize \community{} A genuine surprise of autonomy: the system picked the proof route the \emph{library} prefers, not the one the \emph{paper} prefers.}
\end{frame}

%---------- Frame 15: Wielandt bound ----------
\begin{frame}{The quantum Wielandt bound}
    \begin{block}{Theorem (quantum Wielandt~\cite{Sanz2010Quantum})}
        If $E_A$ is primitive, then for some $L = O(D^2)$
        \[
            T_L(A) \;=\; \spn\{A^{i_1} A^{i_2} \cdots A^{i_L}\} \;=\; \MD .
        \]
    \end{block}
    \vspace{0.4em}
    Wielandt is what makes the canonical-form pipeline \emph{finite}: every normal block becomes injective after $O(D^2)$ blocking steps.
    \vspace{0.4em}
    \\
    \textcolor{dimwhite}{\footnotesize Code: \texttt{vecMulVec\_mem\_wordSpan\_of\_isPrimitivePaper\_of\_noninvertible\_eigenvector} (worked example assembled by scout-then-implement; see architecture section).}
\end{frame}

\section{Intent enforcement --- the real bottleneck}

%---------- Frame 26: The thesis ----------
\begin{frame}{The real bottleneck is not proof search}
    Until a few years ago, \emph{writing Lean proofs at all} was the bottleneck --- every step a manual exercise in syntax, tactic search, and library archaeology. Modern language models close that gap: they read a goal state, suggest tactics, look up Mathlib lemmas, and iterate against the type checker.
    \vspace{0.6em}
    \\
    What they do \emph{not} fix is the layer above syntax.
    \vspace{0.4em}
    \begin{block}{Thesis}
        Lean checks logic, not intent. A formally correct proof of a weaker, unintended statement passes the kernel just the same.
    \end{block}
    \vspace{0.4em}
    \community{Buzzard~\cite{Buzzard2025Formal}:} \emph{``Definitions are more dangerous than proofs.''}
\end{frame}

%---------- Frame 28: Normalisation parenthetical ----------
\begin{frame}{Vignette 1: a parenthetical that turned out to be load-bearing}
    \textbf{Setup.} Cirac et al.~\cite{Cirac2017Annals} normalise canonical-form weights as $|\mu_k| \leq 1$ with at least one $|\mu_1| = 1$. Inserted as a parenthetical \emph{``we will assume from now on''} on line 246 of the source. The formalisation silently dropped it.
    \vspace{0.3em}
    \begin{block}{The catch (paper--Lean cross-check)}
        \footnotesize
        Without the unit-dominant constraint, the proportional-FT power-sum identity
        \[
            \sum_q \mu_{j,q}^N \;=\; \sum_q (\nu_{j,q}\, e^{i\phi_j})^N
        \]
        does \emph{not} separate weights cleanly --- proportional-FT theorems stalled with no obvious local cause.
    \end{block}
    \vspace{0.2em}
    \textbf{Resolution.} 5-PR sequence \#1716 $\to$ \#1725 $\to$ \#1745 $\to$ \#1752 $\to$ \#1758 strengthened \texttt{IsBNTCanonicalForm} with the line-246 normalisation, regenerated the dominant-match API, and lifted unit-block existence to a hypothesis.
    \\
    \textcolor{amber}{\footnotesize \textbf{Buzzard, made concrete:} every stalled sorry was downstream of a single dropped parenthetical.}
\end{frame}

%---------- Frame 29: Doubly stochastic ----------
\begin{frame}{Vignette 2: the doubly-stochastic gauge}
    \textbf{Setup.} The system imposed a simultaneous unital + trace-preserving gauge on $E_A$ and proved a correct spectral-gap theorem under it.
    \vspace{0.3em}
    \begin{block}{The catch (blueprint round 1)}
        \[
            E_A(\Id_D) = \Id_D \quad\text{AND}\quad E_A^*(\Id_D) = \Id_D \quad\text{--- generically impossible.}
        \]
    \end{block}
    \vspace{0.2em}
    \textbf{Analogy.} Asking a generic matrix to be \emph{both} upper-triangular \emph{and} symmetric: only diagonal matrices satisfy both. Imposing left- and right-canonical gauges simultaneously confines $A$ to a measure-zero subset of MPS in exactly that way.
\end{frame}

%---------- B1: Skolem-Noether ----------
\begin{frame}{B1. Skolem--Noether, the precise statement}
    \begin{block}{Skolem--Noether theorem}
        Let $A$ be a finite-dimensional \emph{central simple} algebra over a field $k$, and let $B$ be a simple $k$-subalgebra of $A$. Any two $k$-algebra homomorphisms $B \to A$ are conjugate by an inner automorphism of $A$.
    \end{block}
    \vspace{0.4em}
    \textbf{Specialised to $\MD$.} Take $A = B = \MD$. Every $k$-algebra automorphism $\varphi : \MD \to \MD$ is inner: $\exists X \in \GL_D(\mathbb{C})$ with $\varphi(M) = X M X^{-1}$.
    \vspace{0.4em}
    \begin{itemize}
        \item In our setting: $\varphi$ is the linear extension of $A^i \mapsto B^i$; algebra-hom is forced by cyclic-trace identities; invertibility of $X$ is a side lemma.
        \item Mathlib homes: \texttt{Mathlib.RingTheory.SimpleModule}, \texttt{Mathlib.Algebra.Central.Defs}; our specialisation lives in \texttt{MPS/FundamentalTheorem/Basic.lean}.
    \end{itemize}
\end{frame}

%---------- B2: Kadison-Schwarz + multiplicative domain ----------
\begin{frame}{B2. Kadison--Schwarz and the multiplicative domain}
    \begin{block}{Kadison--Schwarz inequality}
        For a unital CP map $E : \MD \to \MD$ and every $X \in \MD$,
        \[
            E(X^\dagger X) \;\geq\; E(X)^\dagger E(X) \quad (\text{in PSD order}).
        \]
    \end{block}
    \begin{block}{Multiplicative domain}
        \[
            \mathrm{MD}(E) \;=\; \big\{\, X \in \MD : E(X^\dagger Y) = E(X)^\dagger E(Y) \text{ and } E(YX) = E(Y)E(X)\ \forall Y\,\big\}.
        \]
    \end{block}
    \textbf{Tightness $\Leftrightarrow$ membership.} For $X$ with $\|E(X)\| = \|X\|$ and the peripheral eigenvalue having $|\mu|=1$, KS becomes an equality $\Leftrightarrow$ $X \in \mathrm{MD}(E)$. This is the formal version of the side product mentioned in Vignette~2.
\end{frame}

%---------- B4: CFI vs CFII ----------
\begin{frame}{B4. CFI vs CFII --- the two canonical forms}
    \begin{itemize}
        \item \textbf{Canonical Form I (CFI):} block-diagonal modulo a periodic structure on each block. Reached by spectral decomposition + dropping off-diagonal contributions to the trace.
        \item \textbf{Normal CFI:} block $p$ further sites $\Rightarrow$ each block has a \emph{unique} peripheral eigenvalue (normal tensor).
        \item \textbf{Canonical Form II (CFII):} block $p\!\cdot\!k$ further sites with $k \leq D^2 - d + 1$ (quantum Wielandt) $\Rightarrow$ each block is injective.
    \end{itemize}
    \vspace{0.6em}
    \begin{block}{Total blocking factor}
        \[
            L \;=\; p \cdot k, \qquad p = \text{block period}, \qquad k \leq D^2 - d + 1.
        \]
    \end{block}
    \textcolor{dimwhite}{\footnotesize Downstream uses the looser $k \leq D^2$ since the sharp constant is off the FT-MPS critical path.}
\end{frame}

\begin{frame}{B12. Block-separation counterexample}

        $r = 2$ blocks, $\mu = (1,2)$, $d=1$, bond dims $(1,1)$:
        \[
            A_0 = 1,\ A_1 = \tfrac{3}{2}; \qquad B_0 = 3,\ B_1 = \tfrac{1}{2}
            \quad\Longrightarrow\quad

    \end{block}
    \vspace{0.2em}

%---------- B13: Block separation ----------
\begin{frame}{B13. Block separation}
    \begin{block}{Statement}
        Two CFII tensors $\{A_k\}$, $\{B_\ell\}$ generate the same MPS family iff a bijection $\sigma$ aligns them block-by-block; the spectral gap does the separation.
    \end{block}
    \vspace{0.5em}
    \begin{itemize}
        \item \textbf{Mechanism.} Blocks separate because each block's transfer operator has a different spectral radius --- or, at equal spectral radius, distinct peripheral structure.
        \item \textbf{Quantitative form.} Peeling lemma + exponential decay of inter-block overlap + induction on block count.
        \item One paper line $\to$ $\sim$\textbf{800} Lean lines.
    \end{itemize}
\end{frame}

%---------- B14: Peripheral eigenvalues ----------
\begin{frame}{B14. Peripheral eigenvalues: a false generalisation}
    \textbf{Claim (wrong).} Peripheral eigenvalues of \emph{any} CP map are roots of unity --- true for irreducible channels (Evans--H{\o}egh-Krohn; cf.~\cite{Wolf2012Quantum}), false in general.
    \vspace{0.4em}
    \begin{block}{Direct counterexample}
        \footnotesize
        Take $E = E_1 \oplus E_2$ with peripheral spectra $\{e^{i\alpha}\}$, $\{e^{i\beta}\}$ for $\alpha/\beta \notin \mathbb{Q}$. The peripheral set is $\{e^{i\alpha}, e^{i\beta}\}$ --- not a finite group, hence not all roots of unity.
    \end{block}
    \vspace{0.3em}
    \textbf{Separate point --- peripheral eigenvalues can \emph{interact}:}
    \[
        a_N \;=\; 1 + (-1)^N \;\;\text{(constant-modulus summands, no limit).}
    \]
    This is what breaks naive Ces\`aro-type arguments even when each individual peripheral eigenvalue is harmless.
    \vspace{0.3em}
    \\
    \textbf{Resolution.} Caught by a review agent in blueprint round 2, before any Lean attempt.
\end{frame}
