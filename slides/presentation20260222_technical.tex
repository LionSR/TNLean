\section{Background: Matrix Product States}
%======================================================================

\begin{frame}{Matrix Product Vectors (MPV)}
    \begin{columns}[T]
        \column{0.55\textwidth}
        \textbf{Definition:}
        Given a family of matrices $A = (A^k)_{k=1}^d$ where $A^k \in \Mat_D(\mathbb{C})$, the MPV is the sequence of tensors $\ket{V_N(A)} \in (\mathbb{C}^d)^{\otimes N}$:
        \[
            c_{i_1 \dots i_N} = \trace(A^{i_1} \dots A^{i_N})
        \]
        \[
            \ket{V_N(A)} = \sum_{i_1 \dots i_N} \trace(A^{i_1} \dots A^{i_N}) \ket{i_1 \dots i_N}
        \]

        \vspace{1em}
        \textbf{Applications:}
        \begin{itemize}
            \item Ground states of 1D gapped Hamiltonians
            \item Efficient representation of quantum states
        \end{itemize}

        \column{0.4\textwidth}
        \begin{center}
            \begin{tikzpicture}[
                tn/.style={draw=leangreen, fill=cardbg, rounded corners=2pt, minimum size=6mm, text=white},
                bond/.style={thick, white},
                phys/.style={thick, leangreen!80!white}

\end{frame}

\begin{frame}{The Gauge Freedom}
    The map $A \mapsto \ket{V_N(A)}$ is \textbf{not injective}.
    
    \vspace{1em}
    \begin{block}{Gauge Transformation}
        If $X \in \mathrm{GL}_D(\mathbb{C})$ is invertible:
        \[
            B^k = X A^k X^{-1} \implies \trace(\prod B^{i_j}) = \trace(X (\prod A^{i_j}) X^{-1}) = \trace(\prod A^{i_j})
        \]
        Thus $A$ and $B$ generate the \emph{same} MPV.
    \end{block}

    \vspace{1em}
    \textbf{Question:} Is this the \emph{only} redundancy?
    \begin{itemize}
        \item If $\forall N, \ket{V_N(A)} = \ket{V_N(B)}$, are $A$ and $B$ gauge equivalent?
        \item Answer: \textbf{Generally No}. (e.g., $A = 0 \oplus A'$ vs $A'$)
        \item Answer: \textbf{Yes}, if we impose ``Canonical Forms''.
    \end{itemize}
\end{frame}

\begin{frame}{The Fundamental Theorem (Statement)}
    \textbf{Theorem 4.4 (Cirac et al., 2021) \cite{Cirac2021Matrix}:}
    
    Let $A$ and $B$ be two MPS tensors in \emph{canonical form}.
    If they generate the same states $\ket{V_N(A)} = \ket{V_N(B)}$ for all $N$, then there exist:
    \begin{itemize}
        \item A permutation $P$ of the blocks,
        \item Gauge matrices $X_j$ for each block,
        \item Phases $\phi_j$ (roots of unity),
    \end{itemize}
    such that $B$ is the gauge-transformed version of $A$ (up to block permutation and phases).

    \vspace{1em}
    \begin{center}
        \fcolorbox{leangreen}{darkbg}{\textcolor{leangreen}{\textbf{Goal: Formalize this fully in Lean 4.}}}
    \end{center}
\end{frame}

\begin{frame}{Key References}
    The formalization follows the ``modern'' algebraic approach:
    
    \begin{itemize}
        \item \textbf{\cite{Perez-Garcia2007Matrix}}: \emph{Matrix Product State Representations}.
        \item \textbf{\cite{Cirac2017Annals}}: \emph{Matrix Product Density Operators} (Foundational spectral theory).
        \item \textbf{\cite{Cirac2021Matrix}}: \emph{Matrix Product States and Projected Entangled Pair States} (The target theorem).
        \item \textbf{\cite{Sanz2010Quantum}}: Wielandt inequality bounds for primitivity.
    \end{itemize}

    We specifically target the \emph{Normal Form} (block-diagonal with primitive blocks) reduction.
\end{frame}

\begin{frame}{Why is this hard to formalize?}
    \begin{enumerate}
        \item \textbf{Linear Algebra overhead:} Block matrices, direct sums, tensor products, traces, and spectral theory in a typed proof assistant.
        \item \textbf{Analytic vs Algebraic:}
        \begin{itemize}
            \item ``For all $N$'' usually implies an asymptotic argument.
            \item ``Canonical Form'' is an algebraic reduction (Jordan blocks, etc.).
        \end{itemize}
        \item \textbf{Spectral Theory:}
        \begin{itemize}
            \item Need Perron-Frobenius theorem for CP maps.
            \item Need \emph{spectral gap} to prove overlap decay.
        \end{itemize}
        \item \textbf{Combinatorics:} Permuting blocks, matching eigenvalues.
    \end{enumerate}
\end{frame}

\begin{frame}{Pre-Formalization Landscape}
    \begin{itemize}
        \item Most physics proofs rely on ``hand-wavy'' linear algebra ("consider the Jordan form...").
        \item Ambiguities in ``generic'' position vs ``for all $N$''.
        \item Often assume $N \to \infty$ immediately without finite-$N$ bounds.
    \end{itemize}

    \textbf{In Lean:}
    We must be explicit.
    \begin{itemize}
        \item Defined \texttt{BlockTriangular} matrices explicitly.
        \item Defined \texttt{DirectSum} of tensors.
        \item Proved \texttt{SpectralGap} for primitive CP maps.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Definitions in Lean: The MPS Tensor}
    \begin{columns}
        \column{0.6\textwidth}
        \begin{lstlisting}[language=Lean]
abbrev MPSTensor (d D : ℕ) :=
  Fin d → Matrix (Fin D) (Fin D) ℂ

def mpv (A : MPSTensor d D) {N : ℕ}
    (σ : Fin N → Fin d) : ℂ :=
  coeff A (List.ofFn σ)
        \end{lstlisting}
        
        \vspace{0.5em}
        The core object is just a tuple of matrices.
        All properties (irreducibility, primitivity) are predicates on this structure.
        
        \column{0.35\textwidth}
        \textcolor{dimwhite}{
            We treat `MPSTensor` as a linear map
            $A: \mathbb{C}^D \to \mathbb{C}^d \otimes \mathbb{C}^D$
            or as a Kraus operator family.
        }
    \end{columns}
\end{frame}

\begin{frame}[fragile]{Definitions in Lean: Gauge Equivalence}
    \begin{lstlisting}[language=Lean]
def GaugeEquiv (A B : MPSTensor d D) : Prop :=
  ∃ X : GL (Fin D) ℂ,
    ∀ i : Fin d, B i = X * A i * X⁻¹
    \end{lstlisting}

    \vspace{1em}
    This is strict gauge equivalence. The theorem allows for:
    \begin{itemize}
        \item Block permutations (handled via \texttt{Fin rA}~$\simeq$~\texttt{Fin rB})
        \item Phasing ($e^{i\theta} A$)
    \end{itemize}
\end{frame}

\section{Proof Architecture}
%======================================================================

\begin{frame}{High-Level Strategy}
    \begin{center}
        \begin{tikzpicture}[
            node distance=1.2cm,
            process/.style={rectangle, draw=leangreen, fill=cardbg, text=white, text width=3cm, align=center, minimum height=1cm},
            arrow/.style={->, >=Stealth, thick, white}
        ]
            \node[process] (inputs) {Arbitrary MPS $A, B$};
            \node[process, right=of inputs] (canon) {Canonical Form\\(Block Diagonal)};
            \node[process, right=of canon] (primitive) {Primitive Blocks\\(Injectivity)};
            \node[process, right=of primitive] (final) {Global Gauge\\Equivalence};

            \draw[arrow] (inputs) -- node[above, font=\tiny] {Reduction} (canon);
            \draw[arrow] (canon) -- node[above, font=\tiny] {Decomp} (primitive);
            \draw[arrow] (primitive) -- node[above, font=\tiny] {Reassembly} (final);
        \end{tikzpicture}
    \end{center}

    \vspace{1em}
    \textbf{Three Logic Layers:}
    \begin{enumerate}
        \item \textbf{Single Block:} If $A, B$ are primitive and $\braket{V_N(A)|V_N(B)} \approx 1$, then $A \cong B$.
        \item \textbf{Block Separation:} Distinct eigenvalues $\implies$ orthogonal MPVs.
        \item \textbf{Combinatorics:} Matching blocks between $A$ and $B$.
    \end{enumerate}
\end{frame}

\begin{frame}{Step 1: Canonical Form Reduction}
    Any MPS tensor $A$ can be reduced to a form where:
    \begin{itemize}
        \item The transfer matrix $E = \sum_i A^i \otimes \bar{A}^i$ is block upper-triangular.
        \item The diagonal blocks are \emph{irreducible}.
        \item The MPVs depend \emph{only} on the diagonal blocks (nilpotent parts vanish in trace).
    \end{itemize}

    \vspace{0.5em}
    \textbf{Lean Implementation:}
    \texttt{MPSLean.MPS.CanonicalFormReduction}
    
    \begin{itemize}
        \item Uses orthogonal projection theory.
        \item Iteratively splits via invariant projections (strong induction on bond dimension).
        \item \textbf{Result:} We only need to prove the theorem for Block Diagonal MPS where each block is Irreducible.
    \end{itemize}
\end{frame}

\begin{frame}{Step 2: Spectral Gap (The Engine)}
    For an irreducible, primitive block $A$:
    \begin{itemize}
        \item The transfer operator $\mathcal{E}_A(X) = \sum_i A^i X A^{i\dagger}$ has a unique dominant eigenvalue $\rho$.
        \item The spectral gap $\gamma$ ensures convergence to the fixed point.
    \end{itemize}

    \begin{block}{Theorem: Spectral Gap}
        If $\mathcal{E}$ is primitive, the normalised power $\rho^{-N}\mathcal{E}^N$ converges
        to the fixed-point projector $\mathcal{P}$ (spectral radius of the remainder $< 1$).
    \end{block}

    \textbf{Lean:} `MPSLean.Spectral.SpectralGap`
    \begin{itemize}
        \item formalized via \texttt{Complex.abs} bounds on eigenvalues.
        \item used to prove $\braket{V_N|V_N}$ grows as $\rho^N$.
    \end{itemize}
\end{frame}

\begin{frame}{Step 3: Overlap Decay}
    Consider the overlap between two primitive MPS $A$ and $B$:
    \[
        O_N(A,B) = \frac{|\braket{V_N(A) | V_N(B)}|^2}{\braket{V_N(A)|V_N(A)}\braket{V_N(B)|V_N(B)}}
    \]

    \textbf{Key Lemma:}
    If $A$ and $B$ are \textbf{not} gauge equivalent (up to phase), then
    \[ O_N(A,B) \to 0 \quad \text{exponentially}. \]
    
    \textbf{Lean:} `MPSLean.Spectral.MPVOverlapDecay`
    \begin{itemize}
        \item Proves that if spectral radii differ, overlap decays.
        \item If radii match but no gauge equiv, the ``mixed'' transfer operator has spectral radius strictly less than $\sqrt{\rho_A \rho_B}$.
    \end{itemize}
\end{frame}

\begin{frame}{Step 4: Single-Block Fundamental Theorem}
    \textbf{Theorem:}
    If $A, B$ are primitive and generate ``proportional'' MPVs:
    \[ \ket{V_N(A)} = c_N \ket{V_N(B)} \]
    Then $A$ and $B$ are gauge equivalent (and $c_N = e^{i\theta N}$).

    \vspace{1em}
    \textbf{Proof mechanism in Lean:}
    \begin{enumerate}
        \item Construct the ``mixed'' transfer operator $\mathcal{E}_{AB}$.
        \item The overlap condition implies $\mathcal{E}_{AB}$ has spectral radius equal to geometric mean of $\mathcal{E}_A, \mathcal{E}_B$.
        \item This implies the existence of an invertible intertwiner $X$ (Sanz et al., 2010).
    \end{enumerate}
    
    File: `MPSLean.MPS.FundamentalTheoremProportional`
\end{frame}

\begin{frame}{Step 5: Multi-Block Matching (BNT)}
    Now consider $A = \bigoplus_k A_k$ and $B = \bigoplus_j B_j$.
    
    \begin{block}{Basis Normal Form (BNT)}
        We organize blocks such that:
        \begin{itemize}
            \item All $A_k$ are primitive.
            \item Spectral radii are sorted/grouped.
            \item No two blocks are gauge equivalent to each other (redundancy removal).
        \end{itemize}
    \end{block}

    \textbf{The Matching Logic:}
    If $\sum \ket{V_N(A_k)} = \sum \ket{V_N(B_j)}$, then by linear independence of MPVs from inequivalent primitive blocks (proved via overlap decay), there must be a permutation $\sigma$ such that:
    \[ \ket{V_N(A_k)} \approx \ket{V_N(B_{\sigma(k)})} \]
    
    This reduces the multi-block problem to $N$ single-block problems.
\end{frame}

\begin{frame}{Linear Independence of MPVs}
    A crucial intermediate result:
    
    \textbf{Theorem (`BlockSeparation`):}
    If $\{A_1, \dots, A_m\}$ are mutually inequivalent primitive MPS, then for large $N$, the vectors $\{\ket{V_N(A_k)}\}_k$ are \emph{linearly independent}.

    \vspace{0.5em}
    \emph{Proof sketch in Lean:}
    Use Vandermonde-like arguments on the eigenvalues of transfer operators. If spectral radii are distinct, one dominates. If they are equal, the ``mixed'' eigenvalues are sub-dominant, ensuring orthogonality.
\end{frame}

\begin{frame}{Grand Unified Argument}
    Combining it all:
    
    \begin{enumerate}
        \item \textbf{Input:} $A, B$ such that $\ket{V_N(A)} = \ket{V_N(B)}$.
        \item \textbf{Reduce:} $A \to \bigoplus A_i$, $B \to \bigoplus B_j$ (Canonical Form).
        \item \textbf{Cluster:} Group by spectral radius and gauge equivalence class.
        \item \textbf{Separate:} Use linear independence to show the sets of blocks must be identical (up to permutation).
        \item \textbf{Equate:} Apply Single-Block FT to each matched pair.
        \item \textbf{Conclude:} $A \cong B$.
    \end{enumerate}
    
    File: `MPSLean.MPS.FundamentalTheoremFull`
\end{frame}

\section{Gauge Choices \& Deviations}
%======================================================================

\begin{frame}{TP vs DS Gauge}
    \begin{itemize}
        \item \textbf{Trace Preserving (TP):} $\sum A^{i\dagger} A^i = \mathbb{I}$.
        \item \textbf{Doubly Stochastic (DS):} $\sum A^{i\dagger} A^i = \mathbb{I}$ AND $\sum A^i A^{i\dagger} = \mathbb{I}$ (if possible).
    \end{itemize}

    In the paper \cite{Cirac2021Matrix}, proofs often assume TP form.
    In Lean, we avoided assuming TP form globally.
    
    \textbf{Why?}
    Converting to TP requires taking square roots of positive matrices. While possible in Mathlib, it complicates the constructive nature.
    
    \textbf{Our Approach:}
    We work with "General Primitive" tensors and carry around the fixed-point matrices ($\Lambda_L, \Lambda_R$) explicitly.
    \texttt{MPSLean.QPF.PosDef} handles the positive-definiteness of these fixed points.
\end{frame}

\begin{frame}[fragile]{The \texttt{mu\_strict\_anti} Lemma}
    \begin{alertblock}{The Problem}
        Proving uniqueness of the spectral radius often requires monotonicity.
    \end{alertblock}

    We require strict ordering of the block weights by modulus.
    This is captured as a field of the \texttt{IsCanonicalForm} structure:

    \begin{lstlisting}[language=Lean]
-- Field of `IsCanonicalForm`:
mu_strict_anti :
  StrictAnti (fun k : Fin r => (*@$\|\mu(k)\|$@*))
    \end{lstlisting}

    This was critical for handling the ``overlap=1'' condition and ensuring distinct blocks produce orthogonal contributions.
\end{frame}

\begin{frame}{The ``Overlap $\to$ 1'' Assumption}
    Standard literature often states:
    "If overlap is 1, they are equivalent."
    
    In Lean, ``Overlap'' is a ratio of traces.
    We had to rigorously prove:
    \[ \lim_{N \to \infty} \frac{|\trace(E_{AB}^N)|^2}{\trace(E_A^N)\trace(E_B^N)} = 1 \iff \rho(E_{AB})^2 = \rho(E_A)\rho(E_B) \]
    
    This required detailed spectral analysis of the "Mixed Transfer Operator" $E_{AB} = \sum A^i \otimes \bar{B}^i$.
\end{frame}

\begin{frame}{Basis Normal Form (BNT)}
    The paper \cite{Cirac2021Matrix} discusses a ``Normal Form''. We formalized a strict version called \textbf{BNT} (Basis Normal Tensor).
    
    \begin{itemize}
        \item A `BNT` structure in Lean enforces that blocks are sorted and minimal.
        \item We defined a type `BNT` which bundles the list of blocks with proofs of their properties.
        \item This makes the final theorem statement cleaner:
        "Every MPS has a unique BNT representation."
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{The Complete Theorem}

theorem fundamentalTheorem_multiBlock_fromSameMPV(*@$_2$@*)

    (A B : (k : Fin r) → MPSTensor d (dim k))
    (hA : ∀ k, IsInjective (A k))
    (hSame : SameMPV(*@$_2$@*)

        (toTensorFromBlocks (*@$\mu$@*) B))
    (hSep : ∀ k, SameMPV (A k) (B k)) :
    (∀ k, GaugeEquiv (A k) (B k)) (*@$\land$@*)
    GaugeEquiv (toTensorFromBlocks (*@$\mu$@*) A)
               (toTensorFromBlocks (*@$\mu$@*) B) (*@$\land$@*)
    ∃ (σ : Fin r (*@$\simeq$@*) Fin r), ...
    \end{lstlisting}

    Gives per-block and global \texttt{GaugeEquiv} plus an explicit block-permutation
    decomposition of the Pi-algebra automorphism.
    
    This matches the physical intuition:
    \emph{"If two MPS generate the same state, their physical (minimal) degrees of freedom are identical."}
\end{frame}

\begin{frame}{Comparison with Cirac et al. 2021}
    \textbf{Paper Statement (Thm 4.4):}
    "Let $A, B$ be two MPS in canonical form... if they generate the same state... they are related by a gauge transformation, permutation, and phases."

    \textbf{Lean Statement:}
    Exactly matches, but with explicit type-theoretic handling of:
    \begin{itemize}
        \item The permutation (as an \texttt{Equiv}).
        \item The block structure (via \texttt{PiAlgebra}).
        \item The ``for all $N$'' quantifier (via asymptotic analysis).
    \end{itemize}
\end{frame}

\begin{frame}{1. Periodicity and Wielandt}
    Currently, we focus on \emph{aperiodic} (primitive) blocks.
    \begin{itemize}
        \item \textbf{Periodic MPS:} The transfer matrix has eigenvalues $e^{2\pi i k / p} \rho$.
        \item \textbf{Wielandt's Inequality:} We have bounds for primitivity indices, but explicitly handling the periodic case ($p > 1$) is the next generalization.
        \item Ref: de las Cuevas et al.\ (arXiv:1708.00029).
    \end{itemize}
\end{frame}

\begin{frame}{2. Convergence Rates}
    We proved $O_N \to 0$ exponentially.
    \begin{itemize}
        \item Can we extract explicit constants? $C, \gamma$?
        \item Useful for numerical tensor network algorithms (DMRG errors).
        \item Requires making the spectral gap bounds computable/explicit in Mathlib.
    \end{itemize}
\end{frame}

\begin{frame}{3. 2D Tensor Networks (PEPS)}
    The Fundamental Theorem for PEPS (2D) is much harder.
    \begin{itemize}
        \item No ``canonical form'' in general.
        \item Undecidable problems lurk.
        \item However, for \emph{injective} PEPS, a similar theorem holds.
        \item Our `PosMap` and `SpectralGap` libraries are reusable here.
    \end{itemize}
\end{frame}
