\section{Background: Matrix Product States}
%======================================================================

\begin{frame}{Matrix Product Vectors (MPV)}
    \begin{columns}[T]
        \column{0.55\textwidth}
        \textbf{Definition:}
        Given a family of matrices $A = (A^k)_{k=1}^d$ where $A^k \in \Mat_D(\mathbb{C})$, the MPV is the sequence of tensors $\ket{V_N(A)} \in (\mathbb{C}^d)^{\otimes N}$:
        \[
            c_{i_1 \dots i_N} = \trace(A^{i_1} \dots A^{i_N})
        \]
        \[
            \ket{V_N(A)} = \sum_{i_1 \dots i_N} \trace(A^{i_1} \dots A^{i_N}) \ket{i_1 \dots i_N}
        \]

        \vspace{1em}
        \textbf{Applications:}
        \begin{itemize}
            \item Ground states of 1D gapped Hamiltonians
            \item Efficient representation of quantum states
        \end{itemize}

        \column{0.4\textwidth}
        \begin{center}
            \begin{tikzpicture}[
                tn/.style={draw=leangreen, fill=cardbg, rounded corners=2pt, minimum size=6mm, text=white},
                bond/.style={thick, white},
                phys/.style={thick, leangreen!80!white}
            ]

        \end{center}
    \end{columns}
\end{frame}

\begin{frame}{The Gauge Freedom}
    The map $A \mapsto \ket{V_N(A)}$ is \textbf{not injective}.
    
    \vspace{1em}
    \begin{block}{Gauge Transformation}
        If $X \in \mathrm{GL}_D(\mathbb{C})$ is invertible:
        \[
            B^k = X A^k X^{-1} \implies \trace(\prod B^{i_j}) = \trace(X (\prod A^{i_j}) X^{-1}) = \trace(\prod A^{i_j})
        \]
        Thus $A$ and $B$ generate the \emph{same} MPV.
    \end{block}

    \vspace{1em}
    \textbf{Question:} Is this the \emph{only} redundancy?
    \begin{itemize}
        \item If $\forall N, \ket{V_N(A)} = \ket{V_N(B)}$, are $A$ and $B$ gauge equivalent?
        \item Answer: \textbf{Generally No}. (e.g., $A = 0 \oplus A'$ vs $A'$)
        \item Answer: \textbf{Yes}, if we impose ``Canonical Forms''.
    \end{itemize}
\end{frame}

\begin{frame}{The Fundamental Theorem (Statement)}
    \textbf{Theorem 4.4 (Cirac et al., 2021) \cite{Cirac2021Matrix}:}
    
    Let $A$ and $B$ be two MPS tensors in \emph{canonical form}.
    If they generate the same states $\ket{V_N(A)} = \ket{V_N(B)}$ for all $N$, then there exist:
    \begin{itemize}
        \item A permutation $P$ of the blocks,
        \item Gauge matrices $X_j$ for each block,
        \item Phases $\phi_j$ (roots of unity),
    \end{itemize}
    such that $B$ is the gauge-transformed version of $A$ (up to block permutation and phases).

    \vspace{1em}
    \begin{center}
        \fcolorbox{leangreen}{darkbg}{\textcolor{leangreen}{\textbf{Goal: Formalize this in Lean 4.}}}
    \end{center}
\end{frame}

\begin{frame}{Key References}
    The formalization follows the ``modern'' algebraic approach:
    
    \begin{itemize}
        \item \textbf{\cite{Perez-Garcia2007Matrix}}: \emph{Matrix Product State Representations}.
        \item \textbf{\cite{Cirac2017Annals}}: \emph{Matrix Product Density Operators} (Foundational spectral theory).
        \item \textbf{\cite{Cirac2021Matrix}}: \emph{Matrix Product States and Projected Entangled Pair States} (The target theorem).
        \item \textbf{\cite{Sanz2010Quantum}}: Wielandt inequality bounds for primitivity.
    \end{itemize}

    We specifically target the \emph{Normal Form} (block-diagonal with primitive blocks) reduction.
\end{frame}

\begin{frame}{Why is this hard to formalize?}
    \begin{enumerate}
        \item \textbf{Linear Algebra overhead:} Block matrices, direct sums, tensor products, traces, and spectral theory in a typed proof assistant.
        \item \textbf{Analytic vs Algebraic:}
        \begin{itemize}
            \item ``For all $N$'' usually implies an asymptotic argument.
            \item ``Canonical Form'' is an algebraic reduction (Jordan blocks, etc.).
        \end{itemize}
        \item \textbf{Spectral Theory:}
        \begin{itemize}
            \item Need Perron-Frobenius theorem for CP maps.
            \item Need \emph{spectral gap} to prove overlap decay.
        \end{itemize}
        \item \textbf{Combinatorics:} Permuting blocks, matching eigenvalues.
    \end{enumerate}
\end{frame}

\begin{frame}{Pre-Formalization Landscape}
    \begin{itemize}
        \item Most physics proofs rely on ``hand-wavy'' linear algebra ("consider the Jordan form...").
        \item Ambiguities in ``generic'' position vs ``for all $N$''.
        \item Often assume $N \to \infty$ immediately without finite-$N$ bounds.
    \end{itemize}

    \textbf{In Lean:}
    We must be explicit.
    \begin{itemize}
        \item Defined \texttt{BlockTriangular} matrices explicitly.
        \item Defined \texttt{DirectSum} of tensors.
        \item Proved \texttt{SpectralGap} for primitive CP maps.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Definitions in Lean: The MPS Tensor}
    \begin{columns}
        \column{0.6\textwidth}
        \begin{lstlisting}[language=Lean]
abbrev MPSTensor (d D : ℕ) :=
  Fin d → Matrix (Fin D) (Fin D) ℂ

def mpv (A : MPSTensor d D) {N : ℕ}
    (σ : Fin N → Fin d) : ℂ :=
  coeff A (List.ofFn σ)
        \end{lstlisting}
        
        \vspace{0.5em}
        The core object is just a tuple of matrices.
        All properties (irreducibility, primitivity) are predicates on this structure.
        
        \column{0.35\textwidth}
        \textcolor{dimwhite}{
            We treat `MPSTensor` as a linear map
            $A: \mathbb{C}^D \to \mathbb{C}^d \otimes \mathbb{C}^D$
            or as a Kraus operator family.
        }
    \end{columns}
\end{frame}

\begin{frame}[fragile]{Definitions in Lean: Gauge Equivalence}
    \begin{lstlisting}[language=Lean]
def GaugeEquiv (A B : MPSTensor d D) : Prop :=
  ∃ X : GL (Fin D) ℂ,
    ∀ i : Fin d, B i = X * A i * X⁻¹
    \end{lstlisting}

    \vspace{1em}
    This is strict gauge equivalence. The theorem allows for:
    \begin{itemize}
        \item Block permutations (handled via \texttt{Fin rA}~$\simeq$~\texttt{Fin rB})
        \item Phasing ($e^{i\theta} A$)
    \end{itemize}
\end{frame}

%======================================================================

\begin{frame}{High-Level Strategy}
    \begin{center}
        \begin{tikzpicture}[

        ]
            \node[process] (inputs) {Arbitrary MPS $A, B$};
            \node[process, right=of inputs] (canon) {Canonical Form\\(Block Diagonal)};
            \node[process, right=of canon] (primitive) {Primitive Blocks\\(Injectivity)};
            \node[process, right=of primitive] (final) {Global Gauge\\Equivalence};

            \draw[arrow] (inputs) -- node[above, font=\tiny] {Reduction} (canon);
            \draw[arrow] (canon) -- node[above, font=\tiny] {Decomp} (primitive);
            \draw[arrow] (primitive) -- node[above, font=\tiny] {Reassembly} (final);
        \end{tikzpicture}
    \end{center}

    \vspace{0.5em}
    \textbf{Three Logic Layers:}
    \begin{enumerate}
        \item \textbf{Single Block:} Primitive + overlap $\approx 1$ $\implies$ gauge equiv.
        \item \textbf{Block Separation:} Distinct eigenvalues $\implies$ orthogonal MPVs.
        \item \textbf{Combinatorics:} Matching blocks between $A$ and $B$.
    \end{enumerate}
\end{frame}

\begin{frame}{Step 1: Canonical Form Reduction}
    Any MPS tensor $A$ can be reduced to a form where:
    \begin{itemize}
        \item The transfer matrix $E = \sum_i A^i \otimes \bar{A}^i$ is block upper-triangular.
        \item The diagonal blocks are \emph{irreducible}.
        \item The MPVs depend \emph{only} on the diagonal blocks (nilpotent parts vanish in trace).
    \end{itemize}

    \vspace{0.5em}
    \textbf{Lean Implementation:}
    \texttt{TNLean.MPS.CanonicalFormReduction}
    
    \begin{itemize}
        \item Uses orthogonal projection theory.
        \item Iteratively splits via invariant projections (strong induction on bond dimension).
    \end{itemize}
\end{frame}

\begin{frame}{Step 2: Spectral Gap (The Engine)}
    For an irreducible, primitive block $A$:
    \begin{itemize}
        \item The transfer operator $\mathcal{E}_A(X) = \sum_i A^i X A^{i\dagger}$ has a unique dominant eigenvalue $\rho$.
        \item The spectral gap $\gamma$ ensures convergence to the fixed point.
    \end{itemize}

    \begin{block}{Theorem: Spectral Gap}
        If $\mathcal{E}$ is primitive, the normalised power $\rho^{-N}\mathcal{E}^N$ converges
        to the fixed-point projector $\mathcal{P}$ (spectral radius of the remainder $< 1$).
    \end{block}

    \textbf{Lean:} \texttt{TNLean.Spectral.SpectralGap}
    \begin{itemize}
        \item Formalized via \texttt{Complex.abs} bounds on eigenvalues.
        \item Used to prove $\braket{V_N|V_N}$ grows as $\rho^N$.
    \end{itemize}
\end{frame}

\begin{frame}{Step 3: Overlap Decay}
    Consider the overlap between two primitive MPS $A$ and $B$:
    \[
        O_N(A,B) = \frac{|\braket{V_N(A) | V_N(B)}|^2}{\braket{V_N(A)|V_N(A)}\braket{V_N(B)|V_N(B)}}
    \]

    \textbf{Key Lemma:}
    If $A$ and $B$ are \textbf{not} gauge equivalent (up to phase), then
    \[ O_N(A,B) \to 0 \quad \text{exponentially}. \]
    
    \textbf{Lean:} \texttt{TNLean.Spectral.MPVOverlapDecay}
    \begin{itemize}
        \item Proves that if spectral radii differ, overlap decays.
        \item If radii match but no gauge equiv, the ``mixed'' transfer operator has spectral radius strictly less than $\sqrt{\rho_A \rho_B}$.
    \end{itemize}
\end{frame}

\begin{frame}{Step 4: Single-Block Fundamental Theorem}
    \textbf{Theorem:}
    If $A, B$ are primitive and generate ``proportional'' MPVs:
    \[ \ket{V_N(A)} = c_N \ket{V_N(B)} \]
    Then $A$ and $B$ are gauge equivalent (and $c_N = e^{i\theta N}$).

    \vspace{1em}
    \textbf{Proof mechanism in Lean:}
    \begin{enumerate}
        \item Construct the ``mixed'' transfer operator $\mathcal{E}_{AB}$.
        \item The overlap condition implies $\mathcal{E}_{AB}$ has spectral radius equal to geometric mean of $\mathcal{E}_A, \mathcal{E}_B$.
        \item This implies the existence of an invertible intertwiner $X$ (Sanz et al., 2010).
    \end{enumerate}
    
    File: \texttt{TNLean.MPS.FundamentalTheoremProportional}
\end{frame}

\begin{frame}{Step 5: Multi-Block Matching (BNT)}
    \small
    Now consider $A = \bigoplus_k A_k$ and $B = \bigoplus_j B_j$.
    
    \begin{block}{Basis Normal Form (BNT)}
        We organize blocks such that all $A_k$ are primitive,
        spectral radii are sorted/grouped, and no two blocks are gauge equivalent
        (redundancy removal).
    \end{block}

    \textbf{The Matching Logic:}
    If $\sum \ket{V_N(A_k)} = \sum \ket{V_N(B_j)}$, then by linear independence of MPVs from inequivalent primitive blocks (proved via overlap decay), there must be a permutation $\sigma$ such that $\ket{V_N(A_k)} \approx \ket{V_N(B_{\sigma(k)})}$.
\end{frame}

\begin{frame}{Linear Independence of MPVs}
    A crucial intermediate result:
    
    \textbf{Theorem (\texttt{BlockSeparation}):}
    If $\{A_1, \dots, A_m\}$ are mutually inequivalent primitive MPS, then for large $N$, the vectors $\{\ket{V_N(A_k)}\}_k$ are \emph{linearly independent}.

    \vspace{0.5em}
    \emph{Proof sketch in Lean:}
    Use Vandermonde-like arguments on the eigenvalues of transfer operators. If spectral radii are distinct, one dominates. If they are equal, the ``mixed'' eigenvalues are sub-dominant, ensuring orthogonality.
\end{frame}

\begin{frame}{Grand Unified Argument}
    \small
    Combining it all:
    
    \begin{enumerate}
        \item \textbf{Input:} $A, B$ such that $\ket{V_N(A)} = \ket{V_N(B)}$.
        \item \textbf{Reduce:} $A \to \bigoplus A_i$, $B \to \bigoplus B_j$ (Canonical Form).
        \item \textbf{Separate:} Use linear independence to show the sets of blocks must be identical (up to permutation).
        \item \textbf{Equate:} Apply Single-Block FT to each matched pair.
        \item \textbf{Conclude:} $A \cong B$.
    \end{enumerate}
    
    File: \texttt{TNLean.MPS.FundamentalTheoremFull}
\end{frame}

\section{Gauge Choices \& Deviations from the Paper}
%======================================================================

\begin{frame}{TP vs DS Gauge}
    \begin{itemize}
        \item \textbf{Trace Preserving (TP):} $\sum A^{i\dagger} A^i = \mathbb{I}$.
        \item \textbf{Doubly Stochastic (DS):} $\sum A^{i\dagger} A^i = \mathbb{I}$ AND $\sum A^i A^{i\dagger} = \mathbb{I}$ (if possible).
    \end{itemize}

    In the paper \cite{Cirac2021Matrix}, proofs often assume TP form.
    In Lean, we avoided assuming TP form globally.
    
    \textbf{Why?}
    Converting to TP requires taking square roots of positive matrices. While possible in Mathlib, it complicates the constructive nature.
    
    \textbf{Our Approach:}
    We work with "General Primitive" tensors and carry around the fixed-point matrices ($\Lambda_L, \Lambda_R$) explicitly.
    \texttt{TNLean.QPF.PosDef} handles the positive-definiteness of these fixed points.
\end{frame}

\begin{frame}[fragile]{The \texttt{mu\_strict\_anti} Hypothesis}
    \begin{alertblock}{Stronger than the paper}
        The paper \cite{Cirac2021Matrix} allows degenerate $|\mu_k|$ values (blocks with equal spectral radii).
        Our formalization requires \emph{strict} ordering.
    \end{alertblock}

    We require strict ordering of the block weights by modulus.
    This is captured as a field of the \texttt{IsCanonicalForm} structure:

    \begin{lstlisting}[language=Lean]
-- Field of `IsCanonicalForm`:
mu_strict_anti :
  StrictAnti (fun k : Fin r => (*@$\|\mu(k)\|$@*))
    \end{lstlisting}

    This simplifies block separation but excludes cases where distinct
    blocks share the same spectral radius.
\end{frame}

\begin{frame}{The ``Overlap $\to$ 1'' Assumption}
    Standard literature often states:
    "If overlap is 1, they are equivalent."
    
    \begin{alertblock}{Assumed, not derived}
        In the Lean code, \texttt{overlap\_tendsto\_one} is taken as an \emph{explicit hypothesis}
        in \texttt{IsCanonicalForm}, not derived from primitivity of the transfer map.
        Deriving it requires assembling spectral-gap convergence with
        Perron-Frobenius normalization.
    \end{alertblock}

    The mathematical content we \emph{do} prove:
    \[ \frac{|\trace(E_{AB}^N)|^2}{\trace(E_A^N)\trace(E_B^N)} = 1 \iff \rho(E_{AB})^2 = \rho(E_A)\rho(E_B) \]
\end{frame}

\begin{frame}{The \texttt{IsCanonicalFormBNT} Predicate}
    The paper \cite{Cirac2021Matrix} discusses a ``Normal Form''.
    We formalized a strict version called \textbf{BNT} (Basis Normal Tensor).
    
    \begin{itemize}
        \item A \texttt{BNT} structure in Lean enforces that blocks are sorted and minimal.
        \item We defined a type \texttt{BNT} which bundles the list of blocks with proofs of their properties.
    \end{itemize}

    \begin{alertblock}{Extra hypothesis: \texttt{blocks\_not\_equiv}}
        Our \texttt{IsCanonicalFormBNT} adds an explicit
        \texttt{blocks\_not\_equiv} field (no two blocks are gauge equivalent)
        that is implicit/derived in the paper's presentation.
    \end{alertblock}
\end{frame}

    (∀ k, GaugeEquiv (A k) (B k)) (*@$\land$@*)
    GaugeEquiv (toTensorFromBlocks (*@$\mu$@*) A)
               (toTensorFromBlocks (*@$\mu$@*) B) (*@$\land$@*)
    ∃ (σ : Fin r (*@$\simeq$@*) Fin r), ...
    \end{lstlisting}

    Per-block and global \texttt{GaugeEquiv} plus an explicit block-permutation
    decomposition of the Pi-algebra automorphism.
\end{frame}

\begin{frame}{Comparison with Cirac et al.\ 2021}
    \textbf{Paper Statement (Thm 4.4):}
    "Let $A, B$ be two MPS in canonical form... if they generate the same state... they are related by a gauge transformation, permutation, and phases."

    \vspace{0.5em}
    \textbf{Lean Statement:} Covers the core content, with these \textbf{deviations}:
    \begin{itemize}
        \item \texttt{mu\_strict\_anti}: We require $|\mu_0| > |\mu_1| > \cdots$; the paper allows $|\mu_j| = |\mu_k|$ for $j \neq k$.
        \item \texttt{overlap\_tendsto\_one}: Taken as hypothesis; the paper derives it from primitivity.
        \item \texttt{blocks\_not\_equiv}: Extra hypothesis in \texttt{IsCanonicalFormBNT}.
        \item Coefficient convergence (\texttt{haCoeff}): Taken as explicit hypothesis, not derived from $|\mu_k| < |\mu_0|$.
        \item No periodic case (Thm 4.6): We handle only primitive (aperiodic) blocks.
        \item No quantitative Wielandt bound: Injectivity is an assumption, not derived from primitivity.
    \end{itemize}
\end{frame}

\section{Limitations \& Honest Assessment}
%======================================================================

\begin{frame}{What We Actually Proved}
    \textbf{Proved (sorry-free):}
    \begin{itemize}
        \item Single-block FT: primitive + injective $\implies$ gauge equiv (modulo phase).
        \item Multi-block FT: matching blocks under \texttt{IsCanonicalFormBNT}.
        \item Block separation via overlap decay.
        \item Spectral gap for primitive CP maps.
        \item Full library of CP-map, trace, and Kronecker infrastructure.
    \end{itemize}

    \vspace{0.5em}
    \textbf{Not proved / assumed:}
    \begin{itemize}
        \item \texttt{overlap\_tendsto\_one} is \emph{assumed} in \texttt{IsCanonicalForm}, not derived from primitivity.
        \item \texttt{mu\_strict\_anti} is \emph{stronger} than the paper (paper allows degenerate $|\mu|$).
        \item Coefficient convergence $\mu_k^N \to 0$ is taken as explicit hypothesis \texttt{haCoeff}.
        \item No periodicity handling (Thm 4.6 of \cite{Cirac2021Matrix}).
        \item No quantitative Wielandt bound.
    \end{itemize}
\end{frame}

\section{Next Steps: Detailed Roadmap}

\begin{frame}{Target 1: Coefficient Convergence (\texorpdfstring{$\sim$50--100}{~50-100} LOC)}
    \textbf{Mathematical statement:}
    For \texttt{IsCanonicalForm} with \texttt{mu\_strict\_anti}, the coefficients
    $c_j(N) = \sum_q \mu_{j,q}^N$ converge. In particular, $\mu_k^N \to 0$ for $k > 0$
    after normalizing $\mu_0 = 1$, since $|\mu_k| < |\mu_0|$.

    \vspace{0.5em}
    \textbf{Paper ref:} Implicit in \cite{Cirac2017Annals} (arXiv:1606.00608), Appendix~A proof of Thm~4.4.
    The coefficients are used without explicit convergence proof because the paper works with finite power sums.

    \vspace{0.5em}
    \textbf{Lean context:}
    \texttt{fundamentalTheorem\_proportionalMPV\_CFBNT} in \texttt{FundamentalTheoremFull.lean} currently takes
    \texttt{haCoeff : $\forall$ j, Tendsto (fun N $\Rightarrow$ aCoeff N j) atTop (nhds (aLim j))} as an explicit hypothesis.

    \vspace{0.5em}
    \textbf{What's needed:} Apply Mathlib's \texttt{tendsto\_pow\_atTop\_nhds\_zero\_of\_norm\_lt\_one} plus assembly. \textcolor{leangreen}{Easiest target.}
\end{frame}

\begin{frame}{Target 2: PF/$\mu$ Normalization (\texorpdfstring{$\sim$100--200}{~100-200} LOC)}
    \textbf{Mathematical statement:}
    For each irreducible block with transfer map $E_k$, the spectral radius $\rho(E_k)$
    is the Perron-Frobenius eigenvalue. Dividing the block tensor by $\rho(E_k)^{1/2}$
    normalizes the transfer map to spectral radius~1.

    \vspace{0.5em}
    \textbf{Paper ref:} \cite{Cirac2021Matrix} (arXiv:2011.12127) \S IV.A before Def~4.1;
    \cite{Cirac2017Annals} (arXiv:1606.00608) \S 2.3.

    \vspace{0.5em}
    \textbf{Lean context:}
    \texttt{IrreducibleFormII.lean} handles the unitary gauge (TP + diagonal PD fixed point).
    Missing: the spectral-radius normalization \emph{before} applying CF~II.

    \vspace{0.5em}
    \textbf{What's needed:} Show $\text{transferMap}(c \cdot A) = |c|^2 \cdot \text{transferMap}(A)$,
    then choose $c = \rho(E)^{-1/2}$ to normalize. \textcolor{amber}{Moderate difficulty.}
\end{frame}

\begin{frame}{Target 3: Periodic Cases / Thm 4.6 (\texorpdfstring{$\sim$300--600}{~300-600} LOC)}
    \small
    \textbf{Mathematical statement:}
    Without blocking to remove periodicity, two tensors $A, B$ with equal MPVs may differ by
    a diagonal matrix $Z$ with $[Z, A^i] = 0$ for all $i$ and $ZA^i = YB^i Y^{-1}$.
    The $Z$ arises from $p$-periodic blocks with eigenvalues $e^{2\pi iq/p}$.

    \vspace{0.3em}
    \textbf{Paper ref:} \cite{Cirac2021Matrix} Thm~4.6; \cite{Cirac2017Irreducible} for the full periodic version.

    \vspace{0.3em}
    \textbf{Lean context:}
    \texttt{MPS/Blocking.lean} has physical blocking (\texttt{blockTensor}, \texttt{mpv\_blockTensor\_eq\_mpv}).
    \texttt{Channel/Primitive.lean} has the P+N decomposition.
    Missing: peripheral spectrum theory, period detection, $Z$-matrix construction.

    \vspace{0.3em}
    \textbf{What's needed:}
    \begin{enumerate}
        \item Peripheral eigenvalues of irreducible CPTP maps are $p$-th roots of unity (Wolf Thm~6.6 \cite{Wolf2012Quantum}).
        \item Blocking by $p$ removes periodicity (makes transfer primitive).
        \item Define $Z$-matrix as block-diagonal with $p$-th roots.
        \item Compose with existing primitive FT.
    \end{enumerate}
    \textcolor{softred}{Substantial effort.}
\end{frame}

\begin{frame}{Target 4: Quantum Wielandt Bound (\texorpdfstring{$\sim$500--1000}{~500-1000} LOC)}
    \small
    \textbf{Mathematical statement:}
    For a primitive channel with Kraus rank $d$ and dimension $D$, the quantum Wielandt number
    $W$ (smallest $L$ such that length-$L$ products span $M_D(\mathbb{C})$) satisfies
    $W \leq (D^2 - d + 1)(D^2 - 1) + 1$; coarsely $W \leq D^4$.

    \vspace{0.3em}
    \textbf{Paper ref:} \cite{Sanz2010Quantum} (arXiv:0909.5347), Theorem~1 and Corollary~2.
    Also \cite{Wolf2012Quantum} Theorem~7.3.

    \vspace{0.3em}
    \textbf{Lean context:}
    The library uses \texttt{IsNormal A} ($\exists\, N$, \texttt{IsNBlkInjective A N}) and
    \texttt{IsInjective A} as explicit hypotheses.
    Wielandt would give: \texttt{IsPrimitive (transferMap A) $\to$ IsNormal A} with a quantitative bound.

    \vspace{0.3em}
    \textbf{What's needed:}
    \begin{enumerate}
        \item Define \texttt{IsPrimitive} for CPTP maps (trivial peripheral spectrum).
        \item Products of Kraus operators generate a $*$-subalgebra of $M_D(\mathbb{C})$.
        \item Burnside's theorem: irreducibly-acting $*$-subalgebra $= M_D(\mathbb{C})$.
        \item Dimension-counting argument for the $D^4$ bound.
    \end{enumerate}
    \textcolor{softred}{Hardest target; likely a separate project.}
\end{frame}
