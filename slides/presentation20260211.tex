\begin{frame}[fragile]{Key Proof: Injectivity $\Rightarrow$ Irreducible CP Map}
\vspace{-0.5em}
\begin{block}{Theorem (\texttt{injective\_implies\_irreducibleCP}, $\sim$80 lines)}
If $A$ is injective (its matrices span $\MD$), then the transfer map $\mathcal{E}_A$ is irreducible.
\end{block}

\vspace{-0.2em}
{\small\textbf{Proof strategy:} If projection $P$ satisfies $P \cdot \mathcal{E}_A(PXP) \cdot P = \mathcal{E}_A(PXP)$:}

\vspace{-0.2em}
\begin{center}
\begin{tikzpicture}[xscale=0.9,

\node[sbox,fill=leangreen!40!black] (s3) at (6.6,0) {Extend by\\spanning};
\node[sbox,fill=leangreen!50!black] (s4) at (9.9,0) {$P=0$ or\\$P=I$};
\draw[sarr] (s1) -- node[above,font=\tiny,text=amber]{PSD arg} (s2);
\draw[sarr] (s2) -- node[above,font=\tiny,text=amber]{$\{A_i\}$ span} (s3);

\end{tikzpicture}

\begin{lstlisting}[basicstyle=\ttfamily\fontsize{5}{6.5}\selectfont\color{white}]
theorem injective_implies_irreducibleCP (A : MPSTensor d D) (hA : IsInjective A) :
    IsIrreducibleCP (transferMap A) := by
  intro P hProj hInv
  -- Step 1: derive (1-P)*A(*@$_i$@*)*P = 0 for each i (PSD sandwich argument)
  have h_on_A := invariance_implies_complement_zero A P hProj hInv
  -- Step 2: extend to all matrices via span
  have h_all : (*@$\forall$@*) M, (1 - P) * M * P = 0 := by
    intro M
    have hM : M (*@$\in$@*) Submodule.span (*@$\mathbb{C}$@*) (Set.range A) := hA (*@$\blacktriangleright$@*) Submodule.mem_top
    induction hM using Submodule.span_induction with
    | mem x hx => obtain (*@$\langle$@*)i, rfl(*@$\rangle$@*) := hx; exact h_on_A i
    | zero => simp  | add x y _ _ hx hy => ... | smul c x _ hx => ...
  -- Step 3: conclude P = 0 or P = 1
  exact proj_zero_or_one_of_sandwich P h_all
\end{lstlisting}

    mixedTransferMap A B 1 = 0 := by
  by_cases hD : D = 0
  (*@$\cdot$@*) subst hD; ext i; exact i.elim0      -- D = 0: trivial
  (*@$\cdot$@*) exfalso                               -- D (*@$\ge$@*) 1: CONTRADICTION
    have h0 := h_cross 0                   -- h_cross at N = 0:
    simp at h0                             -- tr(F^0(I)) = tr(I) = D = 0
    exact hD h0                            -- but D (*@$\ge$@*) 1 !
\end{lstlisting}

\begin{frame}{Discovery: Missing Hypothesis (Phase Non-Degeneracy)}
\vspace{-0.3em}
{\small During the sprint, we identified a \textbf{missing hypothesis} in the standard physics proof:}

\vspace{0.3em}
\begin{block}{Phase non-degeneracy condition}
The ratios $\mu_k / \mu_\ell$ for $k \neq \ell$ must \textbf{not be roots of unity}.

Without this, blocks with $\mu_k / \mu_\ell = e^{2\pi i p/q}$ can interfere
constructively at system sizes that are multiples of $q$, potentially obstructing
the asymptotic separation argument.
\end{block}

\vspace{0.3em}
\begin{columns}[T]
\column{0.48\textwidth}
{\small\textbf{In the physics literature:}}
\begin{itemize}\itemsep0.1em
  \item {\small Often implicitly assumed}
  \item {\small Canonical form normalizes $|\mu_k| = 1$}
  \item {\small Distinct phases on the unit circle}
  \item {\small The paper \cite{Cirac2021Matrix} handles this via ``blocking'' --- replacing $A$ by $A^{\otimes q}$}
\end{itemize}

\column{0.48\textwidth}
{\small\textbf{In the formalization:}}
\begin{itemize}\itemsep0.1em
  \item {\small Our sorry \#5 takes $\mu$ injective and $\mu_k \neq 0$}
  \item {\small May need to strengthen to:\\
  $\forall k \neq \ell,\ \mu_k/\mu_\ell$ not a root of unity}
  \item {\small Or: work with blocked tensors}
  \item {\small \textcolor{amber}{Open design question}}
\end{itemize}
\end{columns}
\end{frame}

\begin{frame}[fragile]{Backup: Full Proof of \texttt{injective\_implies\_irreducibleCP}}
\vspace{-0.7em}
\begin{lstlisting}[basicstyle=\ttfamily\fontsize{4.5}{5.5}\selectfont\color{white}]
theorem injective_implies_irreducibleCP (A : MPSTensor d D) (hA : IsInjective A) :
    IsIrreducibleCP (transferMap A) := by
  intro P hProj hInv
  -- Step 1: derive (1-P)*A(*@$_i$@*)*P = 0 for each i
  have h_on_A := invariance_implies_complement_zero A P hProj hInv
  -- Step 2: extend to all matrices via span
  have h_all : (*@$\forall$@*) M : Matrix (Fin D) (Fin D) (*@$\mathbb{C}$@*), (1 - P) * M * P = 0 := by
    intro M
    have hM : M (*@$\in$@*) Submodule.span (*@$\mathbb{C}$@*) (Set.range A) := hA (*@$\blacktriangleright$@*) Submodule.mem_top
    induction hM using Submodule.span_induction with
    | mem x hx => obtain (*@$\langle$@*)i, rfl(*@$\rangle$@*) := hx; exact h_on_A i
    | zero => simp
    | add x y _ _ hx hy =>
      calc (1 - P) * (x + y) * P
          = (1 - P) * x * P + (1 - P) * y * P := by noncomm_ring
        _ = 0 + 0 := by rw [hx, hy]
        _ = 0 := add_zero 0
    | smul c x _ hx =>
      calc (1 - P) * (c (*@$\bullet$@*) x) * P
          = c (*@$\bullet$@*) ((1 - P) * x * P) := by rw [mul_smul_comm, smul_mul_assoc]
        _ = c (*@$\bullet$@*) 0 := by rw [hx]
        _ = 0 := smul_zero c
  -- Step 3: conclude P = 0 or P = 1
  exact proj_zero_or_one_of_sandwich P h_all
\end{lstlisting}
\end{frame}

theorem pow_tendsto_zero_of_spectralRadius_lt_one

    (a : A) (h : spectralRadius (*@$\mathbb{C}$@*) a < 1) :
    Filter.Tendsto (fun n => a ^ n) Filter.atTop (nhds 0) := by
  -- Step 1: Find r : NNReal with spectralRadius (*@$\mathbb{C}$@*) a < r < 1
  rw [ENNReal.lt_iff_exists_nnreal_btwn] at h
